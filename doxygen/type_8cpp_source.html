80 /** Utility routine to create a llvm DIArray type of the given number of
81  the given element type. */
82 static llvm::DIType
83 lCreateDIArray(llvm::DIType eltType, int count) {
84  int lowerBound = 0, upperBound = count-1;
85 
86  if (count == 0) {
87  // unsized array -> indicate with low > high
88  lowerBound = 1;
89  upperBound = 0;
90  }
91 
92  llvm::Value *sub = m->diBuilder->getOrCreateSubrange(lowerBound, upperBound);
93  std::vector<llvm::Value *> subs;
94  subs.push_back(sub);
95  llvm::DIArray subArray = m->diBuilder->getOrCreateArray(subs);
96 
97  uint64_t size = eltType.getSizeInBits() * count;
98  uint64_t align = eltType.getAlignInBits();
99 
100  return m->diBuilder->createArrayType(size, align, eltType, subArray);
101 }

574  }
575  else {
576  Assert(variability == Variability::SOA);
577  ArrayType at(GetAsUniformType(), variability.soaWidth);
578  return at.GetDIType(scope);
579  }

1508  // function is going to size an unsized dimension, make sure that
1509  // all of the sub-expression lists are the same length--i.e. issue
1510  // an error if we have something like
1511  // int x[][] = { { 1 }, { 1, 2, 3, 4 } };
1512  unsigned int nextSize = nextList->exprs.size();
1513  for (unsigned int i = 1; i < exprList->exprs.size(); ++i) {
1514  if (exprList->exprs[i] == NULL) {

1516  Assert(m->errorCount > 0);
1517  continue;
1518  }

1521  if (el == NULL || el->exprs.size() != nextSize) {

1524  "make it impossible to size unsized array dimensions.");

1526  }
1527  }
1528  }

1531  // array that we were able to size here.
1532  return new ArrayType(SizeUnsizedArrays(at->GetElementType(), nextList),
1533  at->GetElementCount());

2095

2098  // alignment and size, using that to figure out its offset w.r.t. the

2111  // element starts at an offset that's the correct alignment.

2929

2931  std::vector<llvm::Type *> llvmArgTypes;
2932  for (unsigned int i = 0; i < paramTypes.size(); ++i) {
2933  if (paramTypes[i] == NULL) {
2934  Assert(m->errorCount > 0);
2935  return NULL;
2936  }
2937  Assert(Type::Equal(paramTypes[i], AtomicType::Void) == false);
2938 
2939  llvm::Type *t = paramTypes[i]->LLVMType(ctx);
2940  if (t == NULL) {
2941  Assert(m->errorCount > 0);
2942  return NULL;
2943  }
2944  llvmArgTypes.push_back(t);
2945  }
2946 
2947  // And add the function mask, if asked for
2948  if (!(removeMask || isUnmasked))
2949  llvmArgTypes.push_back(LLVMTypes::MaskType);
2950 
2951  std::vector<llvm::Type *> callTypes;
2952  if (isTask) {
2953  // Tasks take three arguments: a pointer to a struct that holds the
2954  // actual task arguments, the thread index, and the total number of
2955  // threads the tasks system has running. (Task arguments are
2956  // marshalled in a struct so that it's easy to allocate space to
2957  // hold them until the task actually runs.)
2958  llvm::Type *st = llvm::StructType::get(*ctx, llvmArgTypes);
2959  callTypes.push_back(llvm::PointerType::getUnqual(st));
2960  callTypes.push_back(LLVMTypes::Int32Type); // threadIndex
