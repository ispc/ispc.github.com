146 void
147 DeclStmt::EmitCode(FunctionEmitContext *ctx) const {
148  if (!ctx->GetCurrentBasicBlock())
149  return;
150 
151  for (unsigned int i = 0; i < vars.size(); ++i) {
152  Symbol *sym = vars[i].sym;
153  AssertPos(pos, sym != NULL);
154  if (sym->type == NULL)
155  continue;
156  Expr *initExpr = vars[i].init;
157 
158  // Now that we're in the thick of emitting code, it's easy for us
159  // to find out the level of nesting of varying control flow we're
160  // in at this declaration. So we can finally set that
161  // Symbol::varyingCFDepth variable.
162  // @todo It's disgusting to be doing this here.
163  sym->varyingCFDepth = ctx->VaryingCFDepth();
164 
165  ctx->SetDebugPos(sym->pos);
166 
167  // If it's an array that was declared without a size but has an
168  // initializer list, then use the number of elements in the
169  // initializer list to finally set the array's size.
170  sym->type = ArrayType::SizeUnsizedArrays(sym->type, initExpr);
171  if (sym->type == NULL)
172  continue;
173 
174  if (lHasUnsizedArrays(sym->type)) {
175  Error(pos, "Illegal to declare an unsized array variable without "
176  "providing an initializer expression to set its size.");
177  continue;
178  }
179 
180  // References must have initializer expressions as well.
181  if (IsReferenceType(sym->type) == true) {
182  if (initExpr == NULL) {
183  Error(sym->pos, "Must provide initializer for reference-type "
184  "variable \"%s\".", sym->name.c_str());
185  continue;
186  }
187  if (IsReferenceType(initExpr->GetType()) == false) {
188  const Type *initLVType = initExpr->GetLValueType();
189  if (initLVType == NULL) {
190  Error(initExpr->pos, "Initializer for reference-type variable "
191  "\"%s\" must have an lvalue type.", sym->name.c_str());
192  continue;
193  }
194  if (initLVType->IsUniformType() == false) {
195  Error(initExpr->pos, "Initializer for reference-type variable "
196  "\"%s\" must have a uniform lvalue type.", sym->name.c_str());
197  continue;
198  }
199  }
200  }
201 
202  llvm::Type *llvmType = sym->type->LLVMType(g->ctx);
203  if (llvmType == NULL) {
204  AssertPos(pos, m->errorCount > 0);
205  return;
206  }
207 
208  if (sym->storageClass == SC_STATIC) {
209  // For static variables, we need a compile-time constant value
210  // for its initializer; if there's no initializer, we use a
211  // zero value.
212  llvm::Constant *cinit = NULL;
213  if (initExpr != NULL) {
214  if (PossiblyResolveFunctionOverloads(initExpr, sym->type) == false)
215  continue;
216  // FIXME: we only need this for function pointers; it was
217  // already done for atomic types and enums in
218  // DeclStmt::TypeCheck()...
219  if (dynamic_cast<ExprList *>(initExpr) == NULL) {
220  initExpr = TypeConvertExpr(initExpr, sym->type,
221  "initializer");
222  // FIXME: and this is only needed to re-establish
223  // constant-ness so that GetConstant below works for
224  // constant artithmetic expressions...
225  initExpr = ::Optimize(initExpr);
226  }
227 
228  cinit = initExpr->GetConstant(sym->type);
229  if (cinit == NULL)
230  Error(initExpr->pos, "Initializer for static variable "
231  "\"%s\" must be a constant.", sym->name.c_str());
232  }
233  if (cinit == NULL)
234  cinit = llvm::Constant::getNullValue(llvmType);
235 
236  // Allocate space for the static variable in global scope, so
237  // that it persists across function calls
238  sym->storagePtr =
239  new llvm::GlobalVariable(*m->module, llvmType,
240  sym->type->IsConstType(),
241  llvm::GlobalValue::InternalLinkage, cinit,
242  llvm::Twine("static.") +
243  llvm::Twine(sym->pos.first_line) +
244  llvm::Twine(".") + sym->name.c_str());
245  // Tell the FunctionEmitContext about the variable
246  ctx->EmitVariableDebugInfo(sym);
247  }
248  else {
249  // For non-static variables, allocate storage on the stack
250  sym->storagePtr = ctx->AllocaInst(llvmType, sym->name.c_str());
251 
252  // Tell the FunctionEmitContext about the variable; must do
253  // this before the initializer stuff.
254  ctx->EmitVariableDebugInfo(sym);
255 
256  // And then get it initialized...
257  sym->parentFunction = ctx->GetFunction();
258  InitSymbol(sym->storagePtr, sym->type, initExpr, ctx, sym->pos);
259  }
260  }
261 }

264 Stmt *
265 DeclStmt::Optimize() {
266  for (unsigned int i = 0; i < vars.size(); ++i) {
267  Expr *init = vars[i].init;
268  if (init != NULL && dynamic_cast<ExprList *>(init) == NULL) {
269  // If the variable is const-qualified, after we've optimized
270  // the initializer expression, see if we have a ConstExpr. If
271  // so, save it in Symbol::constValue where it can be used in
272  // optimizing later expressions that have this symbol in them.
273  // Note that there are cases where the expression may be
274  // constant but where we don't have a ConstExpr; an example is
275  // const arrays--the ConstExpr implementation just can't
276  // represent an array of values.
277  //
278  // All this is fine in terms of the code that's generated in
279  // the end (LLVM's constant folding stuff is good), but it
280  // means that the ispc compiler's ability to reason about what
281  // is definitely a compile-time constant for things like
282  // computing array sizes from non-trivial expressions is
283  // consequently limited.
284  Symbol *sym = vars[i].sym;
285  if (sym->type && sym->type->IsConstType() &&
286  Type::Equal(init->GetType(), sym->type))
287  sym->constValue = dynamic_cast<ConstExpr *>(init);
288  }
289  }
290  return this;
291 }

294 Stmt *
295 DeclStmt::TypeCheck() {
296  bool encounteredError = false;
297  for (unsigned int i = 0; i < vars.size(); ++i) {
298  if (vars[i].sym == NULL) {
299  encounteredError = true;
300  continue;
301  }
302 
303  if (vars[i].init == NULL)
304  continue;
305 
306  // get the right type for stuff like const float foo = 2; so that
307  // the int->float type conversion is in there and we don't return
308  // an int as the constValue later...
309  const Type *type = vars[i].sym->type;
310  if (CastType<AtomicType>(type) != NULL ||
311  CastType<EnumType>(type) != NULL) {
312  // If it's an expr list with an atomic type, we'll later issue
313  // an error. Need to leave vars[i].init as is in that case so
314  // it is in fact caught later, though.
315  if (dynamic_cast<ExprList *>(vars[i].init) == NULL) {
316  vars[i].init = TypeConvertExpr(vars[i].init, type,
317  "initializer");
318  if (vars[i].init == NULL)
319  encounteredError = true;
320  }
321  }
322  }
323  return encounteredError ? NULL : this;
324 }

398 void
399 IfStmt::EmitCode(FunctionEmitContext *ctx) const {
400  // First check all of the things that might happen due to errors
401  // earlier in compilation and bail out if needed so that we don't
402  // dereference NULL pointers in the below...
403  if (!ctx->GetCurrentBasicBlock())
404  return;
405  if (!test)
406  return;
407  const Type *testType = test->GetType();
408  if (!testType)
409  return;
410 
411  ctx->SetDebugPos(pos);
412  bool isUniform = testType->IsUniformType();
413 
414  llvm::Value *testValue = test->GetValue(ctx);
415  if (testValue == NULL)
416  return;
417 
418  if (isUniform) {
419  ctx->StartUniformIf();
420  if (doAllCheck)
421  Warning(test->pos, "Uniform condition supplied to \"cif\" statement.");
422 
423  // 'If' statements with uniform conditions are relatively
424  // straightforward. We evaluate the condition and then jump to
425  // either the 'then' or 'else' clause depending on its value.
426  llvm::BasicBlock *bthen = ctx->CreateBasicBlock("if_then");
427  llvm::BasicBlock *belse = ctx->CreateBasicBlock("if_else");
428  llvm::BasicBlock *bexit = ctx->CreateBasicBlock("if_exit");
429 
430  // Jump to the appropriate basic block based on the value of
431  // the 'if' test
432  ctx->BranchInst(bthen, belse, testValue);
433 
434  // Emit code for the 'true' case
435  ctx->SetCurrentBasicBlock(bthen);
436  lEmitIfStatements(ctx, trueStmts, "true");
437  if (ctx->GetCurrentBasicBlock())
438  ctx->BranchInst(bexit);
439 
440  // Emit code for the 'false' case
441  ctx->SetCurrentBasicBlock(belse);
442  lEmitIfStatements(ctx, falseStmts, "false");
443  if (ctx->GetCurrentBasicBlock())
444  ctx->BranchInst(bexit);
445 
446  // Set the active basic block to the newly-created exit block
447  // so that subsequent emitted code starts there.
448  ctx->SetCurrentBasicBlock(bexit);
449  ctx->EndIf();
450  }
451  /*
452  // Disabled for performance reasons. Change to an optional compile-time opt switch.
453  else if (lCanApplyBreakOptimization(trueStmts, falseStmts)) {
454  // If we have a simple break statement inside the 'if' and are
455  // under varying control flow, just update the execution mask
456  // directly and don't emit code for the statements. This leads to
457  // better code for this case--this is surprising and should be
458  // root-caused further, but for now this gives us performance
459  // benefit in this case.
460  ctx->SetInternalMaskAndNot(ctx->GetInternalMask(), testValue);
461  }
462  */
463  else
464  emitVaryingIf(ctx, testValue);
465 }

536 /** Emit code for an if test that checks the mask and the test values and
537  tries to be smart about jumping over code that doesn't need to be run.
538  */
539 void
540 IfStmt::emitVaryingIf(FunctionEmitContext *ctx, llvm::Value *ltest) const {
541  llvm::Value *oldMask = ctx->GetInternalMask();
542  if (doAllCheck) {
543  // We can't tell if the mask going into the if is all on at the
544  // compile time. Emit code to check for this and then either run
545  // the code for the 'all on' or the 'mixed' case depending on the
546  // mask's value at runtime.
547  llvm::BasicBlock *bAllOn = ctx->CreateBasicBlock("cif_mask_all");
548  llvm::BasicBlock *bMixedOn = ctx->CreateBasicBlock("cif_mask_mixed");
549  llvm::BasicBlock *bDone = ctx->CreateBasicBlock("cif_done");
550 
551  // Jump to either bAllOn or bMixedOn, depending on the mask's value
552  llvm::Value *maskAllQ = ctx->All(ctx->GetFullMask());
553  ctx->BranchInst(bAllOn, bMixedOn, maskAllQ);
554 
555  // Emit code for the 'mask all on' case
556  ctx->SetCurrentBasicBlock(bAllOn);
557  emitMaskAllOn(ctx, ltest, bDone);
558 
559  // And emit code for the mixed mask case
560  ctx->SetCurrentBasicBlock(bMixedOn);
561  emitMaskMixed(ctx, oldMask, ltest, bDone);
562 
563  // When done, set the current basic block to the block that the two
564  // paths above jump to when they're done.
565  ctx->SetCurrentBasicBlock(bDone);
566  }
567  else if (trueStmts != NULL || falseStmts != NULL) {
568  // If there is nothing that is potentially unsafe to run with all
569  // lanes off in the true and false statements and if the total
570  // complexity of those two is relatively simple, then we'll go
571  // ahead and emit straightline code that runs both sides, updating
572  // the mask accordingly. This is useful for efficiently compiling
573  // things like:
574  //
575  // if (foo) x = 0;
576  // else ++x;
577  //
578  // Where the overhead of checking if any of the program instances wants
579  // to run one side or the other is more than the actual computation.
580  // SafeToRunWithMaskAllOff() checks to make sure that we don't do this
581  // for potentially dangerous code like:
582  //
583  // if (index < count) array[index] = 0;
584  //
585  // where our use of blend for conditional assignments doesn't check
586  // for the 'all lanes' off case.
587  int trueFalseCost = (::EstimateCost(trueStmts) +
588  ::EstimateCost(falseStmts));
589  bool costIsAcceptable = (trueFalseCost <
590  PREDICATE_SAFE_IF_STATEMENT_COST);
591 
592  bool safeToRunWithAllLanesOff = (SafeToRunWithMaskAllOff(trueStmts) &&
593  SafeToRunWithMaskAllOff(falseStmts));
594 
595  Debug(pos, "If statement: true cost %d (safe %d), false cost %d (safe %d).",
596  ::EstimateCost(trueStmts), (int)SafeToRunWithMaskAllOff(trueStmts),
597  ::EstimateCost(falseStmts), (int)SafeToRunWithMaskAllOff(falseStmts));
598 
599  if (safeToRunWithAllLanesOff &&
600  (costIsAcceptable || g->opt.disableCoherentControlFlow)) {
601  ctx->StartVaryingIf(oldMask);
602  emitMaskedTrueAndFalse(ctx, oldMask, ltest);
603  AssertPos(pos, ctx->GetCurrentBasicBlock());
604  ctx->EndIf();
605  }
606  else {
607  llvm::BasicBlock *bDone = ctx->CreateBasicBlock("if_done");
608  emitMaskMixed(ctx, oldMask, ltest, bDone);
609  ctx->SetCurrentBasicBlock(bDone);
610  }
611  }
612 }

615 /** Emits code for 'if' tests under the case where we know that the program
616  mask is all on going into the 'if'.
617  */
618 void
619 IfStmt::emitMaskAllOn(FunctionEmitContext *ctx, llvm::Value *ltest,
620  llvm::BasicBlock *bDone) const {
621  // We start by explicitly storing "all on" into the mask mask. Note
622  // that this doesn't change its actual value, but doing so lets the
623  // compiler see what's going on so that subsequent optimizations for
624  // code emitted here can operate with the knowledge that the mask is
625  // definitely all on (until it modifies the mask itself).
626  AssertPos(pos, !g->opt.disableCoherentControlFlow);
627  if (!g->opt.disableMaskAllOnOptimizations)
628  ctx->SetInternalMask(LLVMMaskAllOn);
629  llvm::Value *oldFunctionMask = ctx->GetFunctionMask();
630  if (!g->opt.disableMaskAllOnOptimizations)
631  ctx->SetFunctionMask(LLVMMaskAllOn);
632 
633  // First, check the value of the test. If it's all on, then we jump to
634  // a basic block that will only have code for the true case.
635  llvm::BasicBlock *bTestAll = ctx->CreateBasicBlock("cif_test_all");
636  llvm::BasicBlock *bTestNoneCheck = ctx->CreateBasicBlock("cif_test_none_check");
637  llvm::Value *testAllQ = ctx->All(ltest);
638  ctx->BranchInst(bTestAll, bTestNoneCheck, testAllQ);
639 
640  // Emit code for the 'test is all true' case
641  ctx->SetCurrentBasicBlock(bTestAll);
642  ctx->StartVaryingIf(LLVMMaskAllOn);
643  lEmitIfStatements(ctx, trueStmts, "if: all on mask, expr all true");
644  ctx->EndIf();
645  if (ctx->GetCurrentBasicBlock() != NULL)
646  // bblock may legitimately be NULL since if there's a return stmt
647  // or break or continue we can actually jump and end emission since
648  // we know all of the lanes are following this path...
649  ctx->BranchInst(bDone);
650 
651  // The test isn't all true. Now emit code to determine if it's all
652  // false, or has mixed values.
653  ctx->SetCurrentBasicBlock(bTestNoneCheck);
654  llvm::BasicBlock *bTestNone = ctx->CreateBasicBlock("cif_test_none");
655  llvm::BasicBlock *bTestMixed = ctx->CreateBasicBlock("cif_test_mixed");
656  llvm::Value *testMixedQ = ctx->Any(ltest);
657  ctx->BranchInst(bTestMixed, bTestNone, testMixedQ);
658 
659  // Emit code for the 'test is all false' case
660  ctx->SetCurrentBasicBlock(bTestNone);
661  ctx->StartVaryingIf(LLVMMaskAllOn);
662  lEmitIfStatements(ctx, falseStmts, "if: all on mask, expr all false");
663  ctx->EndIf();

665  // bblock may be NULL since if there's a return stmt or break or
666  // continue we can actually jump or whatever and end emission...
667  ctx->BranchInst(bDone);
668 
669  // Finally emit code for the 'mixed true/false' case. We unavoidably
670  // need to run both the true and the false statements.
671  ctx->SetCurrentBasicBlock(bTestMixed);
672  ctx->StartVaryingIf(LLVMMaskAllOn);
673  emitMaskedTrueAndFalse(ctx, LLVMMaskAllOn, ltest);
674  // In this case, return/break/continue isn't allowed to jump and end
675  // emission.
676  AssertPos(pos, ctx->GetCurrentBasicBlock());
677  ctx->EndIf();
678  ctx->BranchInst(bDone);
679 
680  ctx->SetCurrentBasicBlock(bDone);
681  ctx->SetFunctionMask(oldFunctionMask);

692  llvm::BasicBlock *bNext = ctx->CreateBasicBlock("safe_if_after_true");

720  ctx->SetCurrentBasicBlock(bRunFalse);
721  if (falseStmts)
722  lEmitIfStatements(ctx, falseStmts, "if: expr mixed, false statements");
723  AssertPos(pos, ctx->GetCurrentBasicBlock());
724

760 /** Preorder callback function for checking for varying breaks or
761  continues. */
762 static bool
763 lVaryingBCPreFunc(ASTNode *node, void *d) {
764  VaryingBCCheckInfo *info = (VaryingBCCheckInfo *)d;
765 
766  // We found a break or continue statement; if we're under varying
767  // control flow, then bingo.
768  if ((dynamic_cast<BreakStmt *>(node) != NULL ||
769  dynamic_cast<ContinueStmt *>(node) != NULL) &&
770  info->varyingControlFlowDepth > 0) {
771  info->foundVaryingBreakOrContinue = true;
772  return false;
773  }
774 
775  // Update the count of the nesting depth of varying control flow if
776  // this is an if statement with a varying condition.
777  if (lIsVaryingFor(node))
778  ++info->varyingControlFlowDepth;
779 
780  if (dynamic_cast<ForStmt *>(node) != NULL ||
781  dynamic_cast<DoStmt *>(node) != NULL ||
782  dynamic_cast<ForeachStmt *>(node) != NULL)
783  // Don't recurse into these guys, since we don't care about varying
784  // breaks or continues within them...
785  return false;
786  else
787  return true;
788 }

823 void DoStmt::EmitCode(FunctionEmitContext *ctx) const {
824  // Check for things that could be NULL due to earlier errors during
825  // compilation.
826  if (!ctx->GetCurrentBasicBlock())
827  return;
828  if (!testExpr || !testExpr->GetType())
829  return;
830 
831  bool uniformTest = testExpr->GetType()->IsUniformType();
832  if (uniformTest && doCoherentCheck)
833  Warning(testExpr->pos, "Uniform condition supplied to \"cdo\" "
834  "statement.");
835 
836  llvm::BasicBlock *bloop = ctx->CreateBasicBlock("do_loop");
837  llvm::BasicBlock *bexit = ctx->CreateBasicBlock("do_exit");
838  llvm::BasicBlock *btest = ctx->CreateBasicBlock("do_test");
839 
840  ctx->StartLoop(bexit, btest, uniformTest);
841 
842  // Start by jumping into the loop body
843  ctx->BranchInst(bloop);
844 
845  // And now emit code for the loop body
846  ctx->SetCurrentBasicBlock(bloop);
847  ctx->SetBlockEntryMask(ctx->GetFullMask());
848  ctx->SetDebugPos(pos);
849  // FIXME: in the StmtList::EmitCode() method takes starts/stops a new
850  // scope around the statements in the list. So if the body is just a
851  // single statement (and thus not a statement list), we need a new
852  // scope, but we don't want two scopes in the StmtList case.
853  if (!dynamic_cast<StmtList *>(bodyStmts))
854  ctx->StartScope();
855 
856  ctx->AddInstrumentationPoint("do loop body");
857  if (doCoherentCheck && !uniformTest) {
858  // Check to see if the mask is all on
859  llvm::BasicBlock *bAllOn = ctx->CreateBasicBlock("do_all_on");
860  llvm::BasicBlock *bMixed = ctx->CreateBasicBlock("do_mixed");
861  ctx->BranchIfMaskAll(bAllOn, bMixed);
862 
863  // If so, emit code for the 'mask all on' case. In particular,
864  // explicitly set the mask to 'all on' (see rationale in
865  // IfStmt::emitCoherentTests()), and then emit the code for the
866  // loop body.
867  ctx->SetCurrentBasicBlock(bAllOn);
868  if (!g->opt.disableMaskAllOnOptimizations)
869  ctx->SetInternalMask(LLVMMaskAllOn);
870  llvm::Value *oldFunctionMask = ctx->GetFunctionMask();
871  if (!g->opt.disableMaskAllOnOptimizations)
872  ctx->SetFunctionMask(LLVMMaskAllOn);
873  if (bodyStmts)
874  bodyStmts->EmitCode(ctx);
875  AssertPos(pos, ctx->GetCurrentBasicBlock());
876  ctx->SetFunctionMask(oldFunctionMask);
877  ctx->BranchInst(btest);
878 
879  // The mask is mixed. Just emit the code for the loop body.
880  ctx->SetCurrentBasicBlock(bMixed);
881  if (bodyStmts)
882  bodyStmts->EmitCode(ctx);
883  AssertPos(pos, ctx->GetCurrentBasicBlock());
884  ctx->BranchInst(btest);
885  }
886  else {
887  // Otherwise just emit the code for the loop body. The current
888  // mask is good.
889  if (bodyStmts)
890  bodyStmts->EmitCode(ctx);
891  if (ctx->GetCurrentBasicBlock())
892  ctx->BranchInst(btest);
893  }
894  // End the scope we started above, if needed.
895  if (!dynamic_cast<StmtList *>(bodyStmts))
896  ctx->EndScope();
897 
898  // Now emit code for the loop test.
899  ctx->SetCurrentBasicBlock(btest);
900  // First, emit code to restore the mask value for any lanes that
901  // executed a 'continue' during the current loop before we go and emit
902  // the code for the test. This is only necessary for varying loops;
903  // 'uniform' loops just jump when they hit a continue statement and
904  // don't mess with the mask.
905  if (!uniformTest)
906  ctx->RestoreContinuedLanes();
907  llvm::Value *testValue = testExpr->GetValue(ctx);
908  if (!testValue)
909  return;
910 
911  if (uniformTest)
912  // For the uniform case, just jump to the top of the loop or the
913  // exit basic block depending on the value of the test.
914  ctx->BranchInst(bloop, bexit, testValue);
915  else {
916  // For the varying case, update the mask based on the value of the
917  // test. If any program instances still want to be running, jump
918  // to the top of the loop. Otherwise, jump out.
919  llvm::Value *mask = ctx->GetInternalMask();
920  ctx->SetInternalMaskAnd(mask, testValue);
921  ctx->BranchIfMaskAny(bloop, bexit);
922  }
923 
924  // ...and we're done. Set things up for subsequent code to be emitted
925  // in the right basic block.
926  ctx->SetCurrentBasicBlock(bexit);
927  ctx->EndLoop();
928 }

932 DoStmt::TypeCheck() {
933  const Type *testType;
934  if (testExpr != NULL && (testType = testExpr->GetType()) != NULL) {
935  // Should the test condition for the loop be uniform or varying?
936  // It can be uniform only if three conditions are met:
937  //
938  // - First and foremost, the type of the test condition must be
939  // uniform.

941  // - Second, the user must not have set the dis-optimization option

948  // loop includes masking stuff, so that we can track which lanes
949  // actually want to be running, accounting for breaks/continues.
950  //
951  bool uniformTest = (testType->IsUniformType() &&
952  !g->opt.disableUniformControlFlow &&
953  !lHasVaryingBreakOrContinue(bodyStmts));
954  testExpr = TypeConvertExpr(testExpr, uniformTest ? AtomicType::UniformBool :
955  AtomicType::VaryingBool,
956  "\"do\" statement");
957  }

997 void
998 ForStmt::EmitCode(FunctionEmitContext *ctx) const {
999  if (!ctx->GetCurrentBasicBlock())
1000  return;
1001 
1002  llvm::BasicBlock *btest = ctx->CreateBasicBlock("for_test");
1003  llvm::BasicBlock *bstep = ctx->CreateBasicBlock("for_step");
1004  llvm::BasicBlock *bloop = ctx->CreateBasicBlock("for_loop");
1005  llvm::BasicBlock *bexit = ctx->CreateBasicBlock("for_exit");
1006 
1007  bool uniformTest = test ? test->GetType()->IsUniformType() :
1008  (!g->opt.disableUniformControlFlow &&
1009  !lHasVaryingBreakOrContinue(stmts));
1010 
1011  ctx->StartLoop(bexit, bstep, uniformTest);
1012  ctx->SetDebugPos(pos);
1013 
1014  // If we have an initiailizer statement, start by emitting the code for
1015  // it and then jump into the loop test code. (Also start a new scope
1016  // since the initiailizer may be a declaration statement).
1017  if (init) {
1018  AssertPos(pos, dynamic_cast<StmtList *>(init) == NULL);
1019  ctx->StartScope();
1020  init->EmitCode(ctx);
1021  }
1022  ctx->BranchInst(btest);
1023 
1024  // Emit code to get the value of the loop test. If no test expression
1025  // was provided, just go with a true value.
1026  ctx->SetCurrentBasicBlock(btest);
1027  llvm::Value *ltest = NULL;
1028  if (test) {
1029  ltest = test->GetValue(ctx);
1030  if (!ltest) {
1031  ctx->EndScope();
1032  ctx->EndLoop();
1033  return;
1034  }
1035  }
1036  else
1037  ltest = uniformTest ? LLVMTrue : LLVMBoolVector(true);
1038 
1039  // Now use the test's value. For a uniform loop, we can either jump to
1040  // the loop body or the loop exit, based on whether it's true or false.
1041  // For a non-uniform loop, we update the mask and jump into the loop if
1042  // any of the mask values are true.
1043  if (uniformTest) {
1044  if (doCoherentCheck)
1045  Warning(test->pos, "Uniform condition supplied to cfor/cwhile "
1046  "statement.");
1047  AssertPos(pos, ltest->getType() == LLVMTypes::BoolType);
1048  ctx->BranchInst(bloop, bexit, ltest);
1049  }
1050  else {
1051  llvm::Value *mask = ctx->GetInternalMask();
1052  ctx->SetInternalMaskAnd(mask, ltest);
1053  ctx->BranchIfMaskAny(bloop, bexit);
1054  }
1055 
1056  // On to emitting the code for the loop body.
1057  ctx->SetCurrentBasicBlock(bloop);
1058  ctx->SetBlockEntryMask(ctx->GetFullMask());
1059  ctx->AddInstrumentationPoint("for loop body");
1060  if (!dynamic_cast<StmtList *>(stmts))
1061  ctx->StartScope();
1062 
1063  if (doCoherentCheck && !uniformTest) {
1064  // For 'varying' loops with the coherence check, we start by
1065  // checking to see if the mask is all on, after it has been updated
1066  // based on the value of the test.
1067  llvm::BasicBlock *bAllOn = ctx->CreateBasicBlock("for_all_on");
1068  llvm::BasicBlock *bMixed = ctx->CreateBasicBlock("for_mixed");
1069  ctx->BranchIfMaskAll(bAllOn, bMixed);
1070 
1071  // Emit code for the mask being all on. Explicitly set the mask to
1072  // be on so that the optimizer can see that it's on (i.e. now that
1073  // the runtime test has passed, make this fact clear for code
1074  // generation at compile time here.)
1075  ctx->SetCurrentBasicBlock(bAllOn);
1076  if (!g->opt.disableMaskAllOnOptimizations)
1077  ctx->SetInternalMask(LLVMMaskAllOn);
1078  llvm::Value *oldFunctionMask = ctx->GetFunctionMask();
1079  if (!g->opt.disableMaskAllOnOptimizations)
1080  ctx->SetFunctionMask(LLVMMaskAllOn);
1081  if (stmts)
1082  stmts->EmitCode(ctx);
1083  AssertPos(pos, ctx->GetCurrentBasicBlock());
1084  ctx->SetFunctionMask(oldFunctionMask);
1085  ctx->BranchInst(bstep);
1086 
1087  // Emit code for the mask being mixed. We should never run the
1088  // loop with the mask all off, based on the BranchIfMaskAny call
1089  // above.
1090  ctx->SetCurrentBasicBlock(bMixed);
1091  if (stmts)
1092  stmts->EmitCode(ctx);
1093  ctx->BranchInst(bstep);
1094  }
1095  else {
1096  // For both uniform loops and varying loops without the coherence
1097  // check, we know that at least one program instance wants to be
1098  // running the loop, so just emit code for the loop body and jump
1099  // to the loop step code.
1100  if (stmts)
1101  stmts->EmitCode(ctx);
1102  if (ctx->GetCurrentBasicBlock())
1103  ctx->BranchInst(bstep);
1104  }
1105  if (!dynamic_cast<StmtList *>(stmts))
1106  ctx->EndScope();
1107 
1108  // Emit code for the loop step. First, restore the lane mask of any
1109  // program instances that executed a 'continue' during the previous
1110  // iteration. Then emit code for the loop step and then jump to the
1111  // test code.
1112  ctx->SetCurrentBasicBlock(bstep);
1113  ctx->RestoreContinuedLanes();
1114  if (step)
1115  step->EmitCode(ctx);
1116  ctx->BranchInst(btest);
1117 
1118  // Set the current emission basic block to the loop exit basic block
1119  ctx->SetCurrentBasicBlock(bexit);
1120  if (init)
1121  ctx->EndScope();
1122  ctx->EndLoop();
1123 }

1268 /* Given a uniform counter value in the memory location pointed to by
1269  uniformCounterPtr, compute the corresponding set of varying counter
1270  values for use within the loop body.
1271  */
1272 static llvm::Value *
1273 lUpdateVaryingCounter(int dim, int nDims, FunctionEmitContext *ctx,
1274  llvm::Value *uniformCounterPtr,
1275  llvm::Value *varyingCounterPtr,
1276  const std::vector<int> &spans) {
1277  // Smear the uniform counter value out to be varying
1278  llvm::Value *counter = ctx->LoadInst(uniformCounterPtr);
1279  llvm::Value *smearCounter = ctx->BroadcastValue(
1280  counter, LLVMTypes::Int32VectorType, "smear_counter");
1281 
1282  // Figure out the offsets; this is a little bit tricky. As an example,
1283  // consider a 2D tiled foreach loop, where we're running 8-wide and
1284  // where the inner dimension has a stride of 4 and the outer dimension
1285  // has a stride of 2. For the inner dimension, we want the offsets
1286  // (0,1,2,3,0,1,2,3), and for the outer dimension we want
1287  // (0,0,0,0,1,1,1,1).
1288  int32_t delta[ISPC_MAX_NVEC];
1289  for (int i = 0; i < g->target->getVectorWidth(); ++i) {
1290  int d = i;
1291  // First, account for the effect of any dimensions at deeper
1292  // nesting levels than the current one.
1293  int prevDimSpanCount = 1;
1294  for (int j = dim; j < nDims-1; ++j)
1295  prevDimSpanCount *= spans[j+1];
1296  d /= prevDimSpanCount;
1297 
1298  // And now with what's left, figure out our own offset
1299  delta[i] = d % spans[dim];
1300  }
1301 
1302  // Add the deltas to compute the varying counter values; store the
1303  // result to memory and then return it directly as well.
1304  llvm::Value *varyingCounter =
1305  ctx->BinaryOperator(llvm::Instruction::Add, smearCounter,
1306  LLVMInt32Vector(delta), "iter_val");
1307  ctx->StoreInst(varyingCounter, varyingCounterPtr);
1308  return varyingCounter;
1309 }

1312 /** Returns the integer log2 of the given integer. */
1313 static int
1314 lLog2(int i) {
1315  int ret = 0;
1316  while (i != 0) {
1317  ++ret;
1318  i >>= 1;
1319  }
1320  return ret-1;
1321 }

1324 /* Figure out how many elements to process in each dimension for each time
1325  through a foreach loop. The untiled case is easy; all of the outer
1326  dimensions up until the innermost one have a span of 1, and the
1327  innermost one takes the entire vector width. For the tiled case, we
1328  give wider spans to the innermost dimensions while also trying to
1329  generate relatively square domains.
1330 
1331  This code works recursively from outer dimensions to inner dimensions.
1332  */
1333 static void
1334 lGetSpans(int dimsLeft, int nDims, int itemsLeft, bool isTiled, int *a) {
1335  if (dimsLeft == 0) {
1336  // Nothing left to do but give all of the remaining work to the
1337  // innermost domain.
1338  *a = itemsLeft;
1339  return;
1340  }
1341 
1342  if (isTiled == false || (dimsLeft >= lLog2(itemsLeft)))
1343  // If we're not tiled, or if there are enough dimensions left that
1344  // giving this one any more than a span of one would mean that a
1345  // later dimension would have to have a span of one, give this one
1346  // a span of one to save the available items for later.
1347  *a = 1;
1348  else if (itemsLeft >= 16 && (dimsLeft == 1))
1349  // Special case to have 4x4 domains for the 2D case when running
1350  // 16-wide.
1351  *a = 4;
1352  else
1353  // Otherwise give this dimension a span of two.
1354  *a = 2;
1355 
1356  lGetSpans(dimsLeft-1, nDims, itemsLeft / *a, isTiled, a+1);
1357 }

1360 /* Emit code for a foreach statement. We effectively emit code to run the
1361  set of n-dimensional nested loops corresponding to the dimensionality of
1362  the foreach statement along with the extra logic to deal with mismatches
1363  between the vector width we're compiling to and the number of elements
1364  to process.
1365  */
1366 void
1367 ForeachStmt::EmitCode(FunctionEmitContext *ctx) const {
1368  if (ctx->GetCurrentBasicBlock() == NULL || stmts == NULL)
1369  return;
1370 
1371  llvm::BasicBlock *bbFullBody = ctx->CreateBasicBlock("foreach_full_body");
1372  llvm::BasicBlock *bbMaskedBody = ctx->CreateBasicBlock("foreach_masked_body");
1373  llvm::BasicBlock *bbExit = ctx->CreateBasicBlock("foreach_exit");
1374 
1375  llvm::Value *oldMask = ctx->GetInternalMask();
1376  llvm::Value *oldFunctionMask = ctx->GetFunctionMask();
1377 
1378  ctx->SetDebugPos(pos);
1379  ctx->StartScope();
1380 
1381  ctx->SetInternalMask(LLVMMaskAllOn);
1382  ctx->SetFunctionMask(LLVMMaskAllOn);
1383 
1384  // This should be caught during typechecking
1385  AssertPos(pos, startExprs.size() == dimVariables.size() &&
1386  endExprs.size() == dimVariables.size());
1387  int nDims = (int)dimVariables.size();
1388 
1389  ///////////////////////////////////////////////////////////////////////
1390  // Setup: compute the number of items we have to work on in each
1391  // dimension and a number of derived values.
1392  std::vector<llvm::BasicBlock *> bbReset, bbStep, bbTest;
1393  std::vector<llvm::Value *> startVals, endVals, uniformCounterPtrs;
1394  std::vector<llvm::Value *> nExtras, alignedEnd, extrasMaskPtrs;
1395 
1396  std::vector<int> span(nDims, 0);
1397  lGetSpans(nDims-1, nDims, g->target->getVectorWidth(), isTiled, &span[0]);
1398 
1399  for (int i = 0; i < nDims; ++i) {
1400  // Basic blocks that we'll fill in later with the looping logic for
1401  // this dimension.
1402  bbReset.push_back(ctx->CreateBasicBlock("foreach_reset"));
1403  if (i < nDims-1)
1404  // stepping for the innermost dimension is handled specially
1405  bbStep.push_back(ctx->CreateBasicBlock("foreach_step"));
1406  bbTest.push_back(ctx->CreateBasicBlock("foreach_test"));
1407 
1408  // Start and end value for this loop dimension
1409  llvm::Value *sv = startExprs[i]->GetValue(ctx);
1410  llvm::Value *ev = endExprs[i]->GetValue(ctx);
1411  if (sv == NULL || ev == NULL)
1412  return;
1413  startVals.push_back(sv);
1414  endVals.push_back(ev);
1415 
1416  // nItems = endVal - startVal
1417  llvm::Value *nItems =
1418  ctx->BinaryOperator(llvm::Instruction::Sub, ev, sv, "nitems");
1419 
1420  // nExtras = nItems % (span for this dimension)
1421  // This gives us the number of extra elements we need to deal with
1422  // at the end of the loop for this dimension that don't fit cleanly
1423  // into a vector width.
1424  nExtras.push_back(ctx->BinaryOperator(llvm::Instruction::SRem, nItems,
1425  LLVMInt32(span[i]), "nextras"));
1426 
1427  // alignedEnd = endVal - nExtras
1428  alignedEnd.push_back(ctx->BinaryOperator(llvm::Instruction::Sub, ev,
1429  nExtras[i], "aligned_end"));
1430 
1431  ///////////////////////////////////////////////////////////////////////
1432  // Each dimension has a loop counter that is a uniform value that
1433  // goes from startVal to endVal, in steps of the span for this
1434  // dimension. Its value is only used internally here for looping
1435  // logic and isn't directly available in the user's program code.
1436  uniformCounterPtrs.push_back(ctx->AllocaInst(LLVMTypes::Int32Type,
1437  "counter"));
1438  ctx->StoreInst(startVals[i], uniformCounterPtrs[i]);
1439 
1440  // There is also a varying variable that holds the set of index
1441  // values for each dimension in the current loop iteration; this is
1442  // the value that is program-visible.
1443  dimVariables[i]->storagePtr =
1444  ctx->AllocaInst(LLVMTypes::Int32VectorType,
1445  dimVariables[i]->name.c_str());
1446  dimVariables[i]->parentFunction = ctx->GetFunction();
1447  ctx->EmitVariableDebugInfo(dimVariables[i]);
1448 
1449  // Each dimension also maintains a mask that represents which of
1450  // the varying elements in the current iteration should be
1451  // processed. (i.e. this is used to disable the lanes that have
1452  // out-of-bounds offsets.)
1453  extrasMaskPtrs.push_back(ctx->AllocaInst(LLVMTypes::MaskType, "extras mask"));
1454  ctx->StoreInst(LLVMMaskAllOn, extrasMaskPtrs[i]);
1455  }
1456 
1457  ctx->StartForeach(FunctionEmitContext::FOREACH_REGULAR);
1458 
1459  // On to the outermost loop's test
1460  ctx->BranchInst(bbTest[0]);
1461 
1462  ///////////////////////////////////////////////////////////////////////////
1463  // foreach_reset: this code runs when we need to reset the counter for
1464  // a given dimension in preparation for running through its loop again,
1465  // after the enclosing level advances its counter.
1466  for (int i = 0; i < nDims; ++i) {
1467  ctx->SetCurrentBasicBlock(bbReset[i]);
1468  if (i == 0)
1469  ctx->BranchInst(bbExit);
1470  else {
1471  ctx->StoreInst(LLVMMaskAllOn, extrasMaskPtrs[i]);
1472  ctx->StoreInst(startVals[i], uniformCounterPtrs[i]);
1473  ctx->BranchInst(bbStep[i-1]);
1474  }
1475  }
1476 
1477  ///////////////////////////////////////////////////////////////////////////
1478  // foreach_step: increment the uniform counter by the vector width.
1479  // Note that we don't increment the varying counter here as well but
1480  // just generate its value when we need it in the loop body. Don't do
1481  // this for the innermost dimension, which has a more complex stepping
1482  // structure..
1483  for (int i = 0; i < nDims-1; ++i) {
1484  ctx->SetCurrentBasicBlock(bbStep[i]);
1485  llvm::Value *counter = ctx->LoadInst(uniformCounterPtrs[i]);
1486  llvm::Value *newCounter =
1487  ctx->BinaryOperator(llvm::Instruction::Add, counter,
1488  LLVMInt32(span[i]), "new_counter");
1489  ctx->StoreInst(newCounter, uniformCounterPtrs[i]);
1490  ctx->BranchInst(bbTest[i]);
1491  }
1492 
1493  ///////////////////////////////////////////////////////////////////////////
1494  // foreach_test (for all dimensions other than the innermost...)
1495  std::vector<llvm::Value *> inExtras;
1496  for (int i = 0; i < nDims-1; ++i) {
1497  ctx->SetCurrentBasicBlock(bbTest[i]);
1498 
1499  llvm::Value *haveExtras =
1500  ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SGT,
1501  endVals[i], alignedEnd[i], "have_extras");
1502 
1503  llvm::Value *counter = ctx->LoadInst(uniformCounterPtrs[i], "counter");
1504  llvm::Value *atAlignedEnd =
1505  ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_EQ,
1506  counter, alignedEnd[i], "at_aligned_end");
1507  llvm::Value *inEx =
1508  ctx->BinaryOperator(llvm::Instruction::And, haveExtras,
1509  atAlignedEnd, "in_extras");
1510 
1511  if (i == 0)
1512  inExtras.push_back(inEx);
1513  else
1514  inExtras.push_back(ctx->BinaryOperator(llvm::Instruction::Or, inEx,
1515  inExtras[i-1], "in_extras_all"));
1516 
1517  llvm::Value *varyingCounter =
1518  lUpdateVaryingCounter(i, nDims, ctx, uniformCounterPtrs[i],
1519  dimVariables[i]->storagePtr, span);
1520 
1521  llvm::Value *smearEnd = ctx->BroadcastValue(
1522  endVals[i], LLVMTypes::Int32VectorType, "smear_end");
1523 
1524  // Do a vector compare of its value to the end value to generate a
1525  // mask for this last bit of work.
1526  llvm::Value *emask =
1527  ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SLT,
1528  varyingCounter, smearEnd);
1529  emask = ctx->I1VecToBoolVec(emask);
1530 
1531  if (i == 0)
1532  ctx->StoreInst(emask, extrasMaskPtrs[i]);
1533  else {
1534  llvm::Value *oldMask = ctx->LoadInst(extrasMaskPtrs[i-1]);
1535  llvm::Value *newMask =
1536  ctx->BinaryOperator(llvm::Instruction::And, oldMask, emask,
1537  "extras_mask");
1538  ctx->StoreInst(newMask, extrasMaskPtrs[i]);
1539  }
1540 
1541  llvm::Value *notAtEnd =
1542  ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SLT,
1543  counter, endVals[i]);
1544  ctx->BranchInst(bbTest[i+1], bbReset[i], notAtEnd);
1545  }
1546 
1547  ///////////////////////////////////////////////////////////////////////////
1548  // foreach_test (for innermost dimension)
1549  //
1550  // All of the outer dimensions are handled generically--basically as a
1551  // for() loop from the start value to the end value, where at each loop
1552  // test, we compute the mask of active elements for the current
1553  // dimension and then update an overall mask that is the AND
1554  // combination of all of the outer ones.
1555  //
1556  // The innermost loop is handled specially, for performance purposes.
1557  // When starting the innermost dimension, we start by checking once
1558  // whether any of the outer dimensions has set the mask to be
1559  // partially-active or not. We follow different code paths for these
1560  // two cases, taking advantage of the knowledge that the mask is all
1561  // on, when this is the case.
1562  //
1563  // In each of these code paths, we start with a loop from the starting
1564  // value to the aligned end value for the innermost dimension; we can
1565  // guarantee that the innermost loop will have an "all on" mask (as far
1566  // as its dimension is concerned) for the duration of this loop. Doing
1567  // so allows us to emit code that assumes the mask is all on (for the
1568  // case where none of the outer dimensions has set the mask to be
1569  // partially on), or allows us to emit code that just uses the mask
1570  // from the outer dimensions directly (for the case where they have).
1571  //
1572  // After this loop, we just need to deal with one vector's worth of
1573  // "ragged extra bits", where the mask used includes the effect of the
1574  // mask for the innermost dimension.
1575  //
1576  // We start out this process by emitting the check that determines
1577  // whether any of the enclosing dimensions is partially active
1578  // (i.e. processing extra elements that don't exactly fit into a
1579  // vector).
1580  llvm::BasicBlock *bbOuterInExtras =
1581  ctx->CreateBasicBlock("outer_in_extras");
1582  llvm::BasicBlock *bbOuterNotInExtras =
1583  ctx->CreateBasicBlock("outer_not_in_extras");
1584 
1585  ctx->SetCurrentBasicBlock(bbTest[nDims-1]);
1586  if (inExtras.size())
1587  ctx->BranchInst(bbOuterInExtras, bbOuterNotInExtras,
1588  inExtras.back());
1589  else
1590  // for a 1D iteration domain, we certainly don't have any enclosing
1591  // dimensions that are processing extra elements.
1592  ctx->BranchInst(bbOuterNotInExtras);
1593 
1594  ///////////////////////////////////////////////////////////////////////////
1595  // One or more outer dimensions in extras, so we need to mask for the loop
1596  // body regardless. We break this into two cases, roughly:
1597  // for (counter = start; counter < alignedEnd; counter += step) {
1598  // // mask is all on for inner, so set mask to outer mask
1599  // // run loop body with mask
1600  // }
1601  // // counter == alignedEnd
1602  // if (counter < end) {
1603  // // set mask to outermask & (counter+programCounter < end)
1604  // // run loop body with mask
1605  // }
1606  llvm::BasicBlock *bbAllInnerPartialOuter =
1607  ctx->CreateBasicBlock("all_inner_partial_outer");
1608  llvm::BasicBlock *bbPartial =
1609  ctx->CreateBasicBlock("both_partial");
1610  ctx->SetCurrentBasicBlock(bbOuterInExtras); {
1611  // Update the varying counter value here, since all subsequent
1612  // blocks along this path need it.
1613  lUpdateVaryingCounter(nDims-1, nDims, ctx, uniformCounterPtrs[nDims-1],
1614  dimVariables[nDims-1]->storagePtr, span);
1615 
1616  // here we just check to see if counter < alignedEnd
1617  llvm::Value *counter = ctx->LoadInst(uniformCounterPtrs[nDims-1], "counter");
1618  llvm::Value *beforeAlignedEnd =
1619  ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SLT,
1620  counter, alignedEnd[nDims-1], "before_aligned_end");
1621  ctx->BranchInst(bbAllInnerPartialOuter, bbPartial, beforeAlignedEnd);
1622  }
1623 
1624  // Below we have a basic block that runs the loop body code for the
1625  // case where the mask is partially but not fully on. This same block
1626  // runs in multiple cases: both for handling any ragged extra data for
1627  // the innermost dimension but also when outer dimensions have set the
1628  // mask to be partially on.
1629  //
1630  // The value stored in stepIndexAfterMaskedBodyPtr is used after each
1631  // execution of the body code to determine whether the innermost index
1632  // value should be incremented by the step (we're running the "for"
1633  // loop of full vectors at the innermost dimension, with outer
1634  // dimensions having set the mask to be partially on), or whether we're
1635  // running once for the ragged extra bits at the end of the innermost
1636  // dimension, in which case we're done with the innermost dimension and
1637  // should step the loop counter for the next enclosing dimension
1638  // instead.
1639  llvm::Value *stepIndexAfterMaskedBodyPtr =
1640  ctx->AllocaInst(LLVMTypes::BoolType, "step_index");
1641 
1642  ///////////////////////////////////////////////////////////////////////////
1643  // We're in the inner loop part where the only masking is due to outer
1644  // dimensions but the innermost dimension fits fully into a vector's
1645  // width. Set the mask and jump to the masked loop body.
1646  ctx->SetCurrentBasicBlock(bbAllInnerPartialOuter); {
1647  llvm::Value *mask;
1648  if (nDims == 1)
1649  // 1D loop; we shouldn't ever get here anyway
1650  mask = LLVMMaskAllOff;
1651  else
1652  mask = ctx->LoadInst(extrasMaskPtrs[nDims-2]);
1653 
1654  ctx->SetInternalMask(mask);
1655 
1656  ctx->StoreInst(LLVMTrue, stepIndexAfterMaskedBodyPtr);
1657  ctx->BranchInst(bbMaskedBody);
1658  }
1659 
1660  ///////////////////////////////////////////////////////////////////////////
1661  // We need to include the effect of the innermost dimension in the mask
1662  // for the final bits here
1663  ctx->SetCurrentBasicBlock(bbPartial); {
1664  llvm::Value *varyingCounter =
1665  ctx->LoadInst(dimVariables[nDims-1]->storagePtr);
1666  llvm::Value *smearEnd = ctx->BroadcastValue(
1667  endVals[nDims-1], LLVMTypes::Int32VectorType, "smear_end");
1668 
1669  llvm::Value *emask =
1670  ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SLT,
1671  varyingCounter, smearEnd);
1672  emask = ctx->I1VecToBoolVec(emask);
1673 
1674  if (nDims == 1) {
1675  ctx->SetInternalMask(emask);
1676  }
1677  else {
1678  llvm::Value *oldMask = ctx->LoadInst(extrasMaskPtrs[nDims-2]);
1679  llvm::Value *newMask =
1680  ctx->BinaryOperator(llvm::Instruction::And, oldMask, emask,
1681  "extras_mask");
1682  ctx->SetInternalMask(newMask);
1683  }
1684 
1685  ctx->StoreInst(LLVMFalse, stepIndexAfterMaskedBodyPtr);
1686  ctx->BranchInst(bbMaskedBody);
1687  }
1688 
1689  ///////////////////////////////////////////////////////////////////////////
1690  // None of the outer dimensions is processing extras; along the lines
1691  // of above, we can express this as:
1692  // for (counter = start; counter < alignedEnd; counter += step) {
1693  // // mask is all on
1694  // // run loop body with mask all on
1695  // }
1696  // // counter == alignedEnd
1697  // if (counter < end) {
1698  // // set mask to (counter+programCounter < end)
1699  // // run loop body with mask
1700  // }
1701  llvm::BasicBlock *bbPartialInnerAllOuter =
1702  ctx->CreateBasicBlock("partial_inner_all_outer");
1703  ctx->SetCurrentBasicBlock(bbOuterNotInExtras); {
1704  llvm::Value *counter = ctx->LoadInst(uniformCounterPtrs[nDims-1], "counter");
1705  llvm::Value *beforeAlignedEnd =
1706  ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SLT,
1707  counter, alignedEnd[nDims-1], "before_aligned_end");
1708  ctx->BranchInst(bbFullBody, bbPartialInnerAllOuter,
1709  beforeAlignedEnd);
1710  }
1711 
1712  ///////////////////////////////////////////////////////////////////////////
1713  // full_body: do a full vector's worth of work. We know that all
1714  // lanes will be running here, so we explicitly set the mask to be 'all
1715  // on'. This ends up being relatively straightforward: just update the
1716  // value of the varying loop counter and have the statements in the
1717  // loop body emit their code.
1718  llvm::BasicBlock *bbFullBodyContinue =
1719  ctx->CreateBasicBlock("foreach_full_continue");
1720  ctx->SetCurrentBasicBlock(bbFullBody); {
1721  ctx->SetInternalMask(LLVMMaskAllOn);
1722  ctx->SetBlockEntryMask(LLVMMaskAllOn);
1723  lUpdateVaryingCounter(nDims-1, nDims, ctx, uniformCounterPtrs[nDims-1],
1724  dimVariables[nDims-1]->storagePtr, span);
1725  ctx->SetContinueTarget(bbFullBodyContinue);
1726  ctx->AddInstrumentationPoint("foreach loop body (all on)");
1727  stmts->EmitCode(ctx);
1728  AssertPos(pos, ctx->GetCurrentBasicBlock() != NULL);
1729  ctx->BranchInst(bbFullBodyContinue);
1730  }
1731  ctx->SetCurrentBasicBlock(bbFullBodyContinue); {
1732  ctx->RestoreContinuedLanes();
1733  llvm::Value *counter = ctx->LoadInst(uniformCounterPtrs[nDims-1]);
1734  llvm::Value *newCounter =
1735  ctx->BinaryOperator(llvm::Instruction::Add, counter,
1736  LLVMInt32(span[nDims-1]), "new_counter");
1737  ctx->StoreInst(newCounter, uniformCounterPtrs[nDims-1]);
1738  ctx->BranchInst(bbOuterNotInExtras);
1739  }
1740 
1741  ///////////////////////////////////////////////////////////////////////////
1742  // We're done running blocks with the mask all on; see if the counter is
1743  // less than the end value, in which case we need to run the body one
1744  // more time to get the extra bits.
1745  llvm::BasicBlock *bbSetInnerMask =
1746  ctx->CreateBasicBlock("partial_inner_only");
1747  ctx->SetCurrentBasicBlock(bbPartialInnerAllOuter); {
1748  llvm::Value *counter = ctx->LoadInst(uniformCounterPtrs[nDims-1], "counter");
1749  llvm::Value *beforeFullEnd =
1750  ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SLT,
1751  counter, endVals[nDims-1], "before_full_end");
1752  ctx->BranchInst(bbSetInnerMask, bbReset[nDims-1], beforeFullEnd);
1753  }
1754 
1755  ///////////////////////////////////////////////////////////////////////////
1756  // The outer dimensions are all on, so the mask is just given by the
1757  // mask for the innermost dimension
1758  ctx->SetCurrentBasicBlock(bbSetInnerMask); {
1759  llvm::Value *varyingCounter =
1760  lUpdateVaryingCounter(nDims-1, nDims, ctx, uniformCounterPtrs[nDims-1],
1761  dimVariables[nDims-1]->storagePtr, span);
1762  llvm::Value *smearEnd = ctx->BroadcastValue(
1763  endVals[nDims-1], LLVMTypes::Int32VectorType, "smear_end");
1764  llvm::Value *emask =
1765  ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SLT,
1766  varyingCounter, smearEnd);
1767  emask = ctx->I1VecToBoolVec(emask);
1768  ctx->SetInternalMask(emask);
1769  ctx->SetBlockEntryMask(emask);
1770 
1771  ctx->StoreInst(LLVMFalse, stepIndexAfterMaskedBodyPtr);
1772  ctx->BranchInst(bbMaskedBody);
1773  }
1774 
1775  ///////////////////////////////////////////////////////////////////////////
1776  // masked_body: set the mask and have the statements emit their
1777  // code again. Note that it's generally worthwhile having two copies
1778  // of the statements' code, since the code above is emitted with the
1779  // mask known to be all-on, which in turn leads to more efficient code
1780  // for that case.
1781  llvm::BasicBlock *bbStepInnerIndex =
1782  ctx->CreateBasicBlock("step_inner_index");
1783  llvm::BasicBlock *bbMaskedBodyContinue =
1784  ctx->CreateBasicBlock("foreach_masked_continue");
1785  ctx->SetCurrentBasicBlock(bbMaskedBody); {
1786  ctx->AddInstrumentationPoint("foreach loop body (masked)");
1787  ctx->SetContinueTarget(bbMaskedBodyContinue);
1788  ctx->DisableGatherScatterWarnings();
1789  ctx->SetBlockEntryMask(ctx->GetFullMask());
1790  stmts->EmitCode(ctx);
1791  ctx->EnableGatherScatterWarnings();
1792  ctx->BranchInst(bbMaskedBodyContinue);
1793  }
1794  ctx->SetCurrentBasicBlock(bbMaskedBodyContinue); {
1795  ctx->RestoreContinuedLanes();
1796  llvm::Value *stepIndex = ctx->LoadInst(stepIndexAfterMaskedBodyPtr);
1797  ctx->BranchInst(bbStepInnerIndex, bbReset[nDims-1], stepIndex);
1798  }
1799 
1800  ///////////////////////////////////////////////////////////////////////////
1801  // step the innermost index, for the case where we're doing the
1802  // innermost for loop over full vectors.
1803  ctx->SetCurrentBasicBlock(bbStepInnerIndex); {
1804  llvm::Value *counter = ctx->LoadInst(uniformCounterPtrs[nDims-1]);
1805  llvm::Value *newCounter =
1806  ctx->BinaryOperator(llvm::Instruction::Add, counter,
1807  LLVMInt32(span[nDims-1]), "new_counter");
1808  ctx->StoreInst(newCounter, uniformCounterPtrs[nDims-1]);
1809  ctx->BranchInst(bbOuterInExtras);
1810  }
1811 
1812  ///////////////////////////////////////////////////////////////////////////
1813  // foreach_exit: All done. Restore the old mask and clean up
1814  ctx->SetCurrentBasicBlock(bbExit);
1815 
1816  ctx->SetInternalMask(oldMask);
1817  ctx->SetFunctionMask(oldFunctionMask);
1818 
1819  ctx->EndForeach();
1820  ctx->EndScope();
1821 }

1930  return;

1933  // variable that holds the current program instance in each loop
1934  // iteration.
1935  if (sym->type == NULL) {
1936  Assert(m->errorCount > 0);
1937  return;
1938  }
1939  Assert(Type::Equal(sym->type,
1940  AtomicType::UniformInt64->GetAsConstType()));

1942 
1943  ctx->SetDebugPos(pos);
1944  ctx->EmitVariableDebugInfo(sym);
1945

1947  llvm::BasicBlock *bbFindNext =
1948  ctx->CreateBasicBlock("foreach_active_find_next");
1949  llvm::BasicBlock *bbBody = ctx->CreateBasicBlock("foreach_active_body");
1950  llvm::BasicBlock *bbCheckForMore =
1951  ctx->CreateBasicBlock("foreach_active_check_for_more");
1952  llvm::BasicBlock *bbDone = ctx->CreateBasicBlock("foreach_active_done");
1953 
1954  // Save the old mask so that we can restore it at the end
1955  llvm::Value *oldInternalMask = ctx->GetInternalMask();
1956 
1957  // Now, *maskBitsPtr will maintain a bitmask for the lanes that remain
1958  // to be processed by a pass through the loop body. It starts out with
1959  // the current execution mask (which should never be all off going in
1960  // to this)...
1961  llvm::Value *oldFullMask = ctx->GetFullMask();

1963  ctx->AllocaInst(LLVMTypes::Int64Type, "mask_bits");
1964  llvm::Value *movmsk = ctx->LaneMask(oldFullMask);
1965  ctx->StoreInst(movmsk, maskBitsPtr);
1966

1968  ctx->StartScope();
1969  ctx->StartForeach(FunctionEmitContext::FOREACH_ACTIVE);
1970  ctx->SetContinueTarget(bbCheckForMore);
1971 
1972  // Onward to find the first set of program instance to run the loop for
1973  ctx->BranchInst(bbFindNext);
1974 
1975  ctx->SetCurrentBasicBlock(bbFindNext); {
1976  // Load the bitmask of the lanes left to be processed
1977  llvm::Value *remainingBits = ctx->LoadInst(maskBitsPtr, "remaining_bits");
1978 
1979  // Find the index of the first set bit in the mask
1980  llvm::Function *ctlzFunc =
1981  m->module->getFunction("__count_trailing_zeros_i64");
1982  Assert(ctlzFunc != NULL);
1983  llvm::Value *firstSet = ctx->CallInst(ctlzFunc, NULL, remainingBits,
1984  "first_set");

1987  // variable.
1988  ctx->StoreInst(firstSet, sym->storagePtr);
1989

1996  llvm::Value *programIndex = ctx->ProgramIndexVector();

1999  llvm::Value *firstSet32 =
2000  ctx->TruncInst(firstSet, LLVMTypes::Int32Type, "first_set32");

2002 
2003  // Now set the execution mask based on doing a vector compare of
2004  // these two
2005  llvm::Value *iterMask =
2006  ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_EQ,
2007  firstSet32Smear, programIndex);
2008  iterMask = ctx->I1VecToBoolVec(iterMask);
2009

2011

2013  // the lane we're about to run.
2014  llvm::Value *setMask =
2015  ctx->BinaryOperator(llvm::Instruction::Shl, LLVMInt64(1),
2016  firstSet, "set_mask");
2017  llvm::Value *notSetMask = ctx->NotOperator(setMask);
2018  llvm::Value *newRemaining =
2019  ctx->BinaryOperator(llvm::Instruction::And, remainingBits,
2020  notSetMask, "new_remaining");

2022 
2023  // and onward to run the loop body...
2024  ctx->BranchInst(bbBody);
2025  }
2026 
2027  ctx->SetCurrentBasicBlock(bbBody); {
2028  ctx->SetBlockEntryMask(ctx->GetFullMask());
2029 
2030  // Run the code in the body of the loop. This is easy now.
2031  if (stmts)
2032  stmts->EmitCode(ctx);
2033 
2034  Assert(ctx->GetCurrentBasicBlock() != NULL);
2035  ctx->BranchInst(bbCheckForMore);
2036  }
2037 
2038  ctx->SetCurrentBasicBlock(bbCheckForMore); {
2039  ctx->RestoreContinuedLanes();
2040  // At the end of the loop body (either due to running the
2041  // statements normally, or a continue statement in the middle of
2042  // the loop that jumps to the end, see if there are any lanes left
2043  // to be processed.
2044  llvm::Value *remainingBits = ctx->LoadInst(maskBitsPtr, "remaining_bits");
2045  llvm::Value *nonZero =
2046  ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_NE,
2047  remainingBits, LLVMInt64(0), "remaining_ne_zero");
2048  ctx->BranchInst(bbFindNext, bbDone, nonZero);
2049  }
2050 
2051  ctx->SetCurrentBasicBlock(bbDone);
2052  ctx->SetInternalMask(oldInternalMask);
2053  ctx->EndForeach();
2054  ctx->EndScope();
2055 }

2464  std::map<llvm::BasicBlock *, llvm::BasicBlock *> nextBlock;

2467  we create the basic block for the next one, we'll use this to update
2468  the nextBlock map<> above. */
2469  llvm::BasicBlock *lastBlock;

2473 static bool
2474 lSwitchASTPreVisit(ASTNode *node, void *d) {
2475  if (dynamic_cast<SwitchStmt *>(node) != NULL)
2476  // don't continue recursively into a nested switch--we only want
2477  // our own case and default statements!
2478  return false;
2479 
2480  CaseStmt *cs = dynamic_cast<CaseStmt *>(node);
2481  DefaultStmt *ds = dynamic_cast<DefaultStmt *>(node);
2482 
2483  SwitchVisitInfo *svi = (SwitchVisitInfo *)d;
2484  llvm::BasicBlock *bb = NULL;
2485  if (cs != NULL) {
2486  // Complain if we've seen a case statement with the same value
2487  // already
2488  for (int i = 0; i < (int)svi->caseBlocks.size(); ++i) {
2489  if (svi->caseBlocks[i].first == cs->value) {
2490  Error(cs->pos, "Duplicate case value \"%d\".", cs->value);
2491  return true;
2492  }
2493  }
2494 
2495  // Otherwise create a new basic block for the code following this
2496  // 'case' statement and record the mappign between the case label
2497  // value and the basic block
2498  char buf[32];
2499  sprintf(buf, "case_%d", cs->value);
2500  bb = svi->ctx->CreateBasicBlock(buf);
2501  svi->caseBlocks.push_back(std::make_pair(cs->value, bb));
2502  }
2503  else if (ds != NULL) {
2504  // And complain if we've seen another 'default' label..
2505  if (svi->defaultBlock != NULL) {
2506  Error(ds->pos, "Multiple \"default\" lables in switch statement.");
2507  return true;
2508  }
2509  else {
2510  // Otherwise create a basic block for the code following the
2511  // "default".
2512  bb = svi->ctx->CreateBasicBlock("default");
2513  svi->defaultBlock = bb;
2514  }
2515  }
2516 
2517  // If we saw a "case" or "default" label, then update the map to record
2518  // that the block we just created follows the block created for the
2519  // previous label in the "switch".
2520  if (bb != NULL) {
2521  svi->nextBlock[svi->lastBlock] = bb;
2522  svi->lastBlock = bb;
2523  }
2524 
2525  return true;
2526 }

2531  if (ctx->GetCurrentBasicBlock() == NULL)
2532  return;
2533

2535  if (expr == NULL || ((type = expr->GetType()) == NULL)) {
2536  AssertPos(pos, m->errorCount > 0);
2537  return;
2538  }
