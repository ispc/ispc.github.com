95 /* Save %ebx in case it's the PIC register */
96 static void __cpuidex(int info[4], int level, int count) {
97  __asm__ __volatile__ ("xchg{l}\t{%%}ebx, %1\n\t"
98  "cpuid\n\t"
99  "xchg{l}\t{%%}ebx, %1\n\t"

106 static const char *
107 lGetSystemISA() {
108  int info[4];
109  __cpuid(info, 1);
110 
111  if ((info[2] & (1 << 28)) != 0) { // AVX
112  // AVX1 for sure....
113  // Ivy Bridge?
114  if ((info[2] & (1 << 29)) != 0 && // F16C
115  (info[2] & (1 << 30)) != 0) { // RDRAND
116  // So far, so good. AVX2?
117  // Call cpuid with eax=7, ecx=0
118  int info2[4];
119  __cpuidex(info2, 7, 0);
120  if ((info2[1] & (1 << 5)) != 0)
121  return "avx2";
122  else
123  return "avx1.1";
124  }
125  // Regular AVX
126  return "avx";
127  }
128  else if ((info[2] & (1 << 19)) != 0)
129  return "sse4";
130  else if ((info[3] & (1 << 26)) != 0)
131  return "sse2";
132  else {
133  fprintf(stderr, "Unable to detect supported SSE/AVX ISA. Exiting.\n");
134  exit(1);
135  }
136 }

592 static bool
593 lGenericTypeLayoutIndeterminate(llvm::Type *type) {
594  if (type->isPrimitiveType() || type->isIntegerTy())
595  return false;
596 
597  if (type == LLVMTypes::BoolVectorType ||
598  type == LLVMTypes::MaskType ||
599  type == LLVMTypes::Int1VectorType)
600  return true;
601 
602  llvm::ArrayType *at =
603  llvm::dyn_cast<llvm::ArrayType>(type);
604  if (at != NULL)
605  return lGenericTypeLayoutIndeterminate(at->getElementType());
606 
607  llvm::PointerType *pt =
608  llvm::dyn_cast<llvm::PointerType>(type);
609  if (pt != NULL)
610  return false;
611 
612  llvm::StructType *st =
613  llvm::dyn_cast<llvm::StructType>(type);
614  if (st != NULL) {
615  for (int i = 0; i < (int)st->getNumElements(); ++i)
616  if (lGenericTypeLayoutIndeterminate(st->getElementType(i)))
617  return true;
618  return false;
619  }
620 
621  Assert(llvm::isa<llvm::VectorType>(type));
622  return true;
623 }
