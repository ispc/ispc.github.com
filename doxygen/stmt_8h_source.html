44 /** @brief Interface class for statements in the ispc language.
45 
46  This abstract base-class encapsulates methods that AST nodes for
47  statements in the language must implement.
48  */
49 class Stmt : public ASTNode {
50 public:
51  Stmt(SourcePos p) : ASTNode(p) { }
52 
53  /** Emit LLVM IR for the statement, using the FunctionEmitContext to create the
54  necessary instructions.
55  */
56  virtual void EmitCode(FunctionEmitContext *ctx) const = 0;
57 
58  /** Print a representation of the statement (and any children AST
59  nodes) to standard output. This method is used for debuggins. */
60  virtual void Print(int indent) const = 0;
61 
62  // Redeclare these methods with Stmt * return values, rather than
63  // ASTNode *s, as in the original ASTNode declarations of them. We'll
64  // also provide a default implementation of Optimize(), since most
65  // Stmts don't have anything to do here.
66  virtual Stmt *Optimize();
67  virtual Stmt *TypeCheck() = 0;
68 };
