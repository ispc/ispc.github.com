148  'from' instruction in the IR to the 'to' instruction.

150  For flexibility, this function takes an llvm::Value rather than an
151  llvm::Instruction for the 'to' parameter; at some places in the code

170 /** We have a protocol with the front-end LLVM IR code generation process
171  that allows us to encode the source file position that corresponds with
172  instructions. (For example, this allows us to issue performance
173  warnings related to things like scatter and gather after optimization
174  has been performed, so that we aren't warning about scatters and
175  gathers that have been improved to stores and loads by optimization
176  passes.) Note that this is slightly redundant with the source file
177  position encoding generated for debugging symbols, though we don't
178  always generate debugging information but we do always generate this
179  position data.
180 
181  This function finds the SourcePos that the metadata in the instruction
182  (if present) corresponds to. See the implementation of
183  FunctionEmitContext::addGSMetadata(), which encodes the source position during
184  code generation.
185 
186  @param inst Instruction to try to find the source position of
187  @param pos Output variable in which to store the position
188  @returns True if source file position metadata was present and *pos
189  has been set. False otherwise.
190 */
191 static bool
192 lGetSourcePosFromMetadata(const llvm::Instruction *inst, SourcePos *pos) {
193  llvm::MDNode *filename = inst->getMetadata("filename");
194  llvm::MDNode *first_line = inst->getMetadata("first_line");
195  llvm::MDNode *first_column = inst->getMetadata("first_column");
196  llvm::MDNode *last_line = inst->getMetadata("last_line");
197  llvm::MDNode *last_column = inst->getMetadata("last_column");
198  if (!filename || !first_line || !first_column || !last_line || !last_column)
199  return false;
200 
201  // All of these asserts are things that FunctionEmitContext::addGSMetadata() is
202  // expected to have done in its operation
203  llvm::MDString *str = llvm::dyn_cast<llvm::MDString>(filename->getOperand(0));
204  Assert(str);
205  llvm::ConstantInt *first_lnum =
206  llvm::dyn_cast<llvm::ConstantInt>(first_line->getOperand(0));
207  Assert(first_lnum);
208  llvm::ConstantInt *first_colnum =
209  llvm::dyn_cast<llvm::ConstantInt>(first_column->getOperand(0));
210  Assert(first_column);
211  llvm::ConstantInt *last_lnum =
212  llvm::dyn_cast<llvm::ConstantInt>(last_line->getOperand(0));
213  Assert(last_lnum);
214  llvm::ConstantInt *last_colnum =
215  llvm::dyn_cast<llvm::ConstantInt>(last_column->getOperand(0));
216  Assert(last_column);
217 
218  *pos = SourcePos(str->getString().data(), (int)first_lnum->getZExtValue(),
219  (int)first_colnum->getZExtValue(), (int)last_lnum->getZExtValue(),
220  (int)last_colnum->getZExtValue());
221  return true;
222 }

284 /** Given a vector of constant values (int, float, or bool) representing an
285  execution mask, convert it to a bitvector where the 0th bit corresponds
286  to the first vector value and so forth.
287 */
288 static uint64_t
289 lConstElementsToMask(const llvm::SmallVector<llvm::Constant *,
290  ISPC_MAX_NVEC> &elements) {
291  Assert(elements.size() <= 64);
292 
293  uint64_t mask = 0;
294  for (unsigned int i = 0; i < elements.size(); ++i) {
295  llvm::APInt intMaskValue;
296  // SSE has the "interesting" approach of encoding blending
297  // masks as <n x float>.
298  llvm::ConstantFP *cf = llvm::dyn_cast<llvm::ConstantFP>(elements[i]);
299  if (cf != NULL) {
300  llvm::APFloat apf = cf->getValueAPF();
301  intMaskValue = apf.bitcastToAPInt();
302  }
303  else {
304  // Otherwise get it as an int
305  llvm::ConstantInt *ci = llvm::dyn_cast<llvm::ConstantInt>(elements[i]);
306  Assert(ci != NULL); // vs return -1 if NULL?
307  intMaskValue = ci->getValue();
308  }
309  // Is the high-bit set? If so, OR in the appropriate bit in
310  // the result mask
311  if (intMaskValue.countLeadingOnes() > 0)
312  mask |= (1ull << i);
313  }
314  return mask;
315 }

318 /** Given an llvm::Value represinting a vector mask, see if the value is a
319  constant. If so, return true and set *bits to be the integer mask
320  found by taking the high bits of the mask values in turn and
321  concatenating them into a single integer. In other words, given the
322  4-wide mask: < 0xffffffff, 0, 0, 0xffffffff >, we have 0b1001 = 9.
323  */
324 static bool
325 lGetMask(llvm::Value *factor, uint64_t *mask) {
326  llvm::ConstantDataVector *cdv = llvm::dyn_cast<llvm::ConstantDataVector>(factor);
327  if (cdv != NULL) {
328  llvm::SmallVector<llvm::Constant *, ISPC_MAX_NVEC> elements;
329  for (int i = 0; i < (int)cdv->getNumElements(); ++i)
330  elements.push_back(cdv->getElementAsConstant(i));
331  *mask = lConstElementsToMask(elements);
332  return true;
333  }
334 
335  llvm::ConstantVector *cv = llvm::dyn_cast<llvm::ConstantVector>(factor);
336  if (cv != NULL) {
337  llvm::SmallVector<llvm::Constant *, ISPC_MAX_NVEC> elements;
338  for (int i = 0; i < (int)cv->getNumOperands(); ++i) {
339  llvm::Constant *c =
340  llvm::dyn_cast<llvm::Constant>(cv->getOperand(i));
341  if (c == NULL)
342  return false;
343  elements.push_back(c);
344  }
345  *mask = lConstElementsToMask(elements);
346  return true;
347  }
348  else if (llvm::isa<llvm::ConstantAggregateZero>(factor)) {
349  *mask = 0;

372 /** Determines if the given mask value is all on, all off, mixed, or
373  unknown at compile time.
374 */
375 static MaskStatus
376 lGetMaskStatus(llvm::Value *mask, int vecWidth = -1) {
377  uint64_t bits;
378  if (lGetMask(mask, &bits) == false)
379  return UNKNOWN;
380

383 
384  if (vecWidth == -1)
385  vecWidth = g->target->getVectorWidth();
386  Assert(vecWidth <= 64);
387 
388  for (int i = 0; i < vecWidth; ++i) {
389  if ((bits & (1ull << i)) == 0)
390  return MIXED;
391  }
392  return ALL_ON;
393 }

710 bool
711 IntrinsicsOpt::runOnBasicBlock(llvm::BasicBlock &bb) {
712  DEBUG_START_PASS("IntrinsicsOpt");
713 
714  llvm::Function *avxMaskedLoad32 =
715  llvm::Intrinsic::getDeclaration(m->module, llvm::Intrinsic::x86_avx_maskload_ps_256);
716  llvm::Function *avxMaskedLoad64 =
717  llvm::Intrinsic::getDeclaration(m->module, llvm::Intrinsic::x86_avx_maskload_pd_256);
718  llvm::Function *avxMaskedStore32 =
719  llvm::Intrinsic::getDeclaration(m->module, llvm::Intrinsic::x86_avx_maskstore_ps_256);
720  llvm::Function *avxMaskedStore64 =
721  llvm::Intrinsic::getDeclaration(m->module, llvm::Intrinsic::x86_avx_maskstore_pd_256);
722  Assert(avxMaskedLoad32 != NULL && avxMaskedStore32 != NULL);
723  Assert(avxMaskedLoad64 != NULL && avxMaskedStore64 != NULL);
724 
725  bool modifiedAny = false;
726  restart:
727  for (llvm::BasicBlock::iterator iter = bb.begin(), e = bb.end(); iter != e; ++iter) {
728  llvm::CallInst *callInst = llvm::dyn_cast<llvm::CallInst>(&*iter);
729  if (callInst == NULL || callInst->getCalledFunction() == NULL)
730  continue;
731 
732  BlendInstruction *blend = matchingBlendInstruction(callInst->getCalledFunction());
733  if (blend != NULL) {
734  llvm::Value *v[2] = { callInst->getArgOperand(blend->op0),
735  callInst->getArgOperand(blend->op1) };
736  llvm::Value *factor = callInst->getArgOperand(blend->opFactor);
737 
738  // If the values are the same, then no need to blend..
739  if (v[0] == v[1]) {
740  llvm::ReplaceInstWithValue(iter->getParent()->getInstList(),
741  iter, v[0]);
742  modifiedAny = true;
743  goto restart;
744  }
745 
746  // If one of the two is undefined, we're allowed to replace
747  // with the value of the other. (In other words, the only
748  // valid case is that the blend factor ends up having a value
749  // that only selects from the defined one of the two operands,
750  // otherwise the result is undefined and any value is fine,
751  // ergo the defined one is an acceptable result.)
752  if (lIsUndef(v[0])) {
753  llvm::ReplaceInstWithValue(iter->getParent()->getInstList(),
754  iter, v[1]);
755  modifiedAny = true;
756  goto restart;
757  }
758  if (lIsUndef(v[1])) {
759  llvm::ReplaceInstWithValue(iter->getParent()->getInstList(),
760  iter, v[0]);
761  modifiedAny = true;
762  goto restart;
763  }
764 
765  uint64_t mask;
766  if (lGetMask(factor, &mask) == true) {
767  llvm::Value *value = NULL;
768  if (mask == 0)
769  // Mask all off -> replace with the first blend value
770  value = v[0];
771  else if (mask == blend->allOnMask)
772  // Mask all on -> replace with the second blend value
773  value = v[1];
774 
775  if (value != NULL) {
776  llvm::ReplaceInstWithValue(iter->getParent()->getInstList(),
777  iter, value);
778  modifiedAny = true;
779  goto restart;
780  }
781  }
782  }
783  else if (matchesMaskInstruction(callInst->getCalledFunction())) {
784  llvm::Value *factor = callInst->getArgOperand(0);
785  uint64_t mask;
786  if (lGetMask(factor, &mask) == true) {
787  // If the vector-valued mask has a known value, replace it
788  // with the corresponding integer mask from its elements
789  // high bits.
790  llvm::Value *value = (callInst->getType() == LLVMTypes::Int32Type) ?
791  LLVMInt32(mask) : LLVMInt64(mask);
792  llvm::ReplaceInstWithValue(iter->getParent()->getInstList(),
793  iter, value);
794  modifiedAny = true;
795  goto restart;
796  }
797  }
798  else if (callInst->getCalledFunction() == avxMaskedLoad32 ||
799  callInst->getCalledFunction() == avxMaskedLoad64) {
800  llvm::Value *factor = callInst->getArgOperand(1);
801  uint64_t mask;
802  if (lGetMask(factor, &mask) == true) {
803  if (mask == 0) {
804  // nothing being loaded, replace with undef value
805  llvm::Type *returnType = callInst->getType();
806  Assert(llvm::isa<llvm::VectorType>(returnType));
807  llvm::Value *undefValue = llvm::UndefValue::get(returnType);
808  llvm::ReplaceInstWithValue(iter->getParent()->getInstList(),
809  iter, undefValue);
810  modifiedAny = true;
811  goto restart;
812  }
813  else if (mask == 0xff) {
814  // all lanes active; replace with a regular load
815  llvm::Type *returnType = callInst->getType();
816  Assert(llvm::isa<llvm::VectorType>(returnType));
817  // cast the i8 * to the appropriate type
818  const char *name = LLVMGetName(callInst->getArgOperand(0), "_cast");
819  llvm::Value *castPtr =
820  new llvm::BitCastInst(callInst->getArgOperand(0),
821  llvm::PointerType::get(returnType, 0),
822  name, callInst);
823  lCopyMetadata(castPtr, callInst);
824  int align;
825  if (g->opt.forceAlignedMemory)
826  align = 0;
827  else
828  align = callInst->getCalledFunction() == avxMaskedLoad32 ? 4 : 8;
829  name = LLVMGetName(callInst->getArgOperand(0), "_load");
830  llvm::Instruction *loadInst =
831  new llvm::LoadInst(castPtr, name, false /* not volatile */,
832  align, (llvm::Instruction *)NULL);
833  lCopyMetadata(loadInst, callInst);
834  llvm::ReplaceInstWithInst(callInst, loadInst);
835  modifiedAny = true;
836  goto restart;
837  }
838  }
839  }
840  else if (callInst->getCalledFunction() == avxMaskedStore32 ||
841  callInst->getCalledFunction() == avxMaskedStore64) {
842  // NOTE: mask is the 2nd parameter, not the 3rd one!!
843  llvm::Value *factor = callInst->getArgOperand(1);
844  uint64_t mask;
845  if (lGetMask(factor, &mask) == true) {
846  if (mask == 0) {
847  // nothing actually being stored, just remove the inst
848  callInst->eraseFromParent();
849  modifiedAny = true;
850  goto restart;
851  }
852  else if (mask == 0xff) {
853  // all lanes storing, so replace with a regular store
854  llvm::Value *rvalue = callInst->getArgOperand(2);
855  llvm::Type *storeType = rvalue->getType();
856  const char *name = LLVMGetName(callInst->getArgOperand(0),
857  "_ptrcast");
858  llvm::Value *castPtr =
859  new llvm::BitCastInst(callInst->getArgOperand(0),
860  llvm::PointerType::get(storeType, 0),
861  name, callInst);
862  lCopyMetadata(castPtr, callInst);
863 
864  llvm::StoreInst *storeInst =
865  new llvm::StoreInst(rvalue, castPtr, (llvm::Instruction *)NULL);
866  int align;
867  if (g->opt.forceAlignedMemory)
868  align = 0;
869  else
870  align = callInst->getCalledFunction() == avxMaskedStore32 ? 4 : 8;
871  storeInst->setAlignment(align);
872  lCopyMetadata(storeInst, callInst);
873  llvm::ReplaceInstWithInst(callInst, storeInst);
874 
875  modifiedAny = true;
876  goto restart;
877  }
878  }
879  }
880  }
881 
882  DEBUG_END_PASS("IntrinsicsOpt");
883 
884  return modifiedAny;
885 }

1026 /** Check to make sure that this value is actually a pointer in the end.
1027  We need to make sure that given an expression like vec(offset) +
1028  ptr2int(ptr), lGetBasePointer() doesn't return vec(offset) for the base
1029  pointer such that we then treat ptr2int(ptr) as an offset. This ends
1030  up being important so that we don't generate LLVM GEP instructions like
1031  "gep inttoptr 8, i64 %ptr", which in turn can lead to incorrect code
1032  since LLVM's pointer aliasing analysis assumes that operands after the
1033  first one to a GEP aren't pointers.
1034  */
1035 static llvm::Value *
1036 lCheckForActualPointer(llvm::Value *v) {
1037  if (v == NULL)
1038  return NULL;
1039  else if (llvm::isa<llvm::PointerType>(v->getType()))
1040  return v;
1041  else if (llvm::isa<llvm::PtrToIntInst>(v))
1042  return v;
1043  else {
1044  llvm::ConstantExpr *uce =
1045  llvm::dyn_cast<llvm::ConstantExpr>(v);
1046  if (uce != NULL &&
1047  uce->getOpcode() == llvm::Instruction::PtrToInt)
1048  return v;
1049  return NULL;
1050  }
1051 }

1054 /** Given a llvm::Value representing a varying pointer, this function
1055  checks to see if all of the elements of the vector have the same value
1056  (i.e. there's a common base pointer). If so, it returns the common
1057  pointer value; otherwise it returns NULL.
1058  */
1059 static llvm::Value *
1060 lGetBasePointer(llvm::Value *v) {
1061  if (llvm::isa<llvm::InsertElementInst>(v) ||
1062  llvm::isa<llvm::ShuffleVectorInst>(v)) {
1063  llvm::Value *elements[ISPC_MAX_NVEC];
1064  LLVMFlattenInsertChain(v, g->target->getVectorWidth(), elements);
1065 
1066  // Make sure none of the elements is undefined.
1067  // TODO: it's probably ok to allow undefined elements and return
1068  // the base pointer if all of the other elements have the same
1069  // value.
1070  for (int i = 0; i < g->target->getVectorWidth(); ++i)
1071  if (elements[i] == NULL)
1072  return NULL;
1073 
1074  // Do all of the elements have the same value?
1075  for (int i = 0; i < g->target->getVectorWidth()-1; ++i)
1076  if (elements[i] != elements[i+1])
1077  return NULL;
1078 
1079  return lCheckForActualPointer(elements[0]);
1080  }
1081 
1082  // This case comes up with global/static arrays
1083  if (llvm::ConstantVector *cv = llvm::dyn_cast<llvm::ConstantVector>(v)) {
1084  return lCheckForActualPointer(cv->getSplatValue());
1085  }
1086  else if (llvm::ConstantDataVector *cdv = llvm::dyn_cast<llvm::ConstantDataVector>(v)) {
1087  return lCheckForActualPointer(cdv->getSplatValue());
1088  }
1089 
1090  return NULL;
1091 }

1094 /** Given the two operands to a constant add expression, see if we have the
1095  form "base pointer + offset", whee op0 is the base pointer and op1 is
1096  the offset; if so return the base and the offset. */
1097 static llvm::Constant *
1098 lGetConstantAddExprBaseOffset(llvm::Constant *op0, llvm::Constant *op1,
1099  llvm::Constant **delta) {
1100  llvm::ConstantExpr *op = llvm::dyn_cast<llvm::ConstantExpr>(op0);
1101  if (op == NULL || op->getOpcode() != llvm::Instruction::PtrToInt)
1102  // the first operand isn't a pointer
1103  return NULL;
1104 
1105  llvm::ConstantInt *opDelta = llvm::dyn_cast<llvm::ConstantInt>(op1);
1106  if (opDelta == NULL)
1107  // the second operand isn't an integer operand
1108  return NULL;
1109 
1110  *delta = opDelta;
1111  return op0;
1112 }

1117  llvm::InsertValueInst *iv = llvm::dyn_cast<llvm::InsertValueInst>(v);
1118  if (iv == NULL)
1119  return NULL;
1120 
1121  Assert(iv->hasIndices() && iv->getNumIndices() == 1);

1124  else
1125  return lExtractFromInserts(iv->getAggregateOperand(), index);
1126 }

1129 /** Given a varying pointer in ptrs, this function checks to see if it can
1130  be determined to be indexing from a common uniform base pointer. If
1131  so, the function returns the base pointer llvm::Value and initializes
1132  *offsets with an int vector of the per-lane offsets
1133  */
1134 static llvm::Value *
1135 lGetBasePtrAndOffsets(llvm::Value *ptrs, llvm::Value **offsets,
1136  llvm::Instruction *insertBefore) {
1137  if (g->debugPrint) {
1138  fprintf(stderr, "lGetBasePtrAndOffsets\n");
1139  LLVMDumpValue(ptrs);
1140  }
1141 
1142  llvm::Value *base = lGetBasePointer(ptrs);
1143  if (base != NULL) {
1144  // We have a straight up varying pointer with no indexing that's
1145  // actually all the same value.
1146  if (g->target->is32Bit())
1147  *offsets = LLVMInt32Vector(0);
1148  else
1149  *offsets = LLVMInt64Vector((int64_t)0);
1150  return base;
1151  }
1152 
1153  llvm::BinaryOperator *bop = llvm::dyn_cast<llvm::BinaryOperator>(ptrs);
1154  if (bop != NULL && bop->getOpcode() == llvm::Instruction::Add) {
1155  // If we have a common pointer plus something, then we're also
1156  // good.
1157  if ((base = lGetBasePtrAndOffsets(bop->getOperand(0),
1158  offsets, insertBefore)) != NULL) {
1159  *offsets =
1160  llvm::BinaryOperator::Create(llvm::Instruction::Add, *offsets,
1161  bop->getOperand(1), "new_offsets",
1162  insertBefore);
1163  return base;
1164  }
1165  else if ((base = lGetBasePtrAndOffsets(bop->getOperand(1),
1166  offsets, insertBefore)) != NULL) {
1167  *offsets =
1168  llvm::BinaryOperator::Create(llvm::Instruction::Add, *offsets,
1169  bop->getOperand(0), "new_offsets",
1170  insertBefore);
1171  return base;
1172  }
1173  }
1174 
1175  llvm::ConstantVector *cv = llvm::dyn_cast<llvm::ConstantVector>(ptrs);
1176  if (cv != NULL) {
1177  // Indexing into global arrays can lead to this form, with
1178  // ConstantVectors..
1179  llvm::SmallVector<llvm::Constant *, ISPC_MAX_NVEC> elements;
1180  for (int i = 0; i < (int)cv->getNumOperands(); ++i) {
1181  llvm::Constant *c =
1182  llvm::dyn_cast<llvm::Constant>(cv->getOperand(i));
1183  if (c == NULL)
1184  return NULL;
1185  elements.push_back(c);
1186  }
1187 
1188  llvm::Constant *delta[ISPC_MAX_NVEC];
1189  for (unsigned int i = 0; i < elements.size(); ++i) {
1190  // For each element, try to decompose it into either a straight
1191  // up base pointer, or a base pointer plus an integer value.
1192  llvm::ConstantExpr *ce = llvm::dyn_cast<llvm::ConstantExpr>(elements[i]);
1193  if (ce == NULL)
1194  return NULL;
1195

2797 /** This function determines whether it makes sense (and is safe) to
2798  generate a vector load of width vectorWidth, starting at *iter. It
2799  returns true if so, setting *newIter to point to the next element in
2800  the set that isn't taken care of by the generated load. If a vector
2801  load of the given width doesn't make sense, then false is returned.
2802  */
2803 static bool
2804 lVectorLoadIsEfficient(std::set<int64_t>::iterator iter,
2805  std::set<int64_t>::iterator end,
2806  std::set<int64_t>::iterator *newIter, int vectorWidth) {
2807  // We're considering a vector load of width vectorWidth, starting at
2808  // the offset "start".
2809  int64_t start = *iter;
2810 
2811  // The basic idea is that we'll look at the subsequent elements in the
2812  // load set after the initial one at start. As long as subsequent
2813  // elements:
2814  //
2815  // 1. Aren't so far separated that they no longer fit into the range
2816  // [start, start+vectorWidth)
2817  //
2818  // 2. And don't have too large a gap in between them (e.g., it's not
2819  // worth generating an 8-wide load for two elements with offsets 0
2820  // and 7, but no loads requested in between).
2821  //
2822  // Then we continue moving forward through the elements until we either
2823  // fill up the vector or run out of elements.
2824 
2825  // lastAccepted holds the last offset we've processed and accepted as
2826  // valid for the vector load underconsideration
2827  int64_t lastAccepted = start;
2828 
2829  while (iter != end) {
2830  // What is the separation in offset values from the last element we
2831  // added to the set for this load?
2832  int64_t delta = *iter - lastAccepted;
2833  if (delta > 3)
2834  // If there's too big a gap, then we won't issue the load
2835  return false;
2836 
2837  int64_t span = *iter - start + 1;
2838 
2839  if (span == vectorWidth) {
2840  // We've extended far enough that we have exactly filled up the
2841  // entire vector width; we can't go any further, so return with
2842  // success. (Update *newIter to point at the next element
2843  // after the last one accepted here.)
2844  *newIter = ++iter;
2845  return true;
2846  }
2847  else if (span > vectorWidth) {
2848  // The current offset won't fit into a vectorWidth-wide load
2849  // starting from start. It's still generally worthwhile
2850  // issuing the load we've been considering, though, since it
2851  // will provide values for a number of previous offsets. This
2852  // load will have one or more elements at the end of its range
2853  // that is not needed by any of the offsets under
2854  // consideration. As such, there are three cases where issuing
2855  // this load is a bad idea:
2856  //
2857  // 1. 2-wide loads: we know that we haven't completely filled
2858  // the 2-wide vector, since otherwise the if() test above
2859  // would have succeeded previously. Therefore, we must have
2860  // a situation with offsets like (4,6,...); it would be a
2861  // silly idea to issue a 2-wide load to get the value for
2862  // the 4 offset, versus failing here and issuing a scalar
2863  // load instead.
2864  //
2865  // 2. If there are too many unnecessary values at the end of
2866  // the load extent (defined as more than half of them)--in
2867  // this case, it'd be better to issue a vector load of
2868  // smaller width anyway.
2869  //
2870  // 3. If the gap between the last accepted offset and the
2871  // current one under consideration is more than the page
2872  // size. In this case we can't be sure whether or not some
2873  // of the unused elements at the end of the load will
2874  // straddle a page boundary and thus lead to an undesirable
2875  // fault. (It's hard to imagine this happening in practice,
2876  // except under contrived circumstances, but better safe
2877  // than sorry.)
2878  const int pageSize = 4096;
2879  if (vectorWidth != 2 &&
2880  (lastAccepted - start) > (vectorWidth / 2) &&
2881  (*iter - lastAccepted) < pageSize) {
2882  *newIter = iter;
2883  return true;
2884  }
2885  else
2886  return false;
2887  }
2888 
2889  // Continue moving forward
2890  lastAccepted = *iter;
2891  ++iter;
2892  }
2893 
2894  return false;
2895 }
