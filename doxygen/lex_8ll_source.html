543 /** Return the integer version of a binary constant from a string.
544  */
545 static uint64_t
546 lParseBinary(const char *ptr, SourcePos pos, char **endPtr) {
547  uint64_t val = 0;
548  bool warned = false;
549 
550  while (*ptr == '0' || *ptr == '1') {
551  if ((val & (((int64_t)1)<<63)) && warned == false) {
552  // We're about to shift out a set bit
553  Warning(pos, "Can't represent binary constant with a 64-bit integer type");
554  warned = true;

565 static int
566 lParseInteger(bool dotdotdot) {
567  int ls = 0, us = 0;
568 
569  char *endPtr = NULL;
570  if (yytext[0] == '0' && yytext[1] == 'b')
571  yylval.intVal = lParseBinary(yytext+2, yylloc, &endPtr);
572  else {
573 #if defined(ISPC_IS_WINDOWS) && !defined(__MINGW32__)
574  yylval.intVal = _strtoui64(yytext, &endPtr, 0);
575 #else
576  // FIXME: should use strtouq and then issue an error if we can't
577  // fit into 64 bits...
578  yylval.intVal = strtoull(yytext, &endPtr, 0);
579 #endif
580  }
581 
582  bool kilo = false, mega = false, giga = false;
583  for (; *endPtr; endPtr++) {
584  if (*endPtr == 'k')
585  kilo = true;
586  else if (*endPtr == 'M')
587  mega = true;
588  else if (*endPtr == 'G')
589  giga = true;
590  else if (*endPtr == 'l' || *endPtr == 'L')
591  ls++;
592  else if (*endPtr == 'u' || *endPtr == 'U')
593  us++;
594  else
595  Assert(dotdotdot && *endPtr == '.');
596  }
597  if (kilo)
598  yylval.intVal *= 1024;
599  if (mega)
600  yylval.intVal *= 1024*1024;
601  if (giga)
602  yylval.intVal *= 1024*1024*1024;
603 
604  if (dotdotdot) {
605  if (ls >= 2)
606  return us ? TOKEN_UINT64DOTDOTDOT_CONSTANT : TOKEN_INT64DOTDOTDOT_CONSTANT;
607  else if (ls == 1)
608  return us ? TOKEN_UINT32DOTDOTDOT_CONSTANT : TOKEN_INT32DOTDOTDOT_CONSTANT;
609 
610  // See if we can fit this into a 32-bit integer...
611  if ((yylval.intVal & 0xffffffff) == yylval.intVal)
612  return us ? TOKEN_UINT32DOTDOTDOT_CONSTANT : TOKEN_INT32DOTDOTDOT_CONSTANT;
613  else
614  return us ? TOKEN_UINT64DOTDOTDOT_CONSTANT : TOKEN_INT64DOTDOTDOT_CONSTANT;
615  }
616  else {
617  if (ls >= 2)
618  return us ? TOKEN_UINT64_CONSTANT : TOKEN_INT64_CONSTANT;
619  else if (ls == 1)
620  return us ? TOKEN_UINT32_CONSTANT : TOKEN_INT32_CONSTANT;
621  else if (us) {
622  // u suffix only
623  if (yylval.intVal <= 0xffffffffL)
624  return TOKEN_UINT32_CONSTANT;
625  else
626  return TOKEN_UINT64_CONSTANT;
627  }
628  else {
629  // No u or l suffix
630  // If we're compiling to an 8-bit mask target and the constant
631  // fits into 8 bits, return an 8-bit int.
632  if (g->target->getMaskBitCount() == 8) {
633  if (yylval.intVal <= 0x7fULL)
634  return TOKEN_INT8_CONSTANT;
635  else if (yylval.intVal <= 0xffULL)
636  return TOKEN_UINT8_CONSTANT;
637  }
638  // And similarly for 16-bit masks and constants
639  if (g->target->getMaskBitCount() == 16) {
640  if (yylval.intVal <= 0x7fffULL)
641  return TOKEN_INT16_CONSTANT;
642  else if (yylval.intVal <= 0xffffULL)
643  return TOKEN_UINT16_CONSTANT;
644  }
645  // Otherwise, see if we can fit this into a 32-bit integer...
646  if (yylval.intVal <= 0x7fffffffULL)
647  return TOKEN_INT32_CONSTANT;
648  else if (yylval.intVal <= 0xffffffffULL)
649  return TOKEN_UINT32_CONSTANT;
650  else if (yylval.intVal <= 0x7fffffffffffffffULL)
651  return TOKEN_INT64_CONSTANT;
652  else
653  return TOKEN_UINT64_CONSTANT;
654  }

734 /** Given a pointer to a position in a string, return the character that it
735  represents, accounting for the escape characters supported in string
736  constants. (i.e. given the literal string "\\", return the character
737  '/'). The return value is the new position in the string and the
738  decoded character is returned in *pChar.
739 */
740 static char *
741 lEscapeChar(char *str, char *pChar, SourcePos *pos)
742 {
743  if (*str != '\\') {
744  *pChar = *str;
745  }
746  else {
747  char *tail;
748  ++str;
749  switch (*str) {
750  case '\'': *pChar = '\''; break;
751  case '\"': *pChar = '\"'; break;
752  case '?': *pChar = '\?'; break;
753  case '\\': *pChar = '\\'; break;
754  case 'a': *pChar = '\a'; break;
755  case 'b': *pChar = '\b'; break;
756  case 'f': *pChar = '\f'; break;
757  case 'n': *pChar = '\n'; break;
758  case 'r': *pChar = '\r'; break;
759  case 't': *pChar = '\t'; break;
760  case 'v': *pChar = '\v'; break;
761  // octal constants \012
762  case '0': case '1': case '2': case '3': case '4':
763  case '5': case '6': case '7':
764  *pChar = (char)strtol(str, &tail, 8);
765  str = tail - 1;
766  break;
767  // hexidecimal constant \xff
768  case 'x':
769  *pChar = (char)strtol(str, &tail, 16);
770  str = tail - 1;
771  break;
772  default:
773  Error(*pos, "Bad character escape sequence: '%s'.", str);
774  break;
775  }
776  }
777  ++str;
778  return str;
779 }

801 /** Compute the value 2^n, where the exponent is given as an integer.
802  There are more efficient ways to do this, for example by just slamming
803  the bits into the appropriate bits of the double, but let's just do the
804  obvious thing.
805 */
806 static double
807 ipow2(int exponent) {
808  if (exponent < 0)
809  return 1. / ipow2(-exponent);
810 
811  double ret = 1.;
812  while (exponent > 16) {
813  ret *= 65536.;
814  exponent -= 16;
815  }
816  while (exponent-- > 0)
817  ret *= 2.;
818  return ret;
819 }

822 /** Parse a hexadecimal-formatted floating-point number (C99 hex float
823  constant-style).
824 */
825 static double
826 lParseHexFloat(const char *ptr) {
827  Assert(ptr != NULL);
828 
829  Assert(ptr[0] == '0' && ptr[1] == 'x');
830  ptr += 2;
831 
832  // Start initializing the mantissa
833  Assert(*ptr == '0' || *ptr == '1');
834  double mantissa = (*ptr == '1') ? 1. : 0.;
835  ++ptr;
836 
837  if (*ptr == '.') {
838  // Is there a fraction part? If so, the i'th digit we encounter
839  // gives the 1/(16^i) component of the mantissa.
840  ++ptr;
841 
842  double scale = 1. / 16.;
843  // Keep going until we come to the 'p', which indicates that we've
844  // come to the exponent
845  while (*ptr != 'p') {
846  // Figure out the raw value from 0-15
847  int digit;
848  if (*ptr >= '0' && *ptr <= '9')
849  digit = *ptr - '0';
850  else if (*ptr >= 'a' && *ptr <= 'f')
851  digit = 10 + *ptr - 'a';
852  else {
853  Assert(*ptr >= 'A' && *ptr <= 'F');
854  digit = 10 + *ptr - 'A';
855  }
856 
857  // And add its contribution to the mantissa
858  mantissa += scale * digit;
859  scale /= 16.;
860  ++ptr;
861  }
862  }
863  else
864  // If there's not a '.', then we better be going straight to the
865  // exponent
866  Assert(*ptr == 'p');
867 
868  ++ptr; // skip the 'p'
869 
870  // interestingly enough, the exponent is provided base 10..
871  int exponent = (int)strtol(ptr, (char **)NULL, 10);
872 
873  // Does stdlib exp2() guarantee exact results for integer n where can
874  // be represented exactly as doubles? I would hope so but am not sure,
875  // so let's be sure.
876  return mantissa * ipow2(exponent);
877 }
