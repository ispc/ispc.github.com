157 /** After compilation completes, there's often a lot of extra debugging
158  metadata left around that isn't needed any more--for example, for
159  static functions that weren't actually used, function information for
160  functions that were inlined, etc. This function takes a llvm::Module
161  and tries to strip out all of this extra stuff.
162  */
163 static void
164 lStripUnusedDebugInfo(llvm::Module *module) {
165  if (g->generateDebuggingSymbols == false)
166  return;
167 
168  // loop over the compile units that contributed to the final module
169  if (llvm::NamedMDNode *cuNodes = module->getNamedMetadata("llvm.dbg.cu")) {
170  for (unsigned i = 0, ie = cuNodes->getNumOperands(); i != ie; ++i) {
171  llvm::MDNode *cuNode = cuNodes->getOperand(i);
172  llvm::DICompileUnit cu(cuNode);
173  llvm::DIArray subprograms = cu.getSubprograms();
174  std::vector<llvm::Value *> usedSubprograms;
175 
176  if (subprograms.getNumElements() == 0)
177  continue;
178 
179  // And now loop over the subprograms inside each compile unit.
180  for (unsigned j = 0, je = subprograms.getNumElements(); j != je; ++j) {
181  llvm::MDNode *spNode =
182  llvm::dyn_cast<llvm::MDNode>(subprograms->getOperand(j));
183  Assert(spNode != NULL);
184  llvm::DISubprogram sp(spNode);
185 
186  // Get the name of the subprogram. Start with the mangled
187  // name; if that's empty then we have an export'ed
188  // function, so grab the unmangled name in that case.
189  std::string name = sp.getLinkageName();
190  if (name == "")
191  name = sp.getName();
192 
193  // Does the llvm::Function for this function survive in the
194  // module?
195  if (module->getFunction(name) != NULL)
196  usedSubprograms.push_back(sp);
197  }
198 
199  Debug(SourcePos(), "%d / %d functions left in module with debug "
200  "info.", (int)usedSubprograms.size(),
201  (int)subprograms.getNumElements());
202 
203  // We'd now like to replace the array of subprograms in the
204  // compile unit with only the ones that actually have function
205  // definitions present. Unfortunately, llvm::DICompileUnit
206  // doesn't provide a method to set the subprograms. Therefore,
207  // we end up needing to directly stuff a new array into the
208  // appropriate slot (number 12) in the MDNode for the compile
209  // unit.
210  //
211  // Because this is all so hard-coded and would break if the
212  // debugging metadata organization on the LLVM side changed,
213  // here is a bunch of asserting to make sure that element 12 of
214  // the compile unit's MDNode has the subprograms array....
215  //
216  // Update: This is not an approved way of working with debug info
217  // metadata. It's not supposed to be deleted. But in out use-case
218  // it's quite useful thing, as we link in bunch of unnecessary
219  // stuff and remove it later on. Removing it is useful, as it
220  // reduces size of the binary significantly (manyfold for small
221  // programs).
222 #if defined(LLVM_3_2)
223  llvm::MDNode *nodeSPMD =
224  llvm::dyn_cast<llvm::MDNode>(cuNode->getOperand(12));
225  Assert(nodeSPMD != NULL);
226  llvm::MDNode *nodeSPMDArray =
227  llvm::dyn_cast<llvm::MDNode>(nodeSPMD->getOperand(0));
228  llvm::DIArray nodeSPs(nodeSPMDArray);
229  Assert(nodeSPs.getNumElements() == subprograms.getNumElements());
230  for (int i = 0; i < (int)nodeSPs.getNumElements(); ++i)
231  Assert(nodeSPs.getElement(i) == subprograms.getElement(i));
232 
233  // And now we can go and stuff it into the node with some
234  // confidence...
235  llvm::Value *usedSubprogramsArray =
236  m->diBuilder->getOrCreateArray(llvm::ArrayRef<llvm::Value *>(usedSubprograms));
237  llvm::MDNode *replNode =
238  llvm::MDNode::get(*g->ctx, llvm::ArrayRef<llvm::Value *>(usedSubprogramsArray));
239  cuNode->replaceOperandWith(12, replNode);
240 #else // LLVM 3.3+
241  llvm::MDNode *nodeSPMDArray =
242  llvm::dyn_cast<llvm::MDNode>(cuNode->getOperand(9));
243  Assert(nodeSPMDArray != NULL);
244  llvm::DIArray nodeSPs(nodeSPMDArray);
245  Assert(nodeSPs.getNumElements() == subprograms.getNumElements());
246  for (int i = 0; i < (int)nodeSPs.getNumElements(); ++i)
247  Assert(nodeSPs.getElement(i) == subprograms.getElement(i));
248 
249  // And now we can go and stuff it into the node with some
250  // confidence...
251  llvm::MDNode *replNode =
252  m->diBuilder->getOrCreateArray(llvm::ArrayRef<llvm::Value *>(usedSubprograms));
253  cuNode->replaceOperandWith(9, replNode);
254 #endif
255  }
256  }
257 
258  // Also, erase a bunch of named metadata detrius; for each function
259  // there is sometimes named metadata llvm.dbg.lv.{funcname} that
260  // doesn't seem to be otherwise needed.
261  std::vector<llvm::NamedMDNode *> toErase;
262  llvm::Module::named_metadata_iterator iter = module->named_metadata_begin();
263  for (; iter != module->named_metadata_end(); ++iter) {
264  if (!strncmp(iter->getName().str().c_str(), "llvm.dbg.lv", 11))
265  toErase.push_back(iter);
266  }
267  for (int i = 0; i < (int)toErase.size(); ++i)
268  module->eraseNamedMetadata(toErase[i]);
269 
270  // Wrap up by running the LLVM pass to remove anything left that's
271  // unused.
272  llvm::PassManager pm;
273  pm.add(llvm::createStripDeadDebugInfoPass());
274  pm.run(*module);
275 }

281 Module::Module(const char *fn) {
282  // It's a hack to do this here, but it must be done after the target
283  // information has been set (so e.g. the vector width is known...) In
284  // particular, if we're compiling to multiple targets with different
285  // vector widths, this needs to be redone each time through.
286  InitLLVMUtil(g->ctx, *g->target);
287 
288  filename = fn;
289  errorCount = 0;
290  symbolTable = new SymbolTable;
291  ast = new AST;
292 
293  lDeclareSizeAndPtrIntTypes(symbolTable);
294 
295  module = new llvm::Module(filename ? filename : "<stdin>", *g->ctx);
296  module->setTargetTriple(g->target->GetTripleString());
297 
298  // DataLayout information supposed to be managed in single place in Target class.
299  module->setDataLayout(g->target->getDataLayout()->getStringRepresentation());
300 
301  if (g->generateDebuggingSymbols) {
302  diBuilder = new llvm::DIBuilder(*module);
303 
304  // Let the DIBuilder know that we're starting a new compilation
305  // unit.
306  if (filename == NULL) {
307  // Unfortunately we can't yet call Error() since the global 'm'
308  // variable hasn't been initialized yet.
309  fprintf(stderr, "Can't emit debugging information with no "
310  "source file on disk.\n");
311  ++errorCount;
312  delete diBuilder;
313  diBuilder = NULL;
314  }
315  else {
316  std::string directory, name;
317  GetDirectoryAndFileName(g->currentDirectory, filename, &directory,
318  &name);
319  char producerString[512];
320 #if defined(BUILD_VERSION) && defined (BUILD_DATE)
321  sprintf(producerString, "ispc version %s (build %s on %s)",
322  ISPC_VERSION, BUILD_VERSION, BUILD_DATE);
323 #else
324  sprintf(producerString, "ispc version %s (built on %s)",
325  ISPC_VERSION, __DATE__);
326 #endif
327 #if !defined(LLVM_3_2) && !defined(LLVM_3_3)
328  diCompileUnit =
329 #endif // LLVM_3_4+
330  diBuilder->createCompileUnit(llvm::dwarf::DW_LANG_C99, /* lang */
331  name, /* filename */
332  directory, /* directory */
333  producerString, /* producer */
334  g->opt.level > 0 /* is optimized */,
335  "-g", /* command line args */
336  0 /* run time version */);
337  }
338  }
339  else
340  diBuilder = NULL;
341 }

420 void
421 Module::AddGlobalVariable(const std::string &name, const Type *type, Expr *initExpr,
422  bool isConst, StorageClass storageClass, SourcePos pos) {
423  // These may be NULL due to errors in parsing; just gracefully return
424  // here if so.
425  if (name == "" || type == NULL) {
426  Assert(errorCount > 0);
427  return;
428  }
429 
430  if (symbolTable->LookupFunction(name.c_str())) {
431  Error(pos, "Global variable \"%s\" shadows previously-declared "
432  "function.", name.c_str());
433  return;
434  }
435 
436  if (storageClass == SC_EXTERN_C) {
437  Error(pos, "extern \"C\" qualifier can only be used for "
438  "functions.");
439  return;
440  }
441 
442  if (type->IsVoidType()) {
443  Error(pos, "\"void\" type global variable is illegal.");
444  return;
445  }
446 
447  type = ArrayType::SizeUnsizedArrays(type, initExpr);
448  if (type == NULL)
449  return;
450 
451  const ArrayType *at = CastType<ArrayType>(type);
452  if (at != NULL && at->TotalElementCount() == 0) {
453  Error(pos, "Illegal to declare a global variable with unsized "
454  "array dimensions that aren't set with an initializer "
455  "expression.");
456  return;
457  }
458 
459 #ifdef ISPC_NVPTX_ENABLED
460  if (g->target->getISA() == Target::NVPTX &&
461 #if 0
462  !type->IsConstType() &&
463 #endif
464 #if 1
465  at != NULL &&
466 #endif
467  type->IsVaryingType())
468  {
469  Error(pos, "Global \"varying\" variables are not yet supported in \"nvptx\" target.");
470  return;
471 #if 0
472  int nel = 32; /* warp-size */
473  if (type->IsArrayType())
474  {
475  const ArrayType *at = CastType<ArrayType>(type);
476  /* we must scale # elements by 4, because a thread-block will run 4 warps
477  * or 128 threads.
478  * ***note-to-me***:please define these value (128threads/4warps)
479  * in nvptx-target definition
480  * instead of compile-time constants
481  */
482  nel *= at->GetElementCount();
483  assert (!type->IsSOAType());
484  type = new ArrayType(at->GetElementType()->GetAsUniformType(), nel);
485  }
486  else
487  type = new ArrayType(type->GetAsUniformType(), nel);
488 #endif
489  }
490 #endif /* ISPC_NVPTX_ENABLED */
491 
492  llvm::Type *llvmType = type->LLVMType(g->ctx);
493  if (llvmType == NULL)
494  return;
495 
496  // See if we have an initializer expression for the global. If so,
497  // make sure it's a compile-time constant!
498  llvm::Constant *llvmInitializer = NULL;
499  ConstExpr *constValue = NULL;
500  if (storageClass == SC_EXTERN || storageClass == SC_EXTERN_C) {
501  if (initExpr != NULL)
502  Error(pos, "Initializer can't be provided with \"extern\" "
503  "global variable \"%s\".", name.c_str());
504  }
505  else {
506  if (initExpr != NULL) {
507  initExpr = TypeCheck(initExpr);
508  if (initExpr != NULL) {
509  // We need to make sure the initializer expression is
510  // the same type as the global. (But not if it's an
511  // ExprList; they don't have types per se / can't type
512  // convert themselves anyway.)
513  if (dynamic_cast<ExprList *>(initExpr) == NULL)
514  initExpr = TypeConvertExpr(initExpr, type, "initializer");
515 
516  if (initExpr != NULL) {
517  initExpr = Optimize(initExpr);
518  // Fingers crossed, now let's see if we've got a
519  // constant value..
520  llvmInitializer = initExpr->GetConstant(type);
521 
522  if (llvmInitializer != NULL) {
523  if (type->IsConstType())
524  // Try to get a ConstExpr associated with
525  // the symbol. This dynamic_cast can
526  // validly fail, for example for types like
527  // StructTypes where a ConstExpr can't
528  // represent their values.
529  constValue = dynamic_cast<ConstExpr *>(initExpr);
530  }
531  else
532  Error(initExpr->pos, "Initializer for global variable \"%s\" "
533  "must be a constant.", name.c_str());
534  }
535  }
536  }
537 
538  // If no initializer was provided or if we couldn't get a value
539  // above, initialize it with zeros..
540  if (llvmInitializer == NULL)
541  llvmInitializer = llvm::Constant::getNullValue(llvmType);
542  }
543 
544  Symbol *sym = symbolTable->LookupVariable(name.c_str());
545  llvm::GlobalVariable *oldGV = NULL;
546  if (sym != NULL) {
547  // We've already seen either a declaration or a definition of this
548  // global.
549 
550  // If the type doesn't match with the previous one, issue an error.
551  if (!Type::Equal(sym->type, type) ||
552  (sym->storageClass != SC_EXTERN &&
553  sym->storageClass != SC_EXTERN_C &&
554  sym->storageClass != storageClass)) {
555  Error(pos, "Definition of variable \"%s\" conflicts with "
556  "definition at %s:%d.", name.c_str(),
557  sym->pos.name, sym->pos.first_line);
558  return;
559  }
560 
561  llvm::GlobalVariable *gv =
562  llvm::dyn_cast<llvm::GlobalVariable>(sym->storagePtr);
563  Assert(gv != NULL);
564 
565  // And issue an error if this is a redefinition of a variable
566  if (gv->hasInitializer() &&
567  sym->storageClass != SC_EXTERN && sym->storageClass != SC_EXTERN_C) {
568  Error(pos, "Redefinition of variable \"%s\" is illegal. "
569  "(Previous definition at %s:%d.)", sym->name.c_str(),
570  sym->pos.name, sym->pos.first_line);
571  return;
572  }
573 
574  // Now, we either have a redeclaration of a global, or a definition
575  // of a previously-declared global. First, save the pointer to the
576  // previous llvm::GlobalVariable
577  oldGV = gv;
578  }
579  else {
580  sym = new Symbol(name, pos, type, storageClass);
581  symbolTable->AddVariable(sym);
582  }
583  sym->constValue = constValue;
584 
585  llvm::GlobalValue::LinkageTypes linkage =
586  (sym->storageClass == SC_STATIC) ? llvm::GlobalValue::InternalLinkage :
587  llvm::GlobalValue::ExternalLinkage;
588 
589  // Note that the NULL llvmInitializer is what leads to "extern"
590  // declarations coming up extern and not defining storage (a bit
591  // subtle)...
592  sym->storagePtr = new llvm::GlobalVariable(*module, llvmType, isConst,
593  linkage, llvmInitializer,
594  sym->name.c_str());
595 
596  // Patch up any references to the previous GlobalVariable (e.g. from a
597  // declaration of a global that was later defined.)
598  if (oldGV != NULL) {
599  oldGV->replaceAllUsesWith(sym->storagePtr);
600  oldGV->removeFromParent();
601  sym->storagePtr->setName(sym->name.c_str());
602  }
603 
604  if (diBuilder) {
605  llvm::DIFile file = pos.GetDIFile();
606  llvm::DIGlobalVariable var =
607 #if !defined(LLVM_3_2) && !defined(LLVM_3_3) && !defined(LLVM_3_4) && !defined(LLVM_3_5)// LLVM 3.6+
608  diBuilder->createGlobalVariable(file,
609  name,
610  name,
611  file,
612  pos.first_line,
613  sym->type->GetDIType(file),
614  (sym->storageClass == SC_STATIC),
615  sym->storagePtr);
616 #else
617  diBuilder->createGlobalVariable(name,
618  file,
619  pos.first_line,
620  sym->type->GetDIType(file),
621  (sym->storageClass == SC_STATIC),
622  sym->storagePtr);
623 #endif
624  Assert(var.Verify());
625  }
626 }

631  code.
632 
633  (Note that it's fine for the original struct or a contained struct to
634  be varying, so long as all of its members have bound 'uniform'
635  variability.)
636 
637  This functions returns true and issues an error if are any illegal
638  types are found and returns false otherwise.
639 */
640 static bool
641 lRecursiveCheckValidParamType(const Type *t, bool vectorOk) {
642  const StructType *st = CastType<StructType>(t);
643  if (st != NULL) {
644  for (int i = 0; i < st->GetElementCount(); ++i)
645  if (!lRecursiveCheckValidParamType(st->GetElementType(i),
646  vectorOk))
647  return false;
648  return true;
649  }
650 
651  // Vector types are also not supported, pending ispc properly
652  // supporting the platform ABI. (Pointers to vector types are ok,
653  // though.) (https://github.com/ispc/ispc/issues/363)...
654  if (vectorOk == false && CastType<VectorType>(t) != NULL)
655  return false;
656 
657  const SequentialType *seqt = CastType<SequentialType>(t);
658  if (seqt != NULL)
659  return lRecursiveCheckValidParamType(seqt->GetElementType(), vectorOk);
660 
661  const PointerType *pt = CastType<PointerType>(t);
662  if (pt != NULL) {
663  // Only allow exported uniform pointers
664  // Uniform pointers to varying data, however, are ok.
665  if (pt->IsVaryingType())
666  return false;
667  else
668  return lRecursiveCheckValidParamType(pt->GetBaseType(), true);
669  }
670 
671  if (t->IsVaryingType() && !vectorOk)
672  return false;
673  else
674  return true;
675 }

737 /** We've got a declaration for a function to process. This function does
738  all the work of creating the corresponding llvm::Function instance,
739  adding the symbol for the function to the symbol table and doing
740  various sanity checks. This function returns true upon success and
741  false if any errors were encountered.
742  */
743 void
744 Module::AddFunctionDeclaration(const std::string &name,
745  const FunctionType *functionType,
746  StorageClass storageClass, bool isInline,
747  SourcePos pos) {
748  Assert(functionType != NULL);
749 
750  // If a global variable with the same name has already been declared
751  // issue an error.
752  if (symbolTable->LookupVariable(name.c_str()) != NULL) {
753  Error(pos, "Function \"%s\" shadows previously-declared global variable. "
754  "Ignoring this definition.",
755  name.c_str());
756  return;
757  }
758 
759  std::vector<Symbol *> overloadFuncs;
760  symbolTable->LookupFunction(name.c_str(), &overloadFuncs);
761  if (overloadFuncs.size() > 0) {
762  for (unsigned int i = 0; i < overloadFuncs.size(); ++i) {
763  Symbol *overloadFunc = overloadFuncs[i];
764 
765  const FunctionType *overloadType =
766  CastType<FunctionType>(overloadFunc->type);
767  if (overloadType == NULL) {
768  Assert(m->errorCount == 0);
769  continue;
770  }
771 
772  // Check for a redeclaration of a function with the same name
773  // and type. This also hits when we have previously declared
774  // the function and are about to define it.
775  if (Type::Equal(overloadFunc->type, functionType))
776  return;
777 
778  if (functionType->isExported || overloadType->isExported)
779  Error(pos, "Illegal to provide \"export\" qualifier for "
780  "functions with the same name but different types. "
781  "(Previous function declaration (%s:%d).)",
782  overloadFunc->pos.name, overloadFunc->pos.first_line);
783 
784  // If all of the parameter types match but the return type is
785  // different, return an error--overloading by return type isn't
786  // allowed.
787  const FunctionType *ofType =
788  CastType<FunctionType>(overloadFunc->type);
789  Assert(ofType != NULL);
790  if (ofType->GetNumParameters() == functionType->GetNumParameters()) {
791  int i;
792  for (i = 0; i < functionType->GetNumParameters(); ++i) {
793  if (Type::Equal(ofType->GetParameterType(i),
794  functionType->GetParameterType(i)) == false)
795  break;
796  }
797  if (i == functionType->GetNumParameters()) {
798  std::string thisRetType = functionType->GetReturnTypeString();
799  std::string otherRetType = ofType->GetReturnTypeString();
800  Error(pos, "Illegal to overload function by return "
801  "type only. This function returns \"%s\" while "
802  "previous declaration at %s:%d returns \"%s\".",
803  thisRetType.c_str(), overloadFunc->pos.name,
804  overloadFunc->pos.first_line, otherRetType.c_str());
805  return;
806  }
807  }
808  }
809  }
810 
811  if (storageClass == SC_EXTERN_C) {
812  // Make sure the user hasn't supplied both an 'extern "C"' and a
813  // 'task' qualifier with the function
814  if (functionType->isTask) {
815  Error(pos, "\"task\" qualifier is illegal with C-linkage extern "
816  "function \"%s\". Ignoring this function.", name.c_str());
817  return;
818  }
819 
820  std::vector<Symbol *> funcs;
821  symbolTable->LookupFunction(name.c_str(), &funcs);
822  if (funcs.size() > 0) {
823  if (funcs.size() > 1) {
824  // Multiple functions with this name have already been declared;
825  // can't overload here
826  Error(pos, "Can't overload extern \"C\" function \"%s\"; "
827  "%d functions with the same name have already been declared.",
828  name.c_str(), (int)funcs.size());
829  return;
830  }
831 
832  // One function with the same name has been declared; see if it
833  // has the same type as this one, in which case it's ok.
834  if (Type::Equal(funcs[0]->type, functionType))
835  return;
836  else {
837  Error(pos, "Can't overload extern \"C\" function \"%s\".",
838  name.c_str());
839  return;
840  }
841  }
842  }
843 
844  // Get the LLVM FunctionType
845  bool disableMask = (storageClass == SC_EXTERN_C);
846  llvm::FunctionType *llvmFunctionType =
847  functionType->LLVMFunctionType(g->ctx, disableMask);
848  if (llvmFunctionType == NULL)
849  return;
850 
851  // And create the llvm::Function
852  llvm::GlobalValue::LinkageTypes linkage = (storageClass == SC_STATIC ||
853  isInline) ?
854  llvm::GlobalValue::InternalLinkage : llvm::GlobalValue::ExternalLinkage;
855 
856  std::string functionName = name;
857  if (storageClass != SC_EXTERN_C) {
858  functionName += functionType->Mangle();
859  if (g->mangleFunctionsWithTarget)
860  functionName += g->target->GetISAString();
861  }
862  llvm::Function *function =
863  llvm::Function::Create(llvmFunctionType, linkage, functionName.c_str(),
864  module);
865 
866  // Set function attributes: we never throw exceptions
867  function->setDoesNotThrow();
868  if (storageClass != SC_EXTERN_C &&
869  isInline)
870 #ifdef LLVM_3_2
871  function->addFnAttr(llvm::Attributes::AlwaysInline);
872 #else // LLVM 3.3+
873  function->addFnAttr(llvm::Attribute::AlwaysInline);
874 #endif
875 
876  if (functionType->isTask)
877 #ifdef ISPC_NVPTX_ENABLED
878  /* evghenii: fails function verification when "if" executed in nvptx target */
879  if (g->target->getISA() != Target::NVPTX)
880 #endif /* ISPC_NVPTX_ENABLED */
881  // This also applies transitively to members I think?
882  function->setDoesNotAlias(1);
883 
884  g->target->markFuncWithTargetAttr(function);
885 
886  // Make sure that the return type isn't 'varying' or vector typed if
887  // the function is 'export'ed.
888  if (functionType->isExported &&
889  lRecursiveCheckValidParamType(functionType->GetReturnType(), false) == false)
890  Error(pos, "Illegal to return a \"varying\" or vector type from "
891  "exported function \"%s\"", name.c_str());
892 
893  if (functionType->isTask &&
894  functionType->GetReturnType()->IsVoidType() == false)
895  Error(pos, "Task-qualified functions must have void return type.");
896 
897 #ifdef ISPC_NVPTX_ENABLED
898  if (g->target->getISA() == Target::NVPTX &&
899  Type::Equal(functionType->GetReturnType(), AtomicType::Void) == false &&
900  functionType->isExported)
901  {
902  Error(pos, "Export-qualified functions must have void return type with \"nvptx\" target.");
903  }
904 #endif /* ISPC_NVPTX_ENABLED */
905 
906  if (functionType->isExported || functionType->isExternC)
907  lCheckForStructParameters(functionType, pos);
908 
909  // Loop over all of the arguments; process default values if present
910  // and do other checks and parameter attribute setting.
911  bool seenDefaultArg = false;
912  int nArgs = functionType->GetNumParameters();
913  for (int i = 0; i < nArgs; ++i) {
914  const Type *argType = functionType->GetParameterType(i);
915  const std::string &argName = functionType->GetParameterName(i);
916  Expr *defaultValue = functionType->GetParameterDefault(i);
917  const SourcePos &argPos = functionType->GetParameterSourcePos(i);
918 
919  // If the function is exported, make sure that the parameter
920  // doesn't have any funky stuff going on in it.
921  // JCB nomosoa - Varying is now a-ok.
922  if (functionType->isExported) {
923  lCheckExportedParameterTypes(argType, argName, argPos);
924  }
925 
926 #ifdef ISPC_NVPTX_ENABLED
927  if (functionType->isTask) {
928  lCheckTaskParameterTypes(argType, argName, argPos);
929  }
930 #endif /* ISPC_NVPTX_ENABLED */
931 
932  // ISPC assumes that no pointers alias. (It should be possible to
933  // specify when this is not the case, but this should be the
934  // default.) Set parameter attributes accordingly. (Only for
935  // uniform pointers, since varying pointers are int vectors...)
936  if (!functionType->isTask &&
937  ((CastType<PointerType>(argType) != NULL &&
938  argType->IsUniformType() &&
939  // Exclude SOA argument because it is a pair {struct *, int}
940  // instead of pointer
941  !CastType<PointerType>(argType)->IsSlice())
942  ||
943 
944  CastType<ReferenceType>(argType) != NULL)) {
945 
946  // NOTE: LLVM indexes function parameters starting from 1.
947  // This is unintuitive.
948  function->setDoesNotAlias(i+1);
949 #if 0
950  int align = 4 * RoundUpPow2(g->target->nativeVectorWidth);
951  function->addAttribute(i+1, llvm::Attribute::constructAlignmentFromInt(align));
952 #endif
953  }
954 
955  if (symbolTable->LookupFunction(argName.c_str()))
956  Warning(argPos, "Function parameter \"%s\" shadows a function "
957  "declared in global scope.", argName.c_str());
958 
959  if (defaultValue != NULL)
960  seenDefaultArg = true;
961  else if (seenDefaultArg) {
962  // Once one parameter has provided a default value, then all of
963  // the following ones must have them as well.
964  Error(argPos, "Parameter \"%s\" is missing default: all "
965  "parameters after the first parameter with a default value "
966  "must have default values as well.", argName.c_str());
967  }
968  }
969 
970  // If llvm gave us back a Function * with a different name than the one
971  // we asked for, then there's already a function with that same
972  // (mangled) name in the llvm::Module. In that case, erase the one we
973  // tried to add and just work with the one it already had.
974  if (function->getName() != functionName) {
975  function->eraseFromParent();
976  function = module->getFunction(functionName);
977  }
978 
979  // Finally, we know all is good and we can add the function to the
980  // symbol table
981  Symbol *funSym = new Symbol(name, pos, functionType, storageClass);
982  funSym->function = function;
983  bool ok = symbolTable->AddFunction(funSym);
984  Assert(ok);
985 }

2612 /** Create the dispatch function for an exported ispc function.
2613  This function checks to see which vector ISAs the system the
2614  code is running on supports and calls out to the best available
2615  variant that was generated at compile time.
2616 
2617  @param module Module in which to create the dispatch function.
2618  @param setISAFunc Pointer to the __set_system_isa() function defined
2619  in builtins-dispatch.ll (which is linked into the
2620  given module before we get here.)
2621  @param systemBestISAPtr Pointer to the module-local __system_best_isa
2622  variable, which holds a value of the Target::ISA
2623  enumerant giving the most capable ISA that the
2624  system supports.
2625  @param name Name of the function for which we're generating a
2626  dispatch function
2627  @param funcs Target-specific variants of the exported function.
2628 */
2629 static void
2630 lCreateDispatchFunction(llvm::Module *module, llvm::Function *setISAFunc,
2631  llvm::Value *systemBestISAPtr, const std::string &name,
2632  FunctionTargetVariants &funcs) {
2633  // The llvm::Function pointers in funcs are pointers to functions in
2634  // different llvm::Modules, so we can't call them directly. Therefore,
2635  // we'll start by generating an 'extern' declaration of each one that
2636  // we have in the current module so that we can then call out to that.
2637  llvm::Function *targetFuncs[Target::NUM_ISAS];
2638  llvm::FunctionType *ftypes[Target::NUM_ISAS];
2639 
2640  for (int i = 0; i < Target::NUM_ISAS; ++i) {
2641  if (funcs.func[i] == NULL) {
2642  targetFuncs[i] = NULL;
2643  ftypes[i] = NULL;
2644  continue;
2645  }
2646 
2647  // Grab the type of the function as well. Note that the various
2648  // functions will have different types if they have arguments that
2649  // are pointers to structs, due to the fact that we mangle LLVM
2650  // struct type names with the target vector width. However,
2651  // because we only allow uniform stuff to pass through the
2652  // export'ed function layer, they should all have the same memory
2653  // layout, so this is benign..
2654  // JCB nomosoa - not anymore...
2655  // add a helper to see if this type has any varying thingies?
2656  // might be hard to detect....
2657  // If so, return a new type with the pointers to those replaced
2658  // by i8 *'s.
2659  // if (ftype == NULL)
2660  ftypes[i] = funcs.func[i]->getFunctionType();
2661 
2662  targetFuncs[i] =
2663  llvm::Function::Create(ftypes[i], llvm::GlobalValue::ExternalLinkage,
2664  funcs.func[i]->getName(), module);
2665  }
2666 
2667  // New helper function checks to see if we need to rewrite the
2668  // type for the dispatch function in case of pointers to varyings
2669  llvm::FunctionType *ftype = lGetVaryingDispatchType(funcs);
2670 
2671  bool voidReturn = ftype->getReturnType()->isVoidTy();
2672 
2673  // Now we can emit the definition of the dispatch function..
2674  llvm::Function *dispatchFunc =
2675  llvm::Function::Create(ftype, llvm::GlobalValue::ExternalLinkage,
2676  name.c_str(), module);
2677  llvm::BasicBlock *bblock =
2678  llvm::BasicBlock::Create(*g->ctx, "entry", dispatchFunc);
2679 
2680  // Start by calling out to the function that determines the system's
2681  // ISA and sets __system_best_isa, if it hasn't been set yet.
2682  llvm::CallInst::Create(setISAFunc, "", bblock);
2683 
2684  // Now we can load the system's ISA enuemrant
2685  llvm::Value *systemISA =
2686  new llvm::LoadInst(systemBestISAPtr, "system_isa", bblock);
2687 
2688  // Now emit code that works backwards though the available variants of
2689  // the function. We'll call out to the first one we find that will run
2690  // successfully on the system the code is running on. In working
2691  // through the candidate ISAs here backward, we're taking advantage of
2692  // the expectation that they are ordered in the Target::ISA enumerant
2693  // from least to most capable.
2694  for (int i = Target::NUM_ISAS-1; i >= 0; --i) {
2695  if (targetFuncs[i] == NULL)
2696  continue;
2697 
2698  // Emit code to see if the system can run the current candidate
2699  // variant successfully--"is the system's ISA enuemrant value >=
2700  // the enumerant value of the current candidate?"
2701  llvm::Value *ok =
2702  llvm::CmpInst::Create(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SGE,
2703  systemISA, LLVMInt32(i), "isa_ok", bblock);
2704  llvm::BasicBlock *callBBlock =
2705  llvm::BasicBlock::Create(*g->ctx, "do_call", dispatchFunc);
2706  llvm::BasicBlock *nextBBlock =
2707  llvm::BasicBlock::Create(*g->ctx, "next_try", dispatchFunc);
2708  llvm::BranchInst::Create(callBBlock, nextBBlock, ok, bblock);
2709 
2710  // Emit the code to make the call call in callBBlock.
2711  // Just pass through all of the args from the dispatch function to
2712  // the target-specific function.
2713  std::vector<llvm::Value *> args;
2714  llvm::Function::arg_iterator argIter = dispatchFunc->arg_begin();
2715  llvm::Function::arg_iterator targsIter = targetFuncs[i]->arg_begin();
2716  for (; argIter != dispatchFunc->arg_end(); ++argIter, ++targsIter) {
2717  // Check to see if we rewrote any types in the dispatch function.
2718  // If so, create bitcasts for the appropriate pointer types.
2719  if (argIter->getType() == targsIter->getType()) {
2720  args.push_back(argIter);
2721  }
2722  else {
2723  llvm::CastInst *argCast =
2724  llvm::CastInst::CreatePointerCast(argIter, targsIter->getType(),
2725  "dpatch_arg_bitcast", callBBlock);
2726  args.push_back(argCast);
2727  }
2728 
2729  }
2730  if (voidReturn) {
2731  llvm::CallInst::Create(targetFuncs[i], args, "", callBBlock);
2732  llvm::ReturnInst::Create(*g->ctx, callBBlock);
2733  }
2734  else {
2735  llvm::Value *retValue =
2736  llvm::CallInst::Create(targetFuncs[i], args, "ret_value",
2737  callBBlock);
2738  llvm::ReturnInst::Create(*g->ctx, retValue, callBBlock);
2739  }
2740 
2741  // Otherwise we'll go on to the next candidate and see about that
2742  // one...
2743  bblock = nextBBlock;
2744  }
2745 
2746  // We couldn't find a match that the current system was capable of
2747  // running. We'll call abort(); this is a bit of a blunt hammer--it
2748  // might be preferable to call a user-supplied callback--ISPCError(...)
2749  // or some such, but we don't want to start imposing too much of a
2750  // runtime library requirement either...
2751  llvm::Function *abortFunc = module->getFunction("abort");
2752  Assert(abortFunc);
2753  llvm::CallInst::Create(abortFunc, "", bblock);
2754 
2755  // Return an undef value from the function here; we won't get to this
2756  // point at runtime, but LLVM needs all of the basic blocks to be
2757  // terminated...
2758  if (voidReturn)
2759  llvm::ReturnInst::Create(*g->ctx, bblock);
2760  else {
2761  llvm::Value *undefRet = llvm::UndefValue::get(ftype->getReturnType());
2762  llvm::ReturnInst::Create(*g->ctx, undefRet, bblock);
2763  }
2764 }
