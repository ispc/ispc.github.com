140 /** After compilation completes, there's often a lot of extra debugging
141  metadata left around that isn't needed any more--for example, for
142  static functions that weren't actually used, function information for
143  functions that were inlined, etc. This function takes a llvm::Module
144  and tries to strip out all of this extra stuff.
145  */
146 static void
147 lStripUnusedDebugInfo(llvm::Module *module) {
148  if (g->generateDebuggingSymbols == false)
149  return;
150 
151  // loop over the compile units that contributed to the final module
152  if (llvm::NamedMDNode *cuNodes = module->getNamedMetadata("llvm.dbg.cu")) {
153  for (unsigned i = 0, ie = cuNodes->getNumOperands(); i != ie; ++i) {
154  llvm::MDNode *cuNode = cuNodes->getOperand(i);
155  llvm::DICompileUnit cu(cuNode);
156  llvm::DIArray subprograms = cu.getSubprograms();
157  std::vector<llvm::Value *> usedSubprograms;
158 
159  if (subprograms.getNumElements() == 0)
160  continue;
161 
162  // And now loop over the subprograms inside each compile unit.
163  for (unsigned j = 0, je = subprograms.getNumElements(); j != je; ++j) {
164  llvm::MDNode *spNode =
165  llvm::dyn_cast<llvm::MDNode>(subprograms->getOperand(j));
166  Assert(spNode != NULL);
167  llvm::DISubprogram sp(spNode);
168 
169  // Get the name of the subprogram. Start with the mangled
170  // name; if that's empty then we have an export'ed
171  // function, so grab the unmangled name in that case.
172  std::string name = sp.getLinkageName();
173  if (name == "")
174  name = sp.getName();
175 
176  // Does the llvm::Function for this function survive in the
177  // module?
178  if (module->getFunction(name) != NULL)
179  usedSubprograms.push_back(sp);
180  }
181 
182  Debug(SourcePos(), "%d / %d functions left in module with debug "
183  "info.", (int)usedSubprograms.size(),
184  (int)subprograms.getNumElements());
185 
186  // We'd now like to replace the array of subprograms in the
187  // compile unit with only the ones that actually have function
188  // definitions present. Unfortunately, llvm::DICompileUnit
189  // doesn't provide a method to set the subprograms. Therefore,
190  // we end up needing to directly stuff a new array into the
191  // appropriate slot (number 12) in the MDNode for the compile
192  // unit.
193  //
194  // Because this is all so hard-coded and would break if the
195  // debugging metadata organization on the LLVM side changed,
196  // here is a bunch of asserting to make sure that element 12 of
197  // the compile unit's MDNode has the subprograms array....
198  //
199  // Update: This is not an approved way of working with debug info
200  // metadata. It's not supposed to be deleted. But in out use-case
201  // it's quite useful thing, as we link in bunch of unnecessary
202  // stuff and remove it later on. Removing it is useful, as it
203  // reduces size of the binary significantly (manyfold for small
204  // programs).
205 #if defined(LLVM_3_1) || defined(LLVM_3_2)
206  llvm::MDNode *nodeSPMD =
207  llvm::dyn_cast<llvm::MDNode>(cuNode->getOperand(12));
208  Assert(nodeSPMD != NULL);
209  llvm::MDNode *nodeSPMDArray =
210  llvm::dyn_cast<llvm::MDNode>(nodeSPMD->getOperand(0));
211  llvm::DIArray nodeSPs(nodeSPMDArray);
212  Assert(nodeSPs.getNumElements() == subprograms.getNumElements());
213  for (int i = 0; i < (int)nodeSPs.getNumElements(); ++i)
214  Assert(nodeSPs.getElement(i) == subprograms.getElement(i));
215 
216  // And now we can go and stuff it into the node with some
217  // confidence...
218  llvm::Value *usedSubprogramsArray =
219  m->diBuilder->getOrCreateArray(llvm::ArrayRef<llvm::Value *>(usedSubprograms));
220  llvm::MDNode *replNode =
221  llvm::MDNode::get(*g->ctx, llvm::ArrayRef<llvm::Value *>(usedSubprogramsArray));
222  cuNode->replaceOperandWith(12, replNode);
223 #else // LLVM 3.3+
224  llvm::MDNode *nodeSPMDArray =
225  llvm::dyn_cast<llvm::MDNode>(cuNode->getOperand(9));
226  Assert(nodeSPMDArray != NULL);
227  llvm::DIArray nodeSPs(nodeSPMDArray);
228  Assert(nodeSPs.getNumElements() == subprograms.getNumElements());
229  for (int i = 0; i < (int)nodeSPs.getNumElements(); ++i)
230  Assert(nodeSPs.getElement(i) == subprograms.getElement(i));
231 
232  // And now we can go and stuff it into the node with some
233  // confidence...
234  llvm::MDNode *replNode =
235  m->diBuilder->getOrCreateArray(llvm::ArrayRef<llvm::Value *>(usedSubprograms));
236  cuNode->replaceOperandWith(9, replNode);
237 #endif
238  }
239  }
240 
241  // Also, erase a bunch of named metadata detrius; for each function
242  // there is sometimes named metadata llvm.dbg.lv.{funcname} that
243  // doesn't seem to be otherwise needed.
244  std::vector<llvm::NamedMDNode *> toErase;
245  llvm::Module::named_metadata_iterator iter = module->named_metadata_begin();
246  for (; iter != module->named_metadata_end(); ++iter) {
247  if (!strncmp(iter->getName().str().c_str(), "llvm.dbg.lv", 11))
248  toErase.push_back(iter);
249  }
250  for (int i = 0; i < (int)toErase.size(); ++i)
251  module->eraseNamedMetadata(toErase[i]);
252 
253  // Wrap up by running the LLVM pass to remove anything left that's
254  // unused.
255  llvm::PassManager pm;
256  pm.add(llvm::createStripDeadDebugInfoPass());
257  pm.run(*module);
258 }

264 Module::Module(const char *fn) {
265  // It's a hack to do this here, but it must be done after the target
266  // information has been set (so e.g. the vector width is known...) In
267  // particular, if we're compiling to multiple targets with different
268  // vector widths, this needs to be redone each time through.
269  InitLLVMUtil(g->ctx, *g->target);
270 
271  filename = fn;
272  errorCount = 0;
273  symbolTable = new SymbolTable;
274  ast = new AST;
275 
276  lDeclareSizeAndPtrIntTypes(symbolTable);
277 
278  module = new llvm::Module(filename ? filename : "<stdin>", *g->ctx);
279  module->setTargetTriple(g->target->GetTripleString());
280 
281  // DataLayout information supposed to be managed in single place in Target class.
282  module->setDataLayout(g->target->getDataLayout()->getStringRepresentation());
283 
284  if (g->generateDebuggingSymbols) {
285  diBuilder = new llvm::DIBuilder(*module);
286 
287  // Let the DIBuilder know that we're starting a new compilation
288  // unit.
289  if (filename == NULL) {
290  // Unfortunately we can't yet call Error() since the global 'm'
291  // variable hasn't been initialized yet.
292  fprintf(stderr, "Can't emit debugging information with no "
293  "source file on disk.\n");
294  ++errorCount;
295  delete diBuilder;
296  diBuilder = NULL;
297  }
298  else {
299  std::string directory, name;
300  GetDirectoryAndFileName(g->currentDirectory, filename, &directory,
301  &name);
302  char producerString[512];
303 #if defined(BUILD_VERSION) && defined (BUILD_DATE)
304  sprintf(producerString, "ispc version %s (build %s on %s)",
305  ISPC_VERSION, BUILD_VERSION, BUILD_DATE);
306 #else
307  sprintf(producerString, "ispc version %s (built on %s)",
308  ISPC_VERSION, __DATE__);
309 #endif
310  diBuilder->createCompileUnit(llvm::dwarf::DW_LANG_C99, /* lang */
311  name, /* filename */
312  directory, /* directory */
313  producerString, /* producer */
314  g->opt.level > 0 /* is optimized */,
315  "-g", /* command line args */
316  0 /* run time version */);
317  }
318  }
319  else
320  diBuilder = NULL;
321 }

400 void
401 Module::AddGlobalVariable(const std::string &name, const Type *type, Expr *initExpr,
402  bool isConst, StorageClass storageClass, SourcePos pos) {
403  // These may be NULL due to errors in parsing; just gracefully return
404  // here if so.
405  if (name == "" || type == NULL) {
406  Assert(errorCount > 0);
407  return;
408  }
409 
410  if (symbolTable->LookupFunction(name.c_str())) {
411  Error(pos, "Global variable \"%s\" shadows previously-declared "
412  "function.", name.c_str());
413  return;
414  }
415 
416  if (storageClass == SC_EXTERN_C) {
417  Error(pos, "extern \"C\" qualifier can only be used for "
418  "functions.");
419  return;
420  }
421 
422  if (Type::Equal(type, AtomicType::Void)) {
423  Error(pos, "\"void\" type global variable is illegal.");
424  return;
425  }
426 
427  type = ArrayType::SizeUnsizedArrays(type, initExpr);
428  if (type == NULL)
429  return;
430 
431  const ArrayType *at = CastType<ArrayType>(type);
432  if (at != NULL && at->TotalElementCount() == 0) {
433  Error(pos, "Illegal to declare a global variable with unsized "
434  "array dimensions that aren't set with an initializer "
435  "expression.");
436  return;
437  }
438 
439  llvm::Type *llvmType = type->LLVMType(g->ctx);
440  if (llvmType == NULL)
441  return;
442 
443  // See if we have an initializer expression for the global. If so,
444  // make sure it's a compile-time constant!
445  llvm::Constant *llvmInitializer = NULL;
446  ConstExpr *constValue = NULL;
447  if (storageClass == SC_EXTERN || storageClass == SC_EXTERN_C) {
448  if (initExpr != NULL)
449  Error(pos, "Initializer can't be provided with \"extern\" "
450  "global variable \"%s\".", name.c_str());
451  }
452  else {
453  if (initExpr != NULL) {
454  initExpr = TypeCheck(initExpr);
455  if (initExpr != NULL) {
456  // We need to make sure the initializer expression is
457  // the same type as the global. (But not if it's an
458  // ExprList; they don't have types per se / can't type
459  // convert themselves anyway.)
460  if (dynamic_cast<ExprList *>(initExpr) == NULL)
461  initExpr = TypeConvertExpr(initExpr, type, "initializer");
462 
463  if (initExpr != NULL) {
464  initExpr = Optimize(initExpr);
465  // Fingers crossed, now let's see if we've got a
466  // constant value..
467  llvmInitializer = initExpr->GetConstant(type);
468 
469  if (llvmInitializer != NULL) {
470  if (type->IsConstType())
471  // Try to get a ConstExpr associated with
472  // the symbol. This dynamic_cast can
473  // validly fail, for example for types like
474  // StructTypes where a ConstExpr can't
475  // represent their values.
476  constValue = dynamic_cast<ConstExpr *>(initExpr);
477  }
478  else
479  Error(initExpr->pos, "Initializer for global variable \"%s\" "
480  "must be a constant.", name.c_str());
481  }
482  }
483  }
484 
485  // If no initializer was provided or if we couldn't get a value
486  // above, initialize it with zeros..
487  if (llvmInitializer == NULL)
488  llvmInitializer = llvm::Constant::getNullValue(llvmType);
489  }
490 
491  Symbol *sym = symbolTable->LookupVariable(name.c_str());
492  llvm::GlobalVariable *oldGV = NULL;
493  if (sym != NULL) {
494  // We've already seen either a declaration or a definition of this
495  // global.
496 
497  // If the type doesn't match with the previous one, issue an error.
498  if (!Type::Equal(sym->type, type) ||
499  (sym->storageClass != SC_EXTERN &&
500  sym->storageClass != SC_EXTERN_C &&
501  sym->storageClass != storageClass)) {
502  Error(pos, "Definition of variable \"%s\" conflicts with "
503  "definition at %s:%d.", name.c_str(),
504  sym->pos.name, sym->pos.first_line);
505  return;
506  }
507 
508  llvm::GlobalVariable *gv =
509  llvm::dyn_cast<llvm::GlobalVariable>(sym->storagePtr);
510  Assert(gv != NULL);
511 
512  // And issue an error if this is a redefinition of a variable
513  if (gv->hasInitializer() &&
514  sym->storageClass != SC_EXTERN && sym->storageClass != SC_EXTERN_C) {
515  Error(pos, "Redefinition of variable \"%s\" is illegal. "
516  "(Previous definition at %s:%d.)", sym->name.c_str(),
517  sym->pos.name, sym->pos.first_line);
518  return;
519  }
520 
521  // Now, we either have a redeclaration of a global, or a definition
522  // of a previously-declared global. First, save the pointer to the
523  // previous llvm::GlobalVariable
524  oldGV = gv;
525  }
526  else {
527  sym = new Symbol(name, pos, type, storageClass);
528  symbolTable->AddVariable(sym);
529  }
530  sym->constValue = constValue;
531 
532  llvm::GlobalValue::LinkageTypes linkage =
533  (sym->storageClass == SC_STATIC) ? llvm::GlobalValue::InternalLinkage :
534  llvm::GlobalValue::ExternalLinkage;
535 
536  // Note that the NULL llvmInitializer is what leads to "extern"
537  // declarations coming up extern and not defining storage (a bit
538  // subtle)...
539  sym->storagePtr = new llvm::GlobalVariable(*module, llvmType, isConst,
540  linkage, llvmInitializer,
541  sym->name.c_str());
542 
543  // Patch up any references to the previous GlobalVariable (e.g. from a
544  // declaration of a global that was later defined.)
545  if (oldGV != NULL) {
546  oldGV->replaceAllUsesWith(sym->storagePtr);
547  oldGV->removeFromParent();
548  sym->storagePtr->setName(sym->name.c_str());
549  }
550 
551  if (diBuilder) {
552  llvm::DIFile file = pos.GetDIFile();
553  llvm::DIGlobalVariable var =
554  diBuilder->createGlobalVariable(name,
555  file,
556  pos.first_line,
557  sym->type->GetDIType(file),
558  (sym->storageClass == SC_STATIC),
559  sym->storagePtr);
560  Assert(var.Verify());
561  }
562 }

565 /** Given an arbitrary type, see if it or any of the leaf types contained
566  in it has a type that's illegal to have exported to C/C++
567  code--specifically, that it has a varying value in memory, or a pointer
568  to SOA data (which has a different representation than a regular
569  pointer.
570 
571  (Note that it's fine for the original struct or a contained struct to
572  be varying, so long as all of its members have bound 'uniform'
573  variability.)
574 
575  This functions returns true and issues an error if are any illegal
576  types are found and returns false otherwise.
577 */
578 static bool
579 lRecursiveCheckValidParamType(const Type *t, bool vectorOk) {
580  const StructType *st = CastType<StructType>(t);
581  if (st != NULL) {
582  for (int i = 0; i < st->GetElementCount(); ++i)
583  if (!lRecursiveCheckValidParamType(st->GetElementType(i),
584  vectorOk))
585  return false;
586  return true;
587  }
588 
589  // Vector types are also not supported, pending ispc properly
590  // supporting the platform ABI. (Pointers to vector types are ok,
591  // though.) (https://github.com/ispc/ispc/issues/363)...
592  if (vectorOk == false && CastType<VectorType>(t) != NULL)
593  return false;
594 
595  const SequentialType *seqt = CastType<SequentialType>(t);
596  if (seqt != NULL)
597  return lRecursiveCheckValidParamType(seqt->GetElementType(), vectorOk);
598 
599  const PointerType *pt = CastType<PointerType>(t);
600  if (pt != NULL) {
601  if (pt->IsSlice() || pt->IsVaryingType())
602  return false;
603  return lRecursiveCheckValidParamType(pt->GetBaseType(), true);
604  }
605 
606  if (t->IsVaryingType())
607  return false;
608  else
609  return true;
610 }

656 /** We've got a declaration for a function to process. This function does
657  all the work of creating the corresponding llvm::Function instance,
658  adding the symbol for the function to the symbol table and doing
659  various sanity checks. This function returns true upon success and
660  false if any errors were encountered.
661  */
662 void
663 Module::AddFunctionDeclaration(const std::string &name,
664  const FunctionType *functionType,
665  StorageClass storageClass, bool isInline,
666  SourcePos pos) {
667  Assert(functionType != NULL);
668 
669  // If a global variable with the same name has already been declared
670  // issue an error.
671  if (symbolTable->LookupVariable(name.c_str()) != NULL) {
672  Error(pos, "Function \"%s\" shadows previously-declared global variable. "
673  "Ignoring this definition.",
674  name.c_str());
675  return;
676  }
677 
678  std::vector<Symbol *> overloadFuncs;
679  symbolTable->LookupFunction(name.c_str(), &overloadFuncs);
680  if (overloadFuncs.size() > 0) {
681  for (unsigned int i = 0; i < overloadFuncs.size(); ++i) {
682  Symbol *overloadFunc = overloadFuncs[i];
683 
684  const FunctionType *overloadType =
685  CastType<FunctionType>(overloadFunc->type);
686  if (overloadType == NULL) {
687  Assert(m->errorCount == 0);
688  continue;
689  }
690 
691  // Check for a redeclaration of a function with the same name
692  // and type. This also hits when we have previously declared
693  // the function and are about to define it.
694  if (Type::Equal(overloadFunc->type, functionType))
695  return;
696 
697  if (functionType->isExported || overloadType->isExported)
698  Error(pos, "Illegal to provide \"export\" qualifier for "
699  "functions with the same name but different types. "
700  "(Previous function declaration (%s:%d).)",
701  overloadFunc->pos.name, overloadFunc->pos.first_line);
702 
703  // If all of the parameter types match but the return type is
704  // different, return an error--overloading by return type isn't
705  // allowed.
706  const FunctionType *ofType =
707  CastType<FunctionType>(overloadFunc->type);
708  Assert(ofType != NULL);
709  if (ofType->GetNumParameters() == functionType->GetNumParameters()) {
710  int i;
711  for (i = 0; i < functionType->GetNumParameters(); ++i) {
712  if (Type::Equal(ofType->GetParameterType(i),
713  functionType->GetParameterType(i)) == false)
714  break;
715  }
716  if (i == functionType->GetNumParameters()) {
717  std::string thisRetType = functionType->GetReturnTypeString();
718  std::string otherRetType = ofType->GetReturnTypeString();
719  Error(pos, "Illegal to overload function by return "
720  "type only. This function returns \"%s\" while "
721  "previous declaration at %s:%d returns \"%s\".",
722  thisRetType.c_str(), overloadFunc->pos.name,
723  overloadFunc->pos.first_line, otherRetType.c_str());
724  return;
725  }
726  }
727  }
728  }
729 
730  if (storageClass == SC_EXTERN_C) {
731  // Make sure the user hasn't supplied both an 'extern "C"' and a
732  // 'task' qualifier with the function
733  if (functionType->isTask) {
734  Error(pos, "\"task\" qualifier is illegal with C-linkage extern "
735  "function \"%s\". Ignoring this function.", name.c_str());
736  return;
737  }
738 
739  std::vector<Symbol *> funcs;
740  symbolTable->LookupFunction(name.c_str(), &funcs);
741  if (funcs.size() > 0) {
742  if (funcs.size() > 1) {
743  // Multiple functions with this name have already been declared;
744  // can't overload here
745  Error(pos, "Can't overload extern \"C\" function \"%s\"; "
746  "%d functions with the same name have already been declared.",
747  name.c_str(), (int)funcs.size());
748  return;
749  }
750 
751  // One function with the same name has been declared; see if it
752  // has the same type as this one, in which case it's ok.
753  if (Type::Equal(funcs[0]->type, functionType))
754  return;
755  else {
756  Error(pos, "Can't overload extern \"C\" function \"%s\".",
757  name.c_str());
758  return;
759  }
760  }
761  }
762 
763  // Get the LLVM FunctionType
764  bool disableMask = (storageClass == SC_EXTERN_C);
765  llvm::FunctionType *llvmFunctionType =
766  functionType->LLVMFunctionType(g->ctx, disableMask);
767  if (llvmFunctionType == NULL)
768  return;
769 
770  // And create the llvm::Function
771  llvm::GlobalValue::LinkageTypes linkage = (storageClass == SC_STATIC ||
772  isInline) ?
773  llvm::GlobalValue::InternalLinkage : llvm::GlobalValue::ExternalLinkage;
774 
775  std::string functionName = name;
776  if (storageClass != SC_EXTERN_C) {
777  functionName += functionType->Mangle();
778  if (g->mangleFunctionsWithTarget)
779  functionName += g->target->GetISAString();
780  }
781  llvm::Function *function =
782  llvm::Function::Create(llvmFunctionType, linkage, functionName.c_str(),
783  module);
784 
785  // Set function attributes: we never throw exceptions
786  function->setDoesNotThrow();
787  if (storageClass != SC_EXTERN_C &&
788  !g->generateDebuggingSymbols &&
789  isInline)
790 #ifdef LLVM_3_2
791  function->addFnAttr(llvm::Attributes::AlwaysInline);
792 #else // LLVM 3.1 and 3.3+
793  function->addFnAttr(llvm::Attribute::AlwaysInline);
794 #endif
795  if (functionType->isTask)
796  // This also applies transitively to members I think?
797 #if defined(LLVM_3_1)
798  function->setDoesNotAlias(1, true);
799 #else // LLVM 3.2+
800  function->setDoesNotAlias(1);
801 #endif
802 
803  g->target->markFuncWithTargetAttr(function);
804 
805  // Make sure that the return type isn't 'varying' or vector typed if
806  // the function is 'export'ed.
807  if (functionType->isExported &&
808  lRecursiveCheckValidParamType(functionType->GetReturnType(), false) == false)
809  Error(pos, "Illegal to return a \"varying\" or vector type from "
810  "exported function \"%s\"", name.c_str());
811 
812  if (functionType->isTask &&
813  Type::Equal(functionType->GetReturnType(), AtomicType::Void) == false)
814  Error(pos, "Task-qualified functions must have void return type.");
815 
816  if (functionType->isExported || functionType->isExternC)
817  lCheckForStructParameters(functionType, pos);
818 
819  // Loop over all of the arguments; process default values if present
820  // and do other checks and parameter attribute setting.
821  bool seenDefaultArg = false;
822  int nArgs = functionType->GetNumParameters();
823  for (int i = 0; i < nArgs; ++i) {
824  const Type *argType = functionType->GetParameterType(i);
825  const std::string &argName = functionType->GetParameterName(i);
826  Expr *defaultValue = functionType->GetParameterDefault(i);
827  const SourcePos &argPos = functionType->GetParameterSourcePos(i);
828 
829  // If the function is exported, make sure that the parameter
830  // doesn't have any varying stuff going on in it.
831  if (functionType->isExported)
832  lCheckExportedParameterTypes(argType, argName, argPos);
833 
834  // ISPC assumes that no pointers alias. (It should be possible to
835  // specify when this is not the case, but this should be the
836  // default.) Set parameter attributes accordingly. (Only for
837  // uniform pointers, since varying pointers are int vectors...)
838  if (!functionType->isTask &&
839  ((CastType<PointerType>(argType) != NULL &&
840  argType->IsUniformType() &&
841  // Exclude SOA argument because it is a pair {struct *, int}
842  // instead of pointer
843  !CastType<PointerType>(argType)->IsSlice())
844  ||
845 
846  CastType<ReferenceType>(argType) != NULL)) {
847 
848  // NOTE: LLVM indexes function parameters starting from 1.
849  // This is unintuitive.
850 #if defined(LLVM_3_1)
851  function->setDoesNotAlias(i+1, true);
852 #else
853  function->setDoesNotAlias(i+1);
854 #endif
855 
856 #if 0
857  int align = 4 * RoundUpPow2(g->target->nativeVectorWidth);
858  function->addAttribute(i+1, llvm::Attribute::constructAlignmentFromInt(align));
859 #endif
860  }
861 
862  if (symbolTable->LookupFunction(argName.c_str()))
863  Warning(argPos, "Function parameter \"%s\" shadows a function "
864  "declared in global scope.", argName.c_str());
865 
866  if (defaultValue != NULL)
867  seenDefaultArg = true;
868  else if (seenDefaultArg) {
869  // Once one parameter has provided a default value, then all of
870  // the following ones must have them as well.
871  Error(argPos, "Parameter \"%s\" is missing default: all "
872  "parameters after the first parameter with a default value "
873  "must have default values as well.", argName.c_str());
874  }
875  }
876 
877  // If llvm gave us back a Function * with a different name than the one
878  // we asked for, then there's already a function with that same
879  // (mangled) name in the llvm::Module. In that case, erase the one we
880  // tried to add and just work with the one it already had.
881  if (function->getName() != functionName) {
882  function->eraseFromParent();
883  function = module->getFunction(functionName);
884  }
885 
886  // Finally, we know all is good and we can add the function to the
887  // symbol table
888  Symbol *funSym = new Symbol(name, pos, functionType, storageClass);
889  funSym->function = function;
890  bool ok = symbolTable->AddFunction(funSym);
891  Assert(ok);
892 }

1122 lEmitStructDecl(const StructType *st, std::vector<const StructType *> *emittedStructs,

1140  emittedStructs->push_back(st);

1152  const Type *type = st->GetElementType(i)->GetAsNonConstType();

1252 /** Add the given type to the vector, if that type isn't already in there.
1253  */

1255 lAddTypeIfNew(const Type *type, std::vector<const T *> *exportedTypes) {
1256  type = type->GetAsNonConstType();
1257

1261  if (Type::Equal((*exportedTypes)[i], type))

1270 /** Given an arbitrary type that appears in the app/ispc interface, add it
1271  to an appropriate vector if it is a struct, enum, or short vector type.
1272  Then, if it's a struct, recursively process its members to do the same.
1273  */
1274 static void
1275 lGetExportedTypes(const Type *type,
1276  std::vector<const StructType *> *exportedStructTypes,
1277  std::vector<const EnumType *> *exportedEnumTypes,
1278  std::vector<const VectorType *> *exportedVectorTypes) {
1279  const ArrayType *arrayType = CastType<ArrayType>(type);
1280  const StructType *structType = CastType<StructType>(type);
1281 
1282  if (CastType<ReferenceType>(type) != NULL)
1283  lGetExportedTypes(type->GetReferenceTarget(), exportedStructTypes,
1284  exportedEnumTypes, exportedVectorTypes);
1285  else if (CastType<PointerType>(type) != NULL)
1286  lGetExportedTypes(type->GetBaseType(), exportedStructTypes,
1287  exportedEnumTypes, exportedVectorTypes);
1288  else if (arrayType != NULL)
1289  lGetExportedTypes(arrayType->GetElementType(), exportedStructTypes,
1290  exportedEnumTypes, exportedVectorTypes);
1291  else if (structType != NULL) {
1292  lAddTypeIfNew(type, exportedStructTypes);
1293  for (int i = 0; i < structType->GetElementCount(); ++i)
1294  lGetExportedTypes(structType->GetElementType(i), exportedStructTypes,
1295  exportedEnumTypes, exportedVectorTypes);
1296  }
1297  else if (CastType<UndefinedStructType>(type) != NULL)
1298  // do nothing
1299  ;
1300  else if (CastType<EnumType>(type) != NULL)
1301  lAddTypeIfNew(type, exportedEnumTypes);
1302  else if (CastType<VectorType>(type) != NULL)
1303  lAddTypeIfNew(type, exportedVectorTypes);
1304  else
1305  Assert(CastType<AtomicType>(type) != NULL);
1306 }

2112  code is running on supports and calls out to the best available
2113  variant that was generated at compile time.

2118  given module before we get here.)

2121  enumerant giving the most capable ISA that the
2122  system supports.
2123  @param name Name of the function for which we're generating a
2124  dispatch function
2125  @param funcs Target-specific variants of the exported function.
2126 */
2127 static void
2128 lCreateDispatchFunction(llvm::Module *module, llvm::Function *setISAFunc,
2129  llvm::Value *systemBestISAPtr, const std::string &name,
2130  FunctionTargetVariants &funcs) {
2131  // The llvm::Function pointers in funcs are pointers to functions in
2132  // different llvm::Modules, so we can't call them directly. Therefore,
2133  // we'll start by generating an 'extern' declaration of each one that
2134  // we have in the current module so that we can then call out to that.
2135  llvm::Function *targetFuncs[Target::NUM_ISAS];
2136  llvm::FunctionType *ftype = NULL;
2137 
2138  for (int i = 0; i < Target::NUM_ISAS; ++i) {
2139  if (funcs.func[i] == NULL) {
2140  targetFuncs[i] = NULL;
2141  continue;
2142  }
2143 
2144  // Grab the type of the function as well. Note that the various
2145  // functions will have different types if they have arguments that
2146  // are pointers to structs, due to the fact that we mangle LLVM
2147  // struct type names with the target vector width. However,
2148  // because we only allow uniform stuff to pass through the
2149  // export'ed function layer, they should all have the same memory
2150  // layout, so this is benign..
2151  if (ftype == NULL)
2152  ftype = funcs.func[i]->getFunctionType();
2153 
2154  targetFuncs[i] =
2155  llvm::Function::Create(ftype, llvm::GlobalValue::ExternalLinkage,
2156  funcs.func[i]->getName(), module);
2157  }
2158 
2159  bool voidReturn = ftype->getReturnType()->isVoidTy();
2160

2162  llvm::Function *dispatchFunc =
2163  llvm::Function::Create(ftype, llvm::GlobalValue::ExternalLinkage,
2164  name.c_str(), module);
2165  llvm::BasicBlock *bblock =
2166  llvm::BasicBlock::Create(*g->ctx, "entry", dispatchFunc);
2167 
2168  // Start by calling out to the function that determines the system's
2169  // ISA and sets __system_best_isa, if it hasn't been set yet.
2170  llvm::CallInst::Create(setISAFunc, "", bblock);
2171 
2172  // Now we can load the system's ISA enuemrant
2173  llvm::Value *systemISA =
2174  new llvm::LoadInst(systemBestISAPtr, "system_isa", bblock);
2175 
2176  // Now emit code that works backwards though the available variants of
2177  // the function. We'll call out to the first one we find that will run
2178  // successfully on the system the code is running on. In working
2179  // through the candidate ISAs here backward, we're taking advantage of
2180  // the expectation that they are ordered in the Target::ISA enumerant
2181  // from least to most capable.
2182  for (int i = Target::NUM_ISAS-1; i >= 0; --i) {
2183  if (targetFuncs[i] == NULL)
2184  continue;

2187  // variant successfully--"is the system's ISA enuemrant value >=

2189  llvm::Value *ok =
2190  llvm::CmpInst::Create(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SGE,
2191  systemISA, LLVMInt32(i), "isa_ok", bblock);

2193  llvm::BasicBlock::Create(*g->ctx, "do_call", dispatchFunc);
2194  llvm::BasicBlock *nextBBlock =
2195  llvm::BasicBlock::Create(*g->ctx, "next_try", dispatchFunc);
2196  llvm::BranchInst::Create(callBBlock, nextBBlock, ok, bblock);
2197

2199  // Just pass through all of the args from the dispatch function to
2200  // the target-specific function.
2201  std::vector<llvm::Value *> args;
2202  llvm::Function::arg_iterator argIter = dispatchFunc->arg_begin();
2203  for (; argIter != dispatchFunc->arg_end(); ++argIter)
2204  args.push_back(argIter);
2205  if (voidReturn) {
2206  llvm::CallInst::Create(targetFuncs[i], args, "", callBBlock);
2207  llvm::ReturnInst::Create(*g->ctx, callBBlock);
2208  }
2209  else {
2210  llvm::Value *retValue =
2211  llvm::CallInst::Create(targetFuncs[i], args, "ret_value",
2212  callBBlock);
2213  llvm::ReturnInst::Create(*g->ctx, retValue, callBBlock);
2214  }
