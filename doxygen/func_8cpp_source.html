157 /** Parameters for tasks are stored in a big structure; this utility
158  function emits code to copy those values out of the task structure into
159  local stack-allocated variables. (Which we expect that LLVM's
160  'mem2reg' pass will in turn promote to SSA registers..
161  */
162 static void
163 lCopyInTaskParameter(int i, llvm::Value *structArgPtr, const
164  std::vector<Symbol *> &args,
165  FunctionEmitContext *ctx) {
166  // We expect the argument structure to come in as a poitner to a
167  // structure. Confirm and figure out its type here.
168  const llvm::Type *structArgType = structArgPtr->getType();
169  Assert(llvm::isa<llvm::PointerType>(structArgType));
170  const llvm::PointerType *pt = llvm::dyn_cast<const llvm::PointerType>(structArgType);
171  Assert(llvm::isa<llvm::StructType>(pt->getElementType()));
172  const llvm::StructType *argStructType =
173  llvm::dyn_cast<const llvm::StructType>(pt->getElementType());
174 
175  // Get the type of the argument we're copying in and its Symbol pointer
176  llvm::Type *argType = argStructType->getElementType(i);
177  Symbol *sym = args[i];
178 
179  if (sym == NULL)
180  // anonymous parameter, so don't worry about it
181  return;
182 
183  // allocate space to copy the parameter in to
184  sym->storagePtr = ctx->AllocaInst(argType, sym->name.c_str());
185 
186  // get a pointer to the value in the struct
187  llvm::Value *ptr = ctx->AddElementOffset(structArgPtr, i, NULL, sym->name.c_str());
188 
189  // and copy the value from the struct and into the local alloca'ed
190  // memory
191  llvm::Value *ptrval = ctx->LoadInst(ptr, sym->name.c_str());
192  ctx->StoreInst(ptrval, sym->storagePtr);
193  ctx->EmitFunctionParameterDebugInfo(sym, i);
194 }

197 /** Given the statements implementing a function, emit the code that
198  implements the function. Most of the work do be done here just
199  involves wiring up the function parameter values to be available in the
200  function body code.
201  */
202 void
203 Function::emitCode(FunctionEmitContext *ctx, llvm::Function *function,
204  SourcePos firstStmtPos) {
205  // Connect the __mask builtin to the location in memory that stores its
206  // value
207  maskSymbol->storagePtr = ctx->GetFullMaskPointer();
208 
209  // add debugging info for __mask
210  maskSymbol->pos = firstStmtPos;
211  ctx->EmitVariableDebugInfo(maskSymbol);
212 
213 #if 0
214  llvm::BasicBlock *entryBBlock = ctx->GetCurrentBasicBlock();
215 #endif
216  const FunctionType *type = CastType<FunctionType>(sym->type);
217  Assert(type != NULL);
218  if (type->isTask == true) {
219  // For tasks, we there should always be three parmeters: the
220  // pointer to the structure that holds all of the arguments, the
221  // thread index, and the thread count variables.
222  llvm::Function::arg_iterator argIter = function->arg_begin();
223  llvm::Value *structParamPtr = argIter++;
224  llvm::Value *threadIndex = argIter++;
225  llvm::Value *threadCount = argIter++;
226  llvm::Value *taskIndex = argIter++;
227  llvm::Value *taskCount = argIter++;
228 
229  // Copy the function parameter values from the structure into local
230  // storage
231  for (unsigned int i = 0; i < args.size(); ++i)
232  lCopyInTaskParameter(i, structParamPtr, args, ctx);
233 
234  if (type->isUnmasked == false) {
235  // Copy in the mask as well.
236  int nArgs = (int)args.size();
237  // The mask is the last parameter in the argument structure
238  llvm::Value *ptr = ctx->AddElementOffset(structParamPtr, nArgs, NULL,
239  "task_struct_mask");
240  llvm::Value *ptrval = ctx->LoadInst(ptr, "mask");
241  ctx->SetFunctionMask(ptrval);
242  }
243 
244  // Copy threadIndex and threadCount into stack-allocated storage so
245  // that their symbols point to something reasonable.
246  threadIndexSym->storagePtr = ctx->AllocaInst(LLVMTypes::Int32Type, "threadIndex");
247  ctx->StoreInst(threadIndex, threadIndexSym->storagePtr);
248 
249  threadCountSym->storagePtr = ctx->AllocaInst(LLVMTypes::Int32Type, "threadCount");
250  ctx->StoreInst(threadCount, threadCountSym->storagePtr);
251 
252  // Copy taskIndex and taskCount into stack-allocated storage so
253  // that their symbols point to something reasonable.
254  taskIndexSym->storagePtr = ctx->AllocaInst(LLVMTypes::Int32Type, "taskIndex");
255  ctx->StoreInst(taskIndex, taskIndexSym->storagePtr);
256 
257  taskCountSym->storagePtr = ctx->AllocaInst(LLVMTypes::Int32Type, "taskCount");
258  ctx->StoreInst(taskCount, taskCountSym->storagePtr);
259  }
260  else {
261  // Regular, non-task function
262  llvm::Function::arg_iterator argIter = function->arg_begin();
263  for (unsigned int i = 0; i < args.size(); ++i, ++argIter) {
264  Symbol *sym = args[i];
265  if (sym == NULL)
266  // anonymous function parameter
267  continue;
268 
269  argIter->setName(sym->name.c_str());
270 
271  // Allocate stack storage for the parameter and emit code
272  // to store the its value there.
273  sym->storagePtr = ctx->AllocaInst(argIter->getType(), sym->name.c_str());
274  ctx->StoreInst(argIter, sym->storagePtr);
275  ctx->EmitFunctionParameterDebugInfo(sym, i);
276  }
277 
278  // If the number of actual function arguments is equal to the
279  // number of declared arguments in decl->functionParams, then we
280  // don't have a mask parameter, so set it to be all on. This
281  // happens for exmaple with 'export'ed functions that the app
282  // calls.
283  if (argIter == function->arg_end()) {
284  Assert(type->isUnmasked || type->isExported);
285  ctx->SetFunctionMask(LLVMMaskAllOn);
286  }
287  else {
288  Assert(type->isUnmasked == false);
289 
290  // Otherwise use the mask to set the entry mask value
291  argIter->setName("__mask");
292  Assert(argIter->getType() == LLVMTypes::MaskType);
293  ctx->SetFunctionMask(argIter);
294  Assert(++argIter == function->arg_end());
295  }
296  }
297 
298  // Finally, we can generate code for the function
299  if (code != NULL) {
300  ctx->SetDebugPos(code->pos);
301  ctx->AddInstrumentationPoint("function entry");
302 
303  int costEstimate = EstimateCost(code);
304  Debug(code->pos, "Estimated cost for function \"%s\" = %d\n",
305  sym->name.c_str(), costEstimate);
306 
307  // If the body of the function is non-trivial, then we wrap the
308  // entire thing inside code that tests to see if the mask is all
309  // on, all off, or mixed. If this is a simple function, then this
310  // isn't worth the code bloat / overhead.
311  bool checkMask = (type->isTask == true) ||
312  (
313 #if defined(LLVM_3_1)
314  (function->hasFnAttr(llvm::Attribute::AlwaysInline) == false)
315 #elif defined(LLVM_3_2)
316  (function->getFnAttributes().hasAttribute(llvm::Attributes::AlwaysInline) == false)
317 #else // LLVM 3.3+
318  (function->getAttributes().getFnAttributes().hasAttribute(llvm::AttributeSet::FunctionIndex, llvm::Attribute::AlwaysInline) == false)
319 #endif
320  &&
321  costEstimate > CHECK_MASK_AT_FUNCTION_START_COST);
322  checkMask &= (type->isUnmasked == false);
323  checkMask &= (g->target->getMaskingIsFree() == false);
324  checkMask &= (g->opt.disableCoherentControlFlow == false);
325 
326  if (checkMask) {
327  llvm::Value *mask = ctx->GetFunctionMask();
328  llvm::Value *allOn = ctx->All(mask);
329  llvm::BasicBlock *bbAllOn = ctx->CreateBasicBlock("all_on");
330  llvm::BasicBlock *bbSomeOn = ctx->CreateBasicBlock("some_on");
331 
332  // Set up basic blocks for goto targets
333  ctx->InitializeLabelMap(code);
334 
335  ctx->BranchInst(bbAllOn, bbSomeOn, allOn);
336  // all on: we've determined dynamically that the mask is all
337  // on. Set the current mask to "all on" explicitly so that
338  // codegen for this path can be improved with this knowledge in
339  // hand...
340  ctx->SetCurrentBasicBlock(bbAllOn);
341  if (!g->opt.disableMaskAllOnOptimizations)
342  ctx->SetFunctionMask(LLVMMaskAllOn);
343  code->EmitCode(ctx);
344  if (ctx->GetCurrentBasicBlock())
345  ctx->ReturnInst();
346 
347  // not all on: however, at least one lane must be running,
348  // since we should never run with all off... some on: reset
349  // the mask to the value it had at function entry and emit the
350  // code. Resetting the mask here is important, due to the "all
351  // on" setting of it for the path above.
352  ctx->SetCurrentBasicBlock(bbSomeOn);
353  ctx->SetFunctionMask(mask);
354 
355  // Set up basic blocks for goto targets again; we want to have
356  // one set of them for gotos in the 'all on' case, and a
357  // distinct set for the 'mixed mask' case.
358  ctx->InitializeLabelMap(code);
359 
360  code->EmitCode(ctx);
361  if (ctx->GetCurrentBasicBlock())
362  ctx->ReturnInst();
363  }
364  else {
365  // Set up basic blocks for goto targets
366  ctx->InitializeLabelMap(code);
367  // No check, just emit the code
368  code->EmitCode(ctx);
369  }
370  }
371 
372  if (ctx->GetCurrentBasicBlock()) {
373  // FIXME: We'd like to issue a warning if we've reached the end of
374  // the function without a return statement (for non-void
375  // functions). But the test below isn't right, since we can have
376  // (with 'x' a varying test) "if (x) return a; else return b;", in
377  // which case we have a valid basic block but its unreachable so ok
378  // to not have return statement.
379 #if 0
380  // If the bblock has no predecessors, then it doesn't matter if it
381  // doesn't have a return; it'll never be reached. If it does,
382  // issue a warning. Also need to warn if it's the entry block for
383  // the function (in which case it will not have predeccesors but is
384  // still reachable.)
385  if (Type::Equal(type->GetReturnType(), AtomicType::Void) == false &&
386  (pred_begin(ec.bblock) != pred_end(ec.bblock) || (ec.bblock == entryBBlock)))
387  Warning(sym->pos, "Missing return statement in function returning \"%s\".",
388  type->rType->GetString().c_str());
389 #endif
390 
391  // FIXME: would like to set the context's current position to
392  // e.g. the end of the function code
393 
394  // if bblock is non-NULL, it hasn't been terminated by e.g. a
395  // return instruction. Need to add a return instruction.
396  ctx->ReturnInst();
397  }

401 void
402 Function::GenerateIR() {
403  if (sym == NULL)
404  // May be NULL due to error earlier in compilation
405  return;
406 
407  llvm::Function *function = sym->function;
408  Assert(function != NULL);
409 
410  // But if that function has a definition, we don't want to redefine it.
411  if (function->empty() == false) {
412  Error(sym->pos, "Ignoring redefinition of function \"%s\".",
413  sym->name.c_str());
414  return;
415  }
416 
417  // Figure out a reasonable source file position for the start of the
418  // function body. If possible, get the position of the first actual
419  // non-StmtList statment...
420  SourcePos firstStmtPos = sym->pos;
421  if (code) {
422  StmtList *sl = dynamic_cast<StmtList *>(code);
423  if (sl && sl->stmts.size() > 0 && sl->stmts[0] != NULL)
424  firstStmtPos = sl->stmts[0]->pos;
425  else
426  firstStmtPos = code->pos;
427  }
428 
429  // And we can now go ahead and emit the code
430  {
431  FunctionEmitContext ec(this, sym, function, firstStmtPos);
432  emitCode(&ec, function, firstStmtPos);
433  }
434 
435  if (m->errorCount == 0) {
436  if (llvm::verifyFunction(*function, llvm::ReturnStatusAction) == true) {
437  if (g->debugPrint)
438  function->dump();
439  FATAL("Function verificication failed");
440  }
441 
442  // If the function is 'export'-qualified, emit a second version of
443  // it without a mask parameter and without name mangling so that
444  // the application can call it
445  const FunctionType *type = CastType<FunctionType>(sym->type);
446  Assert(type != NULL);
447  if (type->isExported) {
448  if (!type->isTask) {
449  llvm::FunctionType *ftype = type->LLVMFunctionType(g->ctx, true);
450  llvm::GlobalValue::LinkageTypes linkage = llvm::GlobalValue::ExternalLinkage;
451  std::string functionName = sym->name;
452  if (g->mangleFunctionsWithTarget)
453  functionName += std::string("_") + g->target->GetISAString();
454  llvm::Function *appFunction =
455  llvm::Function::Create(ftype, linkage, functionName.c_str(), m->module);
456 #if defined(LLVM_3_1)
457  appFunction->setDoesNotThrow(true);
458 #else
459  appFunction->setDoesNotThrow();
460 #endif
461 
462  g->target->markFuncWithTargetAttr(appFunction);
463 
464  if (appFunction->getName() != functionName) {
465  // this was a redefinition for which we already emitted an
466  // error, so don't worry about this one...
467  appFunction->eraseFromParent();
468  }
469  else {
470  // And emit the code again
471  FunctionEmitContext ec(this, sym, appFunction, firstStmtPos);
472  emitCode(&ec, appFunction, firstStmtPos);
473  if (m->errorCount == 0) {
474  sym->exportedFunction = appFunction;
475  if (llvm::verifyFunction(*appFunction,
476  llvm::ReturnStatusAction) == true) {
477  if (g->debugPrint)
478  appFunction->dump();
479  FATAL("Function verificication failed");
480  }
481  }
482  }
483  }
484  }
