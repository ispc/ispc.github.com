76 /** Returns the width of the terminal where the compiler is running.
77  Finding this out may fail in a variety of reasonable situations (piping
78  compiler output to 'less', redirecting output to a file, running the
79  compiler under a debuffer; in this case, just return a reasonable
80  default.
81  */
82 static int
83 lTerminalWidth() {

85  return 1<<30;
86 
87 #if defined(ISPC_IS_WINDOWS)

97  return 80;

99 #endif // ISPC_IS_WINDOWS
100 }

160 /** When printing error messages, we sometimes want to include the source
161  file line for context. This function print the line(s) of the file
162  corresponding to the provided SourcePos and underlines the range of the
163  SourcePos with '^' symbols.
164 */
165 static void
166 lPrintFileLineContext(SourcePos p) {
167  if (p.first_line == 0)
168  return;
169 
170  FILE *f = fopen(p.name, "r");
171  if (!f)
172  return;
173 
174  int c, curLine = 1;
175  while ((c = fgetc(f)) != EOF) {
176  // Don't print more than three lines of context. (More than that,
177  // and we're probably doing the wrong thing...)
178  if (curLine >= std::max(p.first_line, p.last_line-2) &&
179  curLine <= p.last_line)
180  fputc(c, stderr);
181  if (c == '\n')
182  ++curLine;
183  if (curLine > p.last_line)
184  break;
185  }
186 
187  int i = 1;
188  for (; i < p.first_column; ++i)
189  fputc(' ', stderr);
190  fputc('^', stderr);
191  ++i;
192  for (; i < p.last_column; ++i)
193  fputc('^', stderr);
194  fputc('\n', stderr);
195  fputc('\n', stderr);
196 
197  fclose(f);
198 }

201 /** Counts the number of characters into the buf at which the numColons
202  colon character is found. Skips over ANSI escape sequences and doesn't
203  include their characters in the final count.
204  */
205 static int
206 lFindIndent(int numColons, const char *buf) {
207  int indent = 0;
208  while (*buf != '\0') {
209  if (*buf == '\033') {
210  while (*buf != '\0' && *buf != 'm')
211  ++buf;
212  if (*buf == 'm')
213  ++buf;
214  }
215  else {
216  if (*buf == ':') {
217  if (--numColons == 0)
218  break;
219  }
220  ++indent;
221  ++buf;
222  }
223  }
224  return indent + 2;
225 }

228 /** Print the given string to the given FILE, assuming the given output
229  column width. Break words as needed to avoid words spilling past the
230  last column. */
231 static void
232 lPrintWithWordBreaks(const char *buf, int indent, int columnWidth, FILE *out) {
233 #ifdef ISPC_IS_WINDOWS
234  fputs(buf, out);
235  fputs("\n", out);
236 #else
237  int column = 0;
238  int width = std::max(40, columnWidth - 2);
239 
240  // Collect everything into a string and print it all at once at the end
241  // -> try to avoid troubles with mangled error messages with
242  // multi-threaded builds.
243  std::string outStr;
244 
245  const char *msgPos = buf;
246  while (true) {
247  if (*msgPos == '\033') {
248  // handle ANSI color escape: copy it to the output buffer
249  // without charging for the characters it uses
250  do {
251  outStr.push_back(*msgPos++);
252  } while (*msgPos != '\0' && *msgPos != 'm');
253  continue;
254  }
255  else if (*msgPos == '\n') {
256  // Handle newlines cleanly
257  column = indent;
258  outStr.push_back('\n');
259  for (int i = 0; i < indent; ++i)
260  outStr.push_back(' ');
261  // Respect spaces after newlines
262  ++msgPos;
263  while (*msgPos == ' ') {
264  outStr.push_back(' ');
265  ++msgPos;
266  }
267  continue;
268  }
269 
270  while (*msgPos != '\0' && isspace(*msgPos))
271  ++msgPos;
272  if (*msgPos == '\0')
273  break;
274 
275  const char *wordEnd = lFindWordEnd(msgPos);
276  if (column > indent && column + wordEnd - msgPos > width) {
277  // This word would overflow, so start a new line
278  column = indent;
279  outStr.push_back('\n');
280  // Indent to the same column as the ":" at the start of the
281  // message.
282  for (int i = 0; i < indent; ++i)
283  outStr.push_back(' ');
284  }
285 
286  // Finally go and copy the word
287  while (msgPos != wordEnd) {
288  outStr.push_back(*msgPos++);
289  ++column;
290  }
291  outStr.push_back(' ');
292  ++column;
293  }
294  outStr.push_back('\n');
295  fputs(outStr.c_str(), out);
296 #endif
297 }

326 /** Helper function for Error(), Warning(), etc.
327 
328  @param type The type of message being printed (e.g. "Warning")
329  @param p Position in source file that is connected to the message
330  being printed
331  @param fmt printf()-style format string
332  @param args Arguments with values for format string % entries
333 */
334 static void
335 lPrint(const char *type, bool isError, SourcePos p, const char *fmt,
336  va_list args) {
337  char *errorBuf, *formattedBuf;
338  if (vasprintf(&errorBuf, fmt, args) == -1) {
339  fprintf(stderr, "vasprintf() unable to allocate memory!\n");
340  abort();
341  }
342 
343  int indent = 0;
344  if (p.first_line == 0) {
345  // We don't have a valid SourcePos, so create a message without it
346  if (asprintf(&formattedBuf, "%s%s%s%s%s: %s%s", lStartBold(),
347  isError ? lStartRed() : lStartBlue(), type,
348  lResetColor(), lStartBold(), errorBuf,
349  lResetColor()) == -1) {
350  fprintf(stderr, "asprintf() unable to allocate memory!\n");
351  exit(1);
352  }
353  indent = lFindIndent(1, formattedBuf);
354  }
355  else {
356  // Create an error message that includes the file and line number
357  if (asprintf(&formattedBuf, "%s%s:%d:%d: %s%s%s%s: %s%s",
358  lStartBold(), p.name, p.first_line, p.first_column,
359  isError ? lStartRed() : lStartBlue(), type,
360  lResetColor(), lStartBold(), errorBuf,
361  lResetColor()) == -1) {
362  fprintf(stderr, "asprintf() unable to allocate memory!\n");
363  exit(1);
364  }
365  indent = lFindIndent(3, formattedBuf);
366  }
367  // Don't indent too much with long filenames
368  indent = std::min(indent, 8);
369 
370  // Now that we've done all that work, see if we've already printed the
371  // exact same error message. If so, return, so we don't redundantly
372  // print it and annoy the user.
373  static std::set<std::string> printed;
374  if (printed.find(formattedBuf) != printed.end())
375  return;
376  printed.insert(formattedBuf);
377 
378  lPrintWithWordBreaks(formattedBuf, indent, lTerminalWidth(), stderr);
379  lPrintFileLineContext(p);
380 
381  free(errorBuf);
382  free(formattedBuf);
383 }

482 // http://en.wikipedia.org/wiki/Levenshtein_distance
483 int
484 StringEditDistance(const std::string &str1, const std::string &str2, int maxDist) {
485  // Small hack: don't return 0 if the strings are the same; if we've
486  // gotten here, there's been a parsing error, and suggesting the same
487  // string isn't going to actually help things.
488  if (str1 == str2)
489  return maxDist;
490 
491  int n1 = (int)str1.size(), n2 = (int)str2.size();
492  int nmax = std::max(n1, n2);
493 
494  int *current = (int *)alloca((nmax+1) * sizeof(int));
495  int *previous = (int *)alloca((nmax+1) * sizeof(int));
496 
497  for (int i = 0; i <= n2; ++i)
498  previous[i] = i;
499 
500  for (int y = 1; y <= n1; ++y) {
501  current[0] = y;
502  int rowBest = y;
503 
504  for (int x = 1; x <= n2; ++x) {
505  current[x] = std::min(previous[x-1] + (str1[y-1] == str2[x-1] ? 0 : 1),
506  std::min(current[x-1], previous[x])+1);
507  rowBest = std::min(rowBest, current[x]);
508  }
509 
510  if (maxDist != 0 && rowBest > maxDist)
511  return maxDist + 1;
512 
513  std::swap(current, previous);
514  }
515 
516  return previous[n2];
517 }

520 std::vector<std::string>
521 MatchStrings(const std::string &str, const std::vector<std::string> &options) {
522  if (str.size() == 0 || (str.size() == 1 && !isalpha(str[0])))
523  // don't even try...
524  return std::vector<std::string>();
525 
526  const int maxDelta = 2;
527  std::vector<std::string> matches[maxDelta+1];
528 
529  // For all of the options that are up to maxDelta edit distance, store
530  // them in the element of matches[] that corresponds to their edit
531  // distance.
532  for (int i = 0; i < (int)options.size(); ++i) {
533  int dist = StringEditDistance(str, options[i], maxDelta+1);
534  if (dist <= maxDelta)
535  matches[dist].push_back(options[i]);
536  }
537 
538  // And return the first one of them, if any, that has at least one
539  // match.
540  for (int i = 0; i <= maxDelta; ++i) {
541  if (matches[i].size())
542  return matches[i];
543  }
544  return std::vector<std::string>();
545 }

549 GetDirectoryAndFileName(const std::string &currentDirectory,

586 static std::set<std::string> lGetStringArray(const std::string &str) {
587  std::set<std::string> result;
588 
589  Assert(str.find('-') != str.npos);
590 
591  size_t pos_prev = 0, pos;
592  do {
593  pos = str.find('-', pos_prev);
594  std::string substr = str.substr(pos_prev, pos-pos_prev);
595  result.insert(substr);
596  pos_prev = pos;
597  pos_prev++;
598  } while (pos != str.npos);
599 
600  return result;
601 }

604 bool
605 VerifyDataLayoutCompatibility(const std::string &module_dl,
606  const std::string &lib_dl) {
607  if (lib_dl.empty()) {
608  // This is the case for most of library pre-compiled .ll files.
609  return true;
610  }
611 
612  // Get "canonical" form. Instead of looking at "raw" DataLayout string, we
613  // look at the actual representation, as DataLayout class understands it.
614  // In the most cases there's no difference. But on x86 Windows (i386-pc-win32),
615  // clang generates a DataLayout string, which contains two definitions of f80,
616  // which contradic: f80:128:128 followed by f80:32:32. This is a bug, but
617  // correct thing to do is to interpret this exactly how LLVM would treat it,
618  // so we create a DataLayout class and take its string representation.
619 #if defined(LLVM_3_1)
620  llvm::TargetData d1(module_dl);
621  llvm::TargetData d2(lib_dl);
622 #else // LLVM 3.2+
623  llvm::DataLayout d1(module_dl);
624  llvm::DataLayout d2(lib_dl);
625 #endif
626 
627  std::string module_dl_canonic = d1.getStringRepresentation();
628  std::string lib_dl_canonic = d2.getStringRepresentation();
629 
630  // Break down DataLayout strings to separate type definitions.
631  std::set<std::string> module_dl_set = lGetStringArray(module_dl_canonic);
632  std::set<std::string> lib_dl_set = lGetStringArray(lib_dl_canonic);
633 
634  // For each element in library data layout, find matching module element.
635  // If no match is found, then we are in trouble and the library can't be used.
636  for (std::set<std::string>::iterator it = lib_dl_set.begin();
637  it != lib_dl_set.end(); ++it) {
638  // We use the simplest possible definition of "match", which is match exactly.
639  // Ideally it should be relaxed and for triples [p|i|v|f|a|s]<size>:<abi>:<pref>
640  // we should allow <pref> part (preferred alignment) to not match.
641  // But this seems to have no practical value at this point.
642  std::set<std::string>::iterator module_match =
643  std::find(module_dl_set.begin(), module_dl_set.end(), *it);
644  if (module_match == module_dl_set.end()) {
645  // No match for this piece of library DataLayout was found,
646  // return false.
647  return false;
648  }
649  // Remove matching piece from Module set.
650  module_dl_set.erase(module_match);
651  }
652 
653  // We allow extra types to be defined in the Module, but we should check
654  // that it's something that we expect. And we expect vectors and floats.
655  for (std::set<std::string>::iterator it = module_dl_set.begin();
656  it != module_dl_set.end(); ++it) {
657  if ((*it)[0] == 'v' || (*it)[0] == 'f') {
658  continue;
659  }
660  return false;
661  }
662 
663  return true;
664 }
