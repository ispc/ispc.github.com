177 /** We take arguments from both the command line as well as from the
178  ISPC_ARGS environment variable. This function returns a new set of
179  arguments representing the ones from those two sources merged together.
180 */
181 static void lGetAllArgs(int Argc, char *Argv[], int &argc, char *argv[128]) {
182  // Copy over the command line arguments (passed in)
183  for (int i = 0; i < Argc; ++i)
184  argv[i] = Argv[i];
185  argc = Argc;
186 
187  // See if we have any set via the environment variable
188  const char *env = getenv("ISPC_ARGS");
189  if (!env)
190  return;
191  while (true) {
192  // Look for the next space in the string, which delimits the end of
193  // the current argument
194  const char *end = strchr(env, ' ');
195  if (end == NULL)
196  end = env + strlen(env);
197  int len = end - env;
198 
199  // Copy the argument into a newly allocated memory (so we can
200  // NUL-terminate it).
201  char *ptr = new char[len+1];
202  strncpy(ptr, env, len);
203  ptr[len] = '\0';
204 
205  // Add it to the args array and get out of here
206  argv[argc++] = ptr;
207  if (*end == '\0')
208  break;
209 
210  // Advance the starting pointer of the string to the next non-space
211  // character
212  env = end+1;
213  while (*env == ' ')
214  ++env;
215 
216  // Hit the end of the string; get out of here
217  if (*env == '\0')
218  break;
219  }
220 }

229 static int ParsingPhaseName(char * stage) {
230  if (strncmp(stage, "first", 5) == 0) {
231  return 0;
232  }
233  else if (strncmp(stage, "last", 4) == 0) {
234  return LAST_OPT_NUMBER;
235  }
236  else {
237  int t = atoi(stage);
238  if (t < 0 || t > LAST_OPT_NUMBER) {
239  fprintf(stderr, "Phases must be from 0 to %d. %s is incorrect.\n", LAST_OPT_NUMBER, stage);
240  exit(0);
241  }
242  else {
243  return t;
244  }
245  }
246 }

249 static std::set<int> ParsingPhases(char * stages) {
250  std::set<int> phases;
251  int begin = ParsingPhaseName(stages);
252  int end = begin;
253 
254  for (unsigned i = 0; i < strlen(stages); i++) {
255  if ((stages[i] == ',') || (i == strlen(stages) - 1)) {
256  for (int j = begin; j < end + 1; j++) {
257  phases.insert(j);
258  }
259  begin = ParsingPhaseName(stages + i + 1);
260  end = begin;
261  }
262  else if (stages[i] == ':') {
263  end = ParsingPhaseName(stages + i + 1);
264  }
265  }
266  return phases;
267 }

270 static void
271 lParseInclude(const char *path) {
272 #ifdef ISPC_IS_WINDOWS
273  char delim = ';';
274 #else
275  char delim = ':';
276 #endif
277  size_t pos = 0, pos_end;
278  std::string str_path(path);
279  do {
280  pos_end = str_path.find(delim, pos);
281  size_t len = (pos_end == std::string::npos) ?
282  // Not found, copy till end of the string.
283  std::string::npos :
284  // Copy [pos, pos_end).
285  (pos_end - pos);
286  std::string s = str_path.substr(pos, len);
287  g->includePath.push_back(s);
288  pos = pos_end+1;
289  } while (pos_end != std::string::npos);
290 }
