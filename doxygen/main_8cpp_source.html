163 /** We take arguments from both the command line as well as from the
164  ISPC_ARGS environment variable. This function returns a new set of
165  arguments representing the ones from those two sources merged together.
166 */
167 static void lGetAllArgs(int Argc, char *Argv[], int &argc, char *argv[128]) {
168  // Copy over the command line arguments (passed in)
169  for (int i = 0; i < Argc; ++i)
170  argv[i] = Argv[i];
171  argc = Argc;
172 
173  // See if we have any set via the environment variable
174  const char *env = getenv("ISPC_ARGS");
175  if (!env)
176  return;
177  while (true) {
178  // Look for the next space in the string, which delimits the end of
179  // the current argument
180  const char *end = strchr(env, ' ');
181  if (end == NULL)
182  end = env + strlen(env);
183  int len = end - env;
184 
185  // Copy the argument into a newly allocated memory (so we can
186  // NUL-terminate it).
187  char *ptr = new char[len+1];
188  strncpy(ptr, env, len);
189  ptr[len] = '\0';
190 
191  // Add it to the args array and get out of here
192  argv[argc++] = ptr;
193  if (*end == '\0')
194  break;
195 
196  // Advance the starting pointer of the string to the next non-space
197  // character
198  env = end+1;
199  while (*env == ' ')
200  ++env;
201 
202  // Hit the end of the string; get out of here
203  if (*env == '\0')
204  break;
205  }
206 }

215 static void
216 lParseInclude(const char *path) {
217 #ifdef ISPC_IS_WINDOWS
218  char delim = ';';
219 #else
220  char delim = ':';
221 #endif
222  size_t pos = 0, pos_end;
223  std::string str_path(path);
224  do {
225  pos_end = str_path.find(delim, pos);
226  size_t len = (pos_end == std::string::npos) ?
227  // Not found, copy till end of the string.
228  std::string::npos :
229  // Copy [pos, pos_end).
230  (pos_end - pos);
231  std::string s = str_path.substr(pos, len);
232  g->includePath.push_back(s);
233  pos = pos_end+1;
234  } while (pos_end != std::string::npos);
235 }
