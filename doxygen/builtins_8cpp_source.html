80 /** Given an LLVM type, try to find the equivalent ispc type. Note that
81  this is an under-constrained problem due to LLVM's type representations
82  carrying less information than ispc's. (For example, LLVM doesn't
83  distinguish between signed and unsigned integers in its types.)
84 
85  Because this function is only used for generating ispc declarations of
86  functions defined in LLVM bitcode in the builtins-*.ll files, in practice
87  we can get enough of what we need for the relevant cases to make things
88  work, partially with the help of the intAsUnsigned parameter, which
89  indicates whether LLVM integer types should be treated as being signed
90  or unsigned.
91 
92  */
93 static const Type *
94 lLLVMTypeToISPCType(const llvm::Type *t, bool intAsUnsigned) {
95  if (t == LLVMTypes::VoidType)
96  return AtomicType::Void;
97 
98  // uniform
99  else if (t == LLVMTypes::BoolType)
100  return AtomicType::UniformBool;
101  else if (t == LLVMTypes::Int8Type)
102  return intAsUnsigned ? AtomicType::UniformUInt8 : AtomicType::UniformInt8;
103  else if (t == LLVMTypes::Int16Type)
104  return intAsUnsigned ? AtomicType::UniformUInt16 : AtomicType::UniformInt16;
105  else if (t == LLVMTypes::Int32Type)
106  return intAsUnsigned ? AtomicType::UniformUInt32 : AtomicType::UniformInt32;
107  else if (t == LLVMTypes::FloatType)
108  return AtomicType::UniformFloat;
109  else if (t == LLVMTypes::DoubleType)
110  return AtomicType::UniformDouble;
111  else if (t == LLVMTypes::Int64Type)
112  return intAsUnsigned ? AtomicType::UniformUInt64 : AtomicType::UniformInt64;
113 
114  // varying
115  if (LLVMTypes::MaskType != LLVMTypes::Int32VectorType &&
116  t == LLVMTypes::MaskType)
117  return AtomicType::VaryingBool;
118  else if (t == LLVMTypes::Int8VectorType)
119  return intAsUnsigned ? AtomicType::VaryingUInt8 : AtomicType::VaryingInt8;
120  else if (t == LLVMTypes::Int16VectorType)
121  return intAsUnsigned ? AtomicType::VaryingUInt16 : AtomicType::VaryingInt16;
122  else if (t == LLVMTypes::Int32VectorType)
123  return intAsUnsigned ? AtomicType::VaryingUInt32 : AtomicType::VaryingInt32;
124  else if (t == LLVMTypes::FloatVectorType)
125  return AtomicType::VaryingFloat;
126  else if (t == LLVMTypes::DoubleVectorType)
127  return AtomicType::VaryingDouble;
128  else if (t == LLVMTypes::Int64VectorType)
129  return intAsUnsigned ? AtomicType::VaryingUInt64 : AtomicType::VaryingInt64;
130 
131  // pointers to uniform
132  else if (t == LLVMTypes::Int8PointerType)
133  return PointerType::GetUniform(intAsUnsigned ? AtomicType::UniformUInt8 :
134  AtomicType::UniformInt8);
135  else if (t == LLVMTypes::Int16PointerType)
136  return PointerType::GetUniform(intAsUnsigned ? AtomicType::UniformUInt16 :
137  AtomicType::UniformInt16);
138  else if (t == LLVMTypes::Int32PointerType)
139  return PointerType::GetUniform(intAsUnsigned ? AtomicType::UniformUInt32 :
140  AtomicType::UniformInt32);
141  else if (t == LLVMTypes::Int64PointerType)
142  return PointerType::GetUniform(intAsUnsigned ? AtomicType::UniformUInt64 :
143  AtomicType::UniformInt64);
144  else if (t == LLVMTypes::FloatPointerType)
145  return PointerType::GetUniform(AtomicType::UniformFloat);
146  else if (t == LLVMTypes::DoublePointerType)
147  return PointerType::GetUniform(AtomicType::UniformDouble);
148 
149  // pointers to varying
150  else if (t == LLVMTypes::Int8VectorPointerType)
151  return PointerType::GetUniform(intAsUnsigned ? AtomicType::VaryingUInt8 :
152  AtomicType::VaryingInt8);
153  else if (t == LLVMTypes::Int16VectorPointerType)
154  return PointerType::GetUniform(intAsUnsigned ? AtomicType::VaryingUInt16 :
155  AtomicType::VaryingInt16);
156  else if (t == LLVMTypes::Int32VectorPointerType)
157  return PointerType::GetUniform(intAsUnsigned ? AtomicType::VaryingUInt32 :
158  AtomicType::VaryingInt32);
159  else if (t == LLVMTypes::Int64VectorPointerType)
160  return PointerType::GetUniform(intAsUnsigned ? AtomicType::VaryingUInt64 :
161  AtomicType::VaryingInt64);
162  else if (t == LLVMTypes::FloatVectorPointerType)
163  return PointerType::GetUniform(AtomicType::VaryingFloat);
164  else if (t == LLVMTypes::DoubleVectorPointerType)
165  return PointerType::GetUniform(AtomicType::VaryingDouble);
166 
167  return NULL;
168 }

190 /** Given an LLVM function declaration, synthesize the equivalent ispc
191  symbol for the function (if possible). Returns true on success, false
192  on failure.
193  */
194 static bool
195 lCreateISPCSymbol(llvm::Function *func, SymbolTable *symbolTable) {
196  SourcePos noPos;
197  noPos.name = "__stdlib";
198 
199  const llvm::FunctionType *ftype = func->getFunctionType();
200  std::string name = func->getName();
201 
202  if (name.size() < 3 || name[0] != '_' || name[1] != '_')
203  return false;
204 
205  Debug(SourcePos(), "Attempting to create ispc symbol for function \"%s\".",
206  name.c_str());
207 
208  // An unfortunate hack: we want this builtin function to have the
209  // signature "int __sext_varying_bool(bool)", but the ispc function
210  // symbol creation code below assumes that any LLVM vector of i32s is a
211  // varying int32. Here, we need that to be interpreted as a varying
212  // bool, so just have a one-off override for that one...
213  if (g->target->getMaskBitCount() != 1 && name == "__sext_varying_bool") {
214  const Type *returnType = AtomicType::VaryingInt32;
215  llvm::SmallVector<const Type *, 8> argTypes;
216  argTypes.push_back(AtomicType::VaryingBool);
217 
218  FunctionType *funcType = new FunctionType(returnType, argTypes, noPos);
219 
220  Symbol *sym = new Symbol(name, noPos, funcType);
221  sym->function = func;
222  symbolTable->AddFunction(sym);
223  return true;
224  }
225 
226  // If the function has any parameters with integer types, we'll make
227  // two Symbols for two overloaded versions of the function, one with
228  // all of the integer types treated as signed integers and one with all
229  // of them treated as unsigned.
230  for (int i = 0; i < 2; ++i) {
231  bool intAsUnsigned = (i == 1);
232 
233  const Type *returnType = lLLVMTypeToISPCType(ftype->getReturnType(),
234  intAsUnsigned);
235  if (returnType == NULL) {
236  Debug(SourcePos(), "Failed: return type not representable for "
237  "builtin %s.", name.c_str());
238  // return type not representable in ispc -> not callable from ispc
239  return false;
240  }
241 
242  // Iterate over the arguments and try to find their equivalent ispc
243  // types. Track if any of the arguments has an integer type.
244  bool anyIntArgs = false;
245  llvm::SmallVector<const Type *, 8> argTypes;
246  for (unsigned int j = 0; j < ftype->getNumParams(); ++j) {
247  const llvm::Type *llvmArgType = ftype->getParamType(j);
248  const Type *type = lLLVMTypeToISPCType(llvmArgType, intAsUnsigned);
249  if (type == NULL) {
250  Debug(SourcePos(), "Failed: type of parameter %d not "
251  "representable for builtin %s", j, name.c_str());
252  return false;
253  }
254  anyIntArgs |=
255  (Type::Equal(type, lLLVMTypeToISPCType(llvmArgType, !intAsUnsigned)) == false);
256  argTypes.push_back(type);
257  }
258 
259  // Always create the symbol the first time through, in particular
260  // so that we get symbols for things with no integer types!
261  if (i == 0 || anyIntArgs == true)
262  lCreateSymbol(name, returnType, argTypes, ftype, func, symbolTable);
263  }
264 
265  return true;
266 }

287 /** In many of the builtins-*.ll files, we have declarations of various LLVM
288  intrinsics that are then used in the implementation of various target-
289  specific functions. This function loops over all of the intrinsic
290  declarations and makes sure that the signature we have in our .ll file
291  matches the signature of the actual intrinsic.
292 */
293 static void
294 lCheckModuleIntrinsics(llvm::Module *module) {
295  llvm::Module::iterator iter;
296  for (iter = module->begin(); iter != module->end(); ++iter) {
297  llvm::Function *func = iter;
298  if (!func->isIntrinsic())
299  continue;
300 
301  const std::string funcName = func->getName().str();
302  // Work around http://llvm.org/bugs/show_bug.cgi?id=10438; only
303  // check the llvm.x86.* intrinsics for now...
304  if (!strncmp(funcName.c_str(), "llvm.x86.", 9)) {
305  llvm::Intrinsic::ID id = (llvm::Intrinsic::ID)func->getIntrinsicID();
306  Assert(id != 0);
307  llvm::Type *intrinsicType =
308  llvm::Intrinsic::getType(*g->ctx, id);
309  intrinsicType = llvm::PointerType::get(intrinsicType, 0);
310  Assert(func->getType() == intrinsicType);
311  }
312  }
313 }

612 /** This utility function takes serialized binary LLVM bitcode and adds its
613  definitions to the given module. Functions in the bitcode that can be
614  mapped to ispc functions are also added to the symbol table.
615 
616  @param bitcode Binary LLVM bitcode (e.g. the contents of a *.bc file)
617  @param length Length of the bitcode buffer
618  @param module Module to link the bitcode into
619  @param symbolTable Symbol table to add definitions to
620  */
621 void
622 AddBitcodeToModule(const unsigned char *bitcode, int length,
623  llvm::Module *module, SymbolTable *symbolTable) {
624  std::string bcErr;
625  llvm::StringRef sb = llvm::StringRef((char *)bitcode, length);
626  llvm::MemoryBuffer *bcBuf = llvm::MemoryBuffer::getMemBuffer(sb);
627  llvm::Module *bcModule = llvm::ParseBitcodeFile(bcBuf, *g->ctx, &bcErr);
628  if (!bcModule)
629  Error(SourcePos(), "Error parsing stdlib bitcode: %s", bcErr.c_str());
630  else {
631  // FIXME: this feels like a bad idea, but the issue is that when we
632  // set the llvm::Module's target triple in the ispc Module::Module
633  // constructor, we start by calling llvm::sys::getHostTriple() (and
634  // then change the arch if needed). Somehow that ends up giving us
635  // strings like 'x86_64-apple-darwin11.0.0', while the stuff we
636  // compile to bitcode with clang has module triples like
637  // 'i386-apple-macosx10.7.0'. And then LLVM issues a warning about
638  // linking together modules with incompatible target triples..
639  llvm::Triple mTriple(m->module->getTargetTriple());
640  llvm::Triple bcTriple(bcModule->getTargetTriple());
641  Assert(bcTriple.getArch() == llvm::Triple::UnknownArch ||
642  mTriple.getArch() == bcTriple.getArch());
643  Assert(bcTriple.getVendor() == llvm::Triple::UnknownVendor ||
644  mTriple.getVendor() == bcTriple.getVendor());
645  bcModule->setTargetTriple(mTriple.str());
646 
647  // We unconditionally set module DataLayout to library, but we must
648  // ensure that library and module DataLayouts are compatible.
649  // If they are not, we should recompile the library for problematic
650  // architecture and investigate what happened.
651  // Generally we allow library DataLayout to be subset of module
652  // DataLayout or library DataLayout to be empty.
653  if (!VerifyDataLayoutCompatibility(module->getDataLayout(),
654  bcModule->getDataLayout())) {
655  Error(SourcePos(), "Module DataLayout is incompatible with library DataLayout:\n"
656  "Module DL: %s\n"
657  "Library DL: %s\n",
658  module->getDataLayout().c_str(), bcModule->getDataLayout().c_str());
659  }
660 
661  bcModule->setDataLayout(module->getDataLayout());
662 
663  std::string(linkError);
664  if (llvm::Linker::LinkModules(module, bcModule,
665  llvm::Linker::DestroySource,
666  &linkError))
667  Error(SourcePos(), "Error linking stdlib bitcode: %s", linkError.c_str());
668  lSetInternalFunctions(module);
669  if (symbolTable != NULL)
670  lAddModuleSymbols(module, symbolTable);
671  lCheckModuleIntrinsics(module);
672  }

691  llvm::GlobalValue::LinkageTypes linkage = g->generateDebuggingSymbols ?
692  llvm::GlobalValue::WeakODRLinkage : llvm::GlobalValue::InternalLinkage;

694  linit, name);

702  // have the DW_AT_artifical attribute. It's not clear if this
703  // matters for anything though.
704  llvm::DIGlobalVariable var =
705  m->diBuilder->createGlobalVariable(name,
