82 /** Given an LLVM type, try to find the equivalent ispc type. Note that
83  this is an under-constrained problem due to LLVM's type representations
84  carrying less information than ispc's. (For example, LLVM doesn't
85  distinguish between signed and unsigned integers in its types.)
86 
87  Because this function is only used for generating ispc declarations of
88  functions defined in LLVM bitcode in the builtins-*.ll files, in practice
89  we can get enough of what we need for the relevant cases to make things
90  work, partially with the help of the intAsUnsigned parameter, which
91  indicates whether LLVM integer types should be treated as being signed
92  or unsigned.
93 
94  */
95 static const Type *
96 lLLVMTypeToISPCType(const llvm::Type *t, bool intAsUnsigned) {
97  if (t == LLVMTypes::VoidType)
98  return AtomicType::Void;
99 
100  // uniform
101  else if (t == LLVMTypes::BoolType)
102  return AtomicType::UniformBool;
103  else if (t == LLVMTypes::Int8Type)
104  return intAsUnsigned ? AtomicType::UniformUInt8 : AtomicType::UniformInt8;
105  else if (t == LLVMTypes::Int16Type)
106  return intAsUnsigned ? AtomicType::UniformUInt16 : AtomicType::UniformInt16;
107  else if (t == LLVMTypes::Int32Type)
108  return intAsUnsigned ? AtomicType::UniformUInt32 : AtomicType::UniformInt32;
109  else if (t == LLVMTypes::FloatType)
110  return AtomicType::UniformFloat;
111  else if (t == LLVMTypes::DoubleType)
112  return AtomicType::UniformDouble;
113  else if (t == LLVMTypes::Int64Type)
114  return intAsUnsigned ? AtomicType::UniformUInt64 : AtomicType::UniformInt64;
115 
116  // varying
117  if (t == LLVMTypes::Int8VectorType)
118  return intAsUnsigned ? AtomicType::VaryingUInt8 : AtomicType::VaryingInt8;
119  else if (t == LLVMTypes::Int16VectorType)
120  return intAsUnsigned ? AtomicType::VaryingUInt16 : AtomicType::VaryingInt16;
121  else if (t == LLVMTypes::Int32VectorType)
122  return intAsUnsigned ? AtomicType::VaryingUInt32 : AtomicType::VaryingInt32;
123  else if (t == LLVMTypes::FloatVectorType)
124  return AtomicType::VaryingFloat;
125  else if (t == LLVMTypes::DoubleVectorType)
126  return AtomicType::VaryingDouble;
127  else if (t == LLVMTypes::Int64VectorType)
128  return intAsUnsigned ? AtomicType::VaryingUInt64 : AtomicType::VaryingInt64;
129  else if (t == LLVMTypes::MaskType)
130  return AtomicType::VaryingBool;
131 
132  // pointers to uniform
133  else if (t == LLVMTypes::Int8PointerType)
134  return PointerType::GetUniform(intAsUnsigned ? AtomicType::UniformUInt8 :
135  AtomicType::UniformInt8);
136  else if (t == LLVMTypes::Int16PointerType)
137  return PointerType::GetUniform(intAsUnsigned ? AtomicType::UniformUInt16 :
138  AtomicType::UniformInt16);
139  else if (t == LLVMTypes::Int32PointerType)
140  return PointerType::GetUniform(intAsUnsigned ? AtomicType::UniformUInt32 :
141  AtomicType::UniformInt32);
142  else if (t == LLVMTypes::Int64PointerType)
143  return PointerType::GetUniform(intAsUnsigned ? AtomicType::UniformUInt64 :
144  AtomicType::UniformInt64);
145  else if (t == LLVMTypes::FloatPointerType)
146  return PointerType::GetUniform(AtomicType::UniformFloat);
147  else if (t == LLVMTypes::DoublePointerType)
148  return PointerType::GetUniform(AtomicType::UniformDouble);
149 
150  // pointers to varying
151  else if (t == LLVMTypes::Int8VectorPointerType)
152  return PointerType::GetUniform(intAsUnsigned ? AtomicType::VaryingUInt8 :
153  AtomicType::VaryingInt8);
154  else if (t == LLVMTypes::Int16VectorPointerType)
155  return PointerType::GetUniform(intAsUnsigned ? AtomicType::VaryingUInt16 :
156  AtomicType::VaryingInt16);
157  else if (t == LLVMTypes::Int32VectorPointerType)
158  return PointerType::GetUniform(intAsUnsigned ? AtomicType::VaryingUInt32 :
159  AtomicType::VaryingInt32);
160  else if (t == LLVMTypes::Int64VectorPointerType)
161  return PointerType::GetUniform(intAsUnsigned ? AtomicType::VaryingUInt64 :
162  AtomicType::VaryingInt64);
163  else if (t == LLVMTypes::FloatVectorPointerType)
164  return PointerType::GetUniform(AtomicType::VaryingFloat);
165  else if (t == LLVMTypes::DoubleVectorPointerType)
166  return PointerType::GetUniform(AtomicType::VaryingDouble);
167 
168  return NULL;
169 }

191 /** Given an LLVM function declaration, synthesize the equivalent ispc
192  symbol for the function (if possible). Returns true on success, false
193  on failure.
194  */
195 static bool
196 lCreateISPCSymbol(llvm::Function *func, SymbolTable *symbolTable) {
197  SourcePos noPos;
198  noPos.name = "__stdlib";
199 
200  const llvm::FunctionType *ftype = func->getFunctionType();
201  std::string name = func->getName();
202 
203  if (name.size() < 3 || name[0] != '_' || name[1] != '_')
204  return false;
205 
206  Debug(SourcePos(), "Attempting to create ispc symbol for function \"%s\".",
207  name.c_str());
208 
209  // An unfortunate hack: we want this builtin function to have the
210  // signature "int __sext_varying_bool(bool)", but the ispc function
211  // symbol creation code below assumes that any LLVM vector of i32s is a
212  // varying int32. Here, we need that to be interpreted as a varying
213  // bool, so just have a one-off override for that one...
214  if (g->target->getMaskBitCount() != 1 && name == "__sext_varying_bool") {
215  const Type *returnType = AtomicType::VaryingInt32;
216  llvm::SmallVector<const Type *, 8> argTypes;
217  argTypes.push_back(AtomicType::VaryingBool);
218 
219  FunctionType *funcType = new FunctionType(returnType, argTypes, noPos);
220 
221  Symbol *sym = new Symbol(name, noPos, funcType);
222  sym->function = func;
223  symbolTable->AddFunction(sym);
224  return true;
225  }
226 
227  // If the function has any parameters with integer types, we'll make
228  // two Symbols for two overloaded versions of the function, one with
229  // all of the integer types treated as signed integers and one with all
230  // of them treated as unsigned.
231  for (int i = 0; i < 2; ++i) {
232  bool intAsUnsigned = (i == 1);
233 
234  const Type *returnType = lLLVMTypeToISPCType(ftype->getReturnType(),
235  intAsUnsigned);
236  if (returnType == NULL) {
237  Debug(SourcePos(), "Failed: return type not representable for "
238  "builtin %s.", name.c_str());
239  // return type not representable in ispc -> not callable from ispc
240  return false;
241  }
242 
243  // Iterate over the arguments and try to find their equivalent ispc
244  // types. Track if any of the arguments has an integer type.
245  bool anyIntArgs = false;
246  llvm::SmallVector<const Type *, 8> argTypes;
247  for (unsigned int j = 0; j < ftype->getNumParams(); ++j) {
248  const llvm::Type *llvmArgType = ftype->getParamType(j);
249  const Type *type = lLLVMTypeToISPCType(llvmArgType, intAsUnsigned);
250  if (type == NULL) {
251  Debug(SourcePos(), "Failed: type of parameter %d not "
252  "representable for builtin %s", j, name.c_str());
253  return false;
254  }
255  anyIntArgs |=
256  (Type::Equal(type, lLLVMTypeToISPCType(llvmArgType, !intAsUnsigned)) == false);
257  argTypes.push_back(type);
258  }
259 
260  // Always create the symbol the first time through, in particular
261  // so that we get symbols for things with no integer types!
262  if (i == 0 || anyIntArgs == true)
263  lCreateSymbol(name, returnType, argTypes, ftype, func, symbolTable);
264  }
265 
266  return true;
267 }

292 /** In many of the builtins-*.ll files, we have declarations of various LLVM
293  intrinsics that are then used in the implementation of various target-
294  specific functions. This function loops over all of the intrinsic
295  declarations and makes sure that the signature we have in our .ll file
296  matches the signature of the actual intrinsic.
297 */
298 static void
299 lCheckModuleIntrinsics(llvm::Module *module) {
300  llvm::Module::iterator iter;
301  for (iter = module->begin(); iter != module->end(); ++iter) {
302 #if ISPC_LLVM_VERSION <= ISPC_LLVM_3_7 /* 3.2, 3.3, 3.4, 3.5, 3.6, 3.7 */
303  llvm::Function *func = iter;
304 #else /* LLVM 3.8+ */
305  llvm::Function *func = &*iter;
306 #endif
307  if (!func->isIntrinsic())
308  continue;
309 
310  const std::string funcName = func->getName().str();
311  // Work around http://llvm.org/bugs/show_bug.cgi?id=10438; only
312  // check the llvm.x86.* intrinsics for now...
313  if (!strncmp(funcName.c_str(), "llvm.x86.", 9)) {
314  llvm::Intrinsic::ID id = (llvm::Intrinsic::ID)func->getIntrinsicID();
315  if (id == 0) fprintf(stderr, "FATAL: intrinsic is not found: %s \n", funcName.c_str());
316  Assert(id != 0);
317  llvm::Type *intrinsicType =
318  llvm::Intrinsic::getType(*g->ctx, id);
319  intrinsicType = llvm::PointerType::get(intrinsicType, 0);
320  Assert(func->getType() == intrinsicType);
321  }
322  }
323 }

780 /** This utility function takes serialized binary LLVM bitcode and adds its
781  definitions to the given module. Functions in the bitcode that can be
782  mapped to ispc functions are also added to the symbol table.
783 
784  @param bitcode Binary LLVM bitcode (e.g. the contents of a *.bc file)
785  @param length Length of the bitcode buffer
786  @param module Module to link the bitcode into
787  @param symbolTable Symbol table to add definitions to
788  */
789 void
790 AddBitcodeToModule(const unsigned char *bitcode, int length,
791  llvm::Module *module, SymbolTable *symbolTable, bool warn) {
792  llvm::StringRef sb = llvm::StringRef((char *)bitcode, length);
793 #if ISPC_LLVM_VERSION <= ISPC_LLVM_3_5
794  llvm::MemoryBuffer *bcBuf = llvm::MemoryBuffer::getMemBuffer(sb);
795 #else // LLVM 3.6+
796  llvm::MemoryBufferRef bcBuf = llvm::MemoryBuffer::getMemBuffer(sb)->getMemBufferRef();
797 #endif
798 
799 #if ISPC_LLVM_VERSION >= ISPC_LLVM_3_7 // LLVM 3.7+
800  llvm::ErrorOr<std::unique_ptr<llvm::Module>> ModuleOrErr = llvm::parseBitcodeFile(bcBuf, *g->ctx);
801  if (std::error_code EC = ModuleOrErr.getError())
802  Error(SourcePos(), "Error parsing stdlib bitcode: %s", EC.message().c_str());
803  else {
804  llvm::Module *bcModule = ModuleOrErr.get().release();
805 #elif ISPC_LLVM_VERSION == ISPC_LLVM_3_5 || ISPC_LLVM_VERSION == ISPC_LLVM_3_6
806  llvm::ErrorOr<llvm::Module *> ModuleOrErr = llvm::parseBitcodeFile(bcBuf, *g->ctx);
807  if (std::error_code EC = ModuleOrErr.getError())
808  Error(SourcePos(), "Error parsing stdlib bitcode: %s", EC.message().c_str());
809  else {
810  llvm::Module *bcModule = ModuleOrErr.get();
811 #else // LLVM 3.2 - 3.4
812  std::string bcErr;
813  llvm::Module *bcModule = llvm::ParseBitcodeFile(bcBuf, *g->ctx, &bcErr);
814  if (!bcModule)
815  Error(SourcePos(), "Error parsing stdlib bitcode: %s", bcErr.c_str());
816  else {
817 #endif
818  // FIXME: this feels like a bad idea, but the issue is that when we
819  // set the llvm::Module's target triple in the ispc Module::Module
820  // constructor, we start by calling llvm::sys::getHostTriple() (and
821  // then change the arch if needed). Somehow that ends up giving us
822  // strings like 'x86_64-apple-darwin11.0.0', while the stuff we
823  // compile to bitcode with clang has module triples like
824  // 'i386-apple-macosx10.7.0'. And then LLVM issues a warning about
825  // linking together modules with incompatible target triples..
826  llvm::Triple mTriple(m->module->getTargetTriple());
827  llvm::Triple bcTriple(bcModule->getTargetTriple());
828  Debug(SourcePos(), "module triple: %s\nbitcode triple: %s\n",
829  mTriple.str().c_str(), bcTriple.str().c_str());
830 #if defined(ISPC_ARM_ENABLED) && !defined(__arm__)
831  // FIXME: More ugly and dangerous stuff. We really haven't set up
832  // proper build and runtime infrastructure for ispc to do
833  // cross-compilation, yet it's at minimum useful to be able to emit
834  // ARM code from x86 for ispc development. One side-effect is that
835  // when the build process turns builtins/builtins.c to LLVM bitcode
836  // for us to link in at runtime, that bitcode has been compiled for
837  // an IA target, which in turn causes the checks in the following
838  // code to (appropraitely) fail.
839  //
840  // In order to be able to have some ability to generate ARM code on
841  // IA, we'll just skip those tests in that case and allow the
842  // setTargetTriple() and setDataLayout() calls below to shove in
843  // the values for an ARM target. This maybe won't cause problems
844  // in the generated code, since bulitins.c doesn't do anything too
845  // complex w.r.t. struct layouts, etc.
846  if (g->target->getISA() != Target::NEON32 &&
847  g->target->getISA() != Target::NEON16 &&
848  g->target->getISA() != Target::NEON8)
849 #endif // !__arm__
850 #ifdef ISPC_NVPTX_ENABLED
851  if (g->target->getISA() != Target::NVPTX)
852 #endif /* ISPC_NVPTX_ENABLED */
853  {
854  Assert(bcTriple.getArch() == llvm::Triple::UnknownArch ||
855  mTriple.getArch() == bcTriple.getArch());
856  Assert(bcTriple.getVendor() == llvm::Triple::UnknownVendor ||
857  mTriple.getVendor() == bcTriple.getVendor());
858 
859  // We unconditionally set module DataLayout to library, but we must
860  // ensure that library and module DataLayouts are compatible.
861  // If they are not, we should recompile the library for problematic
862  // architecture and investigate what happened.
863  // Generally we allow library DataLayout to be subset of module
864  // DataLayout or library DataLayout to be empty.
865 #if ISPC_LLVM_VERSION >= ISPC_LLVM_3_5
866  if (!VerifyDataLayoutCompatibility(module->getDataLayoutStr(),
867  bcModule->getDataLayoutStr())
868  && warn) {
869  Warning(SourcePos(), "Module DataLayout is incompatible with "
870  "library DataLayout:\n"
871  "Module DL: %s\n"
872  "Library DL: %s\n",
873  module->getDataLayoutStr().c_str(),
874  bcModule->getDataLayoutStr().c_str());
875  }
876 #else
877  if (!VerifyDataLayoutCompatibility(module->getDataLayout(),
878  bcModule->getDataLayout())
879  && warn) {
880  Warning(SourcePos(), "Module DataLayout is incompatible with "
881  "library DataLayout:\n"
882  "Module DL: %s\n"
883  "Library DL: %s\n",
884  module->getDataLayout().c_str(),
885  bcModule->getDataLayout().c_str());
886  }
887 #endif
888  }
889 
890  bcModule->setTargetTriple(mTriple.str());
891  bcModule->setDataLayout(module->getDataLayout());
892 
893 #if ISPC_LLVM_VERSION <= ISPC_LLVM_3_5 // 3.2-3.5
894  std::string(linkError);
895 
896  if (llvm::Linker::LinkModules(module, bcModule,
897  llvm::Linker::DestroySource,
898  &linkError))
899  Error(SourcePos(), "Error linking stdlib bitcode: %s", linkError.c_str());
900 #elif ISPC_LLVM_VERSION <= ISPC_LLVM_3_7 // 3.6-3.7
901  llvm::Linker::LinkModules(module, bcModule);
902 #else // LLVM 3.8+
903  // A hack to move over declaration, which have no definition.
904  // New linker is kind of smart and think it knows better what to do, so
905  // it removes unused declarations without definitions.
906  // This trick should be legal, as both modules use the same LLVMContext.
907  for (llvm::Function& f : *bcModule) {
908  if (f.isDeclaration()) {
909  // Declarations with uses will be moved by Linker.
910  if (f.getNumUses() > 0)
911  continue;
912  module->getOrInsertFunction(f.getName(), f.getFunctionType(),
913  f.getAttributes());
914  }
915  }
916 
917  std::unique_ptr<llvm::Module> M(bcModule);
918  if (llvm::Linker::linkModules(*module, std::move(M))) {
919  Error(SourcePos(), "Error linking stdlib bitcode.");
920  }
921 #endif
922 
923  lSetInternalFunctions(module);
924  if (symbolTable != NULL)
925  lAddModuleSymbols(module, symbolTable);
926  lCheckModuleIntrinsics(module);
927  }

1112  bool warn = g->target->getISA() != Target::GENERIC;
1113 
1114 #define EXPORT_MODULE_COND_WARN(export_module, warnings) \
1115  extern unsigned char export_module[]; \
1116  extern int export_module##_length; \

1119

1121  extern unsigned char export_module[]; \
1122  extern int export_module##_length; \
1123  AddBitcodeToModule(export_module, export_module##_length, \
1124  module, symbolTable, true);
1125 
1126  // Add the definitions from the compiled builtins.c file.
1127  // When compiling for "generic" target family, data layout warnings for
1128  // "builtins_bitcode_c" have to be switched off: its DL is incompatible
1129  // with the DL of "generic". Anyway, AddBitcodeToModule() corrects this
1130  // automatically if DLs differ (by copying module`s DL to export`s DL).
1131  if (runtime32) {
1132  EXPORT_MODULE_COND_WARN(builtins_bitcode_c_32, warn);
1133  }
