49 /** Rounds up the given value to the next power of two, if it isn't a power
50  of two already. */
51 inline uint32_t RoundUpPow2(uint32_t v) {
52  v--;
53  v |= v >> 1;
54  v |= v >> 2;
55  v |= v >> 4;
56  v |= v >> 8;
57  v |= v >> 16;
58  return v+1;
59 }
