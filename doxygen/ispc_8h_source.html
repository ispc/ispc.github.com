355 /** @brief Structure that collects optimization options
356 
357  This structure collects all of the options related to optimization of
358  generated code.
359 */
360 struct Opt {
361  Opt();
362 
363  /** Optimization level. Currently, the only valid values are 0,
364  indicating essentially no optimization, and 1, indicating as much
365  optimization as possible. */
366  int level;
367 
368  /** Indicates whether "fast and loose" numerically unsafe optimizations
369  should be performed. This is false by default. */
370  bool fastMath;
371 
372  /** Indicates whether an vector load should be issued for masked loads
373  on platforms that don't have a native masked vector load. (This may
374  lead to accessing memory up to programCount-1 elements past the end of
375  arrays, so is unsafe in general.) */
376  bool fastMaskedVload;
377 
378  /** Indicates when loops should be unrolled (when doing so seems like
379  it will make sense. */
380  bool unrollLoops;
381 
382  /** Indicates if addressing math will be done with 32-bit math, even on
383  64-bit systems. (This is generally noticably more efficient,
384  though at the cost of addressing >2GB).
385  */
386  bool force32BitAddressing;
387 
388  /** Indicates whether Assert() statements should be ignored (for
389  performance in the generated code). */
390  bool disableAsserts;
391 
392  /** Indicates whether FMA instructions should be disabled (on targets
393  that support them). */
394  bool disableFMA;
395 
396  /** Always generate aligned vector load/store instructions; this
397  implies a guarantee that all dynamic access through pointers that
398  becomes a vector load/store will be a cache-aligned sequence of
399  locations. */
400  bool forceAlignedMemory;
401 
402  /** If enabled, disables the various optimizations that kick in when
403  the execution mask can be determined to be "all on" at compile
404  time. */
405  bool disableMaskAllOnOptimizations;
406 
407  /** If enabled, the various __pseudo* memory ops (gather/scatter,
408  masked load/store) are left in their __pseudo* form, for better
409  understanding of the structure of generated code when reading
410  it. */
411  bool disableHandlePseudoMemoryOps;
412 
413  /** On targets that don't have a masked store instruction but do have a
414  blending instruction, by default, we simulate masked stores by
415  loading the old value, blending, and storing the result. This can
416  potentially be unsafe in multi-threaded code, in that it writes to
417  locations that aren't supposed to be written to. Setting this
418  value to true disables this work-around, and instead implements
419  masked stores by 'scalarizing' them, so that we iterate over the
420  ISIMD lanes and do a scalar write for the ones that are running. */
421  bool disableBlendedMaskedStores;
422 
423  /** Disables the 'coherent control flow' constructs in the
424  language. (e.g. this causes "cif" statements to be demoted to "if"
425  statements.) This is likely only useful for measuring the impact
426  of coherent control flow. */
427  bool disableCoherentControlFlow;
428 
429  /** Disables uniform control flow optimizations (e.g. this changes an
430  "if" statement with a uniform condition to have a varying
431  condition). This is likely only useful for measuring the impact of
432  uniform control flow. */
433  bool disableUniformControlFlow;
434 
435  /** Disables the backend optimizations related to gather/scatter
436  (e.g. transforming gather from sequential locations to an unaligned
437  load, etc.) This is likely only useful for measuring the impact of
438  these optimizations. */
439  bool disableGatherScatterOptimizations;
440 
441  /** Disables the optimization that demotes masked stores to regular
442  stores when the store is happening at the same control flow level
443  where the variable was declared. This is likely only useful for
444  measuring the impact of this optimization. */
445  bool disableMaskedStoreToStore;
446 
447  /** Disables the optimization that detects when the execution mask is
448  all on and emits code for gathers and scatters that doesn't loop
449  over the SIMD lanes but just does the scalar loads and stores
450  directly. */
451  bool disableGatherScatterFlattening;
452 
453  /** Disables the optimizations that detect when arrays are being
454  indexed with 'uniform' values and issue scalar loads/stores rather
455  than gathers/scatters. This is likely only useful for measuring
456  the impact of this optimization. */
457  bool disableUniformMemoryOptimizations;
458 
459  /** Disables optimizations that coalesce incoherent scalar memory
460  access from gathers into wider vector operations, when possible. */
461  bool disableCoalescing;
462 };
