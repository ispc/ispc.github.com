386 /** @brief Structure that collects optimization options
387 
388  This structure collects all of the options related to optimization of
389  generated code.
390 */
391 struct Opt {
392  Opt();
393 
394  /** Optimization level. Currently, the only valid values are 0,
395  indicating essentially no optimization, and 1, indicating as much
396  optimization as possible. */
397  int level;
398 
399  /** Indicates whether "fast and loose" numerically unsafe optimizations
400  should be performed. This is false by default. */
401  bool fastMath;
402 
403  /** Indicates whether an vector load should be issued for masked loads
404  on platforms that don't have a native masked vector load. (This may
405  lead to accessing memory up to programCount-1 elements past the end of
406  arrays, so is unsafe in general.) */
407  bool fastMaskedVload;
408 
409  /** Indicates when loops should be unrolled (when doing so seems like
410  it will make sense. */
411  bool unrollLoops;
412 
413  /** Indicates if addressing math will be done with 32-bit math, even on
414  64-bit systems. (This is generally noticably more efficient,
415  though at the cost of addressing >2GB).
416  */
417  bool force32BitAddressing;
418 
419  /** Indicates whether Assert() statements should be ignored (for
420  performance in the generated code). */
421  bool disableAsserts;
422 
423  /** Indicates whether FMA instructions should be disabled (on targets
424  that support them). */
425  bool disableFMA;
426 
427  /** Always generate aligned vector load/store instructions; this
428  implies a guarantee that all dynamic access through pointers that
429  becomes a vector load/store will be a cache-aligned sequence of
430  locations. */
431  bool forceAlignedMemory;
432 
433  /** If enabled, disables the various optimizations that kick in when
434  the execution mask can be determined to be "all on" at compile
435  time. */
436  bool disableMaskAllOnOptimizations;
437 
438  /** If enabled, the various __pseudo* memory ops (gather/scatter,
439  masked load/store) are left in their __pseudo* form, for better
440  understanding of the structure of generated code when reading
441  it. */
442  bool disableHandlePseudoMemoryOps;
443 
444  /** On targets that don't have a masked store instruction but do have a
445  blending instruction, by default, we simulate masked stores by
446  loading the old value, blending, and storing the result. This can
447  potentially be unsafe in multi-threaded code, in that it writes to
448  locations that aren't supposed to be written to. Setting this
449  value to true disables this work-around, and instead implements
450  masked stores by 'scalarizing' them, so that we iterate over the
451  ISIMD lanes and do a scalar write for the ones that are running. */
452  bool disableBlendedMaskedStores;
453 
454  /** Disables the 'coherent control flow' constructs in the
455  language. (e.g. this causes "cif" statements to be demoted to "if"
456  statements.) This is likely only useful for measuring the impact
457  of coherent control flow. */
458  bool disableCoherentControlFlow;
459 
460  /** Disables uniform control flow optimizations (e.g. this changes an
461  "if" statement with a uniform condition to have a varying
462  condition). This is likely only useful for measuring the impact of
463  uniform control flow. */
464  bool disableUniformControlFlow;
465 
466  /** Disables the backend optimizations related to gather/scatter
467  (e.g. transforming gather from sequential locations to an unaligned
468  load, etc.) This is likely only useful for measuring the impact of
469  these optimizations. */
470  bool disableGatherScatterOptimizations;
471 
472  /** Disables the optimization that demotes masked stores to regular
473  stores when the store is happening at the same control flow level
474  where the variable was declared. This is likely only useful for
475  measuring the impact of this optimization. */
476  bool disableMaskedStoreToStore;
477 
478  /** Disables the optimization that detects when the execution mask is
479  all on and emits code for gathers and scatters that doesn't loop
480  over the SIMD lanes but just does the scalar loads and stores
481  directly. */
482  bool disableGatherScatterFlattening;
483 
484  /** Disables the optimizations that detect when arrays are being
485  indexed with 'uniform' values and issue scalar loads/stores rather
486  than gathers/scatters. This is likely only useful for measuring
487  the impact of this optimization. */
488  bool disableUniformMemoryOptimizations;
489 
490  /** Disables optimizations that coalesce incoherent scalar memory
491  access from gathers into wider vector operations, when possible. */
492  bool disableCoalescing;
493 };
