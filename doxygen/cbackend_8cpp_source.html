128  void run(const llvm::Module &M) {
129  // Get types from global variables.
130  for (llvm::Module::const_global_iterator I = M.global_begin(),
131  E = M.global_end(); I != E; ++I) {
132  incorporateType(I->getType());
133  if (I->hasInitializer())
134  incorporateValue(I->getInitializer());
135  }
136 
137  // Get types from aliases.
138  for (llvm::Module::const_alias_iterator I = M.alias_begin(),
139  E = M.alias_end(); I != E; ++I) {
140  incorporateType(I->getType());
141  if (const llvm::Value *Aliasee = I->getAliasee())
142  incorporateValue(Aliasee);
143  }
144 
145  llvm::SmallVector<std::pair<unsigned, llvm::MDNode*>, 4> MDForInst;
146 
147  // Get types from functions.
148  for (llvm::Module::const_iterator FI = M.begin(), E = M.end(); FI != E; ++FI) {
149  incorporateType(FI->getType());
150 
151  for (llvm::Function::const_iterator BB = FI->begin(), E = FI->end();
152  BB != E;++BB)
153  for (llvm::BasicBlock::const_iterator II = BB->begin(),
154  E = BB->end(); II != E; ++II) {
155  const llvm::Instruction &I = *II;
156 
157  // Operands of SwitchInsts changed format after 3.1
158  // Seems like there ought to be better way to do what we
159  // want here. For now, punt on SwitchInsts.
160  if (llvm::isa<llvm::SwitchInst>(&I)) continue;
161 
162  // Incorporate the type of the instruction and all its operands.
163  incorporateType(I.getType());
164  for (llvm::User::const_op_iterator OI = I.op_begin(), OE = I.op_end();
165  OI != OE; ++OI)
166  incorporateValue(*OI);
167 
168  // Incorporate types hiding in metadata.
169  I.getAllMetadataOtherThanDebugLoc(MDForInst);
170  for (unsigned i = 0, e = MDForInst.size(); i != e; ++i)
171  incorporateMDNode(MDForInst[i].second);
172  MDForInst.clear();
173  }
174  }
175 
176  for (llvm::Module::const_named_metadata_iterator I = M.named_metadata_begin(),
177  E = M.named_metadata_end(); I != E; ++I) {
178  const llvm::NamedMDNode *NMD = I;
179  for (unsigned i = 0, e = NMD->getNumOperands(); i != e; ++i)
180  incorporateMDNode(NMD->getOperand(i));
181  }
182  }
183 
184  private:
185  void incorporateType(llvm::Type *Ty) {
186  // Check to see if we're already visited this type.
187  if (!VisitedTypes.insert(Ty).second)
188  return;
189 
190  if (llvm::ArrayType *ATy = llvm::dyn_cast<llvm::ArrayType>(Ty))
191  ArrayTypes.push_back(ATy);
192 
193  // Recursively walk all contained types.
194  for (llvm::Type::subtype_iterator I = Ty->subtype_begin(),
195  E = Ty->subtype_end(); I != E; ++I)
196  incorporateType(*I);
197  }
198 
199  /// incorporateValue - This method is used to walk operand lists finding
200  /// types hiding in constant expressions and other operands that won't be
201  /// walked in other ways. GlobalValues, basic blocks, instructions, and
202  /// inst operands are all explicitly enumerated.
203  void incorporateValue(const llvm::Value *V) {
204  if (const llvm::MDNode *M = llvm::dyn_cast<llvm::MDNode>(V))
205  return incorporateMDNode(M);
206  if (!llvm::isa<llvm::Constant>(V) || llvm::isa<llvm::GlobalValue>(V)) return;
207 
208  // Already visited?
209  if (!VisitedConstants.insert(V).second)
210  return;
211 
212  // Check this type.
213  incorporateType(V->getType());
214 
215  // Look in operands for types.
216  const llvm::User *U = llvm::cast<llvm::User>(V);
217  for (llvm::Constant::const_op_iterator I = U->op_begin(),
218  E = U->op_end(); I != E;++I)
219  incorporateValue(*I);
220  }
221 
222  void incorporateMDNode(const llvm::MDNode *V) {
223 
224  // Already visited?
225  if (!VisitedConstants.insert(V).second)
226  return;
227 
228  // Look in operands for types.
229  for (unsigned i = 0, e = V->getNumOperands(); i != e; ++i)
230  if (llvm::Value *Op = V->getOperand(i))
231  incorporateValue(Op);
232  }
233  };

435  // isInlinableInst - Attempt to inline instructions into their uses to build
436  // trees as much as possible. To do this, we have to consistently decide
437  // what is acceptable to inline, so that variable declarations don't get
438  // printed and an extra copy of the expr is not emitted.
439  //
440  static bool isInlinableInst(const llvm::Instruction &I) {
441  // Always inline cmp instructions, even if they are shared by multiple
442  // expressions. GCC generates horrible code if we don't.
443  if (llvm::isa<llvm::CmpInst>(I) && llvm::isa<llvm::VectorType>(I.getType()) == false)
444  return true;
445 
446  // Must be an expression, must be used exactly once. If it is dead, we
447  // emit it inline where it would go.
448  if (I.getType() == llvm::Type::getVoidTy(I.getContext()) || !I.hasOneUse() ||
449  llvm::isa<llvm::TerminatorInst>(I) || llvm::isa<llvm::CallInst>(I) || llvm::isa<llvm::PHINode>(I) ||
450  llvm::isa<llvm::LoadInst>(I) || llvm::isa<llvm::VAArgInst>(I) || llvm::isa<llvm::InsertElementInst>(I) ||
451  llvm::isa<llvm::InsertValueInst>(I) || llvm::isa<llvm::ExtractValueInst>(I) || llvm::isa<llvm::SelectInst>(I))
452  // Don't inline a load across a store or other bad things!
453  return false;
454 
455  // Must not be used in inline asm, extractelement, or shufflevector.
456  if (I.hasOneUse()) {
457  const llvm::Instruction &User = llvm::cast<llvm::Instruction>(*I.use_back());
458  if (isInlineAsm(User) || llvm::isa<llvm::ExtractElementInst>(User) ||
459  llvm::isa<llvm::ShuffleVectorInst>(User) || llvm::isa<llvm::AtomicRMWInst>(User) ||
460  llvm::isa<llvm::AtomicCmpXchgInst>(User))
461  return false;
462  }
463 
464  // Only inline instruction it if it's use is in the same BB as the inst.
465  return I.getParent() == llvm::cast<llvm::Instruction>(I.use_back())->getParent();
466  }

993 // isFPCSafeToPrint - Returns true if we may assume that CFP may be written out
994 // textually as a double (rather than as a reference to a stack-allocated
995 // variable). We decide this by converting CFP to a string and back into a
996 // double, and then checking whether the conversion results in a bit-equal
997 // double to the original value of CFP. This depends on us and the target C
998 // compiler agreeing on the conversion process (which is pretty likely since we
999 // only deal in IEEE FP).
1000 //
1001 static bool isFPCSafeToPrint(const llvm::ConstantFP *CFP) {
1002  bool ignored;
1003  // Do long doubles in hex for now.
1004  if (CFP->getType() != llvm::Type::getFloatTy(CFP->getContext()) &&
1005  CFP->getType() != llvm::Type::getDoubleTy(CFP->getContext()))
1006  return false;
1007  llvm::APFloat APF = llvm::APFloat(CFP->getValueAPF()); // copy
1008  if (CFP->getType() == llvm::Type::getFloatTy(CFP->getContext()))
1009  APF.convert(llvm::APFloat::IEEEdouble, llvm::APFloat::rmNearestTiesToEven, &ignored);
1010 #if HAVE_PRINTF_A && ENABLE_CBE_PRINTF_A
1011  char Buffer[100];
1012  sprintf(Buffer, "%a", APF.convertToDouble());
1013  if (!strncmp(Buffer, "0x", 2) ||
1014  !strncmp(Buffer, "-0x", 3) ||
1015  !strncmp(Buffer, "+0x", 3))
1016  return APF.bitwiseIsEqual(llvm::APFloat(atof(Buffer)));
1017  return false;
1018 #else
1019  std::string StrVal = ftostr(APF);
1020 
1021  while (StrVal[0] == ' ')
1022  StrVal.erase(StrVal.begin());
1023 
1024  // Check to make sure that the stringized number is not some string like "Inf"
1025  // or NaN. Check that the string matches the "[-+]?[0-9]" regex.
1026  if ((StrVal[0] >= '0' && StrVal[0] <= '9') ||
1027  ((StrVal[0] == '-' || StrVal[0] == '+') &&
1028  (StrVal[1] >= '0' && StrVal[1] <= '9')))
1029  // Reparse stringized version!
1030  return APF.bitwiseIsEqual(llvm::APFloat(atof(StrVal.c_str())));
1031  return false;
1032 #endif
1033 }
