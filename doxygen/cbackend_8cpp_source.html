128  void run(const llvm::Module &M) {
129  // Get types from global variables.
130  for (llvm::Module::const_global_iterator I = M.global_begin(),
131  E = M.global_end(); I != E; ++I) {
132  incorporateType(I->getType());
133  if (I->hasInitializer())
134  incorporateValue(I->getInitializer());
135  }
136 
137  // Get types from aliases.
138  for (llvm::Module::const_alias_iterator I = M.alias_begin(),
139  E = M.alias_end(); I != E; ++I) {
140  incorporateType(I->getType());
141  if (const llvm::Value *Aliasee = I->getAliasee())
142  incorporateValue(Aliasee);
143  }
144 
145  llvm::SmallVector<std::pair<unsigned, llvm::MDNode*>, 4> MDForInst;
146 
147  // Get types from functions.
148  for (llvm::Module::const_iterator FI = M.begin(), E = M.end(); FI != E; ++FI) {
149  incorporateType(FI->getType());
150 
151  for (llvm::Function::const_iterator BB = FI->begin(), E = FI->end();
152  BB != E;++BB)
153  for (llvm::BasicBlock::const_iterator II = BB->begin(),
154  E = BB->end(); II != E; ++II) {
155  const llvm::Instruction &I = *II;
156 
157  // Operands of SwitchInsts changed format after 3.1
158  // Seems like there ought to be better way to do what we
159  // want here. For now, punt on SwitchInsts.
160  if (llvm::isa<llvm::SwitchInst>(&I)) continue;
161 
162  // Incorporate the type of the instruction and all its operands.
163  incorporateType(I.getType());
164  for (llvm::User::const_op_iterator OI = I.op_begin(), OE = I.op_end();
165  OI != OE; ++OI)
166  incorporateValue(*OI);
167 
168  // Incorporate types hiding in metadata.
169  I.getAllMetadataOtherThanDebugLoc(MDForInst);
170  for (unsigned i = 0, e = MDForInst.size(); i != e; ++i)
171  incorporateMDNode(MDForInst[i].second);
172  MDForInst.clear();
173  }
174  }
175 
176  for (llvm::Module::const_named_metadata_iterator I = M.named_metadata_begin(),
177  E = M.named_metadata_end(); I != E; ++I) {
178  const llvm::NamedMDNode *NMD = I;
179  for (unsigned i = 0, e = NMD->getNumOperands(); i != e; ++i)
180  incorporateMDNode(NMD->getOperand(i));
181  }
182  }
183 
184  private:
185  void incorporateType(llvm::Type *Ty) {
186  // Check to see if we're already visited this type.
187  if (!VisitedTypes.insert(Ty).second)
188  return;
189 
190  if (llvm::ArrayType *ATy = llvm::dyn_cast<llvm::ArrayType>(Ty))
191  ArrayTypes.push_back(ATy);
192 
193  // Recursively walk all contained types.
194  for (llvm::Type::subtype_iterator I = Ty->subtype_begin(),
195  E = Ty->subtype_end(); I != E; ++I)
196  incorporateType(*I);
197  }
198 
199  /// incorporateValue - This method is used to walk operand lists finding
200  /// types hiding in constant expressions and other operands that won't be
201  /// walked in other ways. GlobalValues, basic blocks, instructions, and
202  /// inst operands are all explicitly enumerated.
203  void incorporateValue(const llvm::Value *V) {
204  if (const llvm::MDNode *M = llvm::dyn_cast<llvm::MDNode>(V))
205  return incorporateMDNode(M);
206  if (!llvm::isa<llvm::Constant>(V) || llvm::isa<llvm::GlobalValue>(V)) return;
207 
208  // Already visited?
209  if (!VisitedConstants.insert(V).second)
210  return;
211 
212  // Check this type.
213  incorporateType(V->getType());
214 
215  // Look in operands for types.
216  const llvm::User *U = llvm::cast<llvm::User>(V);
217  for (llvm::Constant::const_op_iterator I = U->op_begin(),
218  E = U->op_end(); I != E;++I)
219  incorporateValue(*I);
220  }
221 
222  void incorporateMDNode(const llvm::MDNode *V) {
223 
224  // Already visited?
225  if (!VisitedConstants.insert(V).second)
226  return;
227 
228  // Look in operands for types.
229  for (unsigned i = 0, e = V->getNumOperands(); i != e; ++i)
230  if (llvm::Value *Op = V->getOperand(i))
231  incorporateValue(Op);
232  }
233  };

439  // isInlinableInst - Attempt to inline instructions into their uses to build
440  // trees as much as possible. To do this, we have to consistently decide
441  // what is acceptable to inline, so that variable declarations don't get
442  // printed and an extra copy of the expr is not emitted.
443  //
444  static bool isInlinableInst(const llvm::Instruction &I) {
445  // Always inline cmp instructions, even if they are shared by multiple
446  // expressions. GCC generates horrible code if we don't.
447  if (llvm::isa<llvm::CmpInst>(I) && llvm::isa<llvm::VectorType>(I.getType()) == false)
448  return true;
449 
450  // Must be an expression, must be used exactly once. If it is dead, we
451  // emit it inline where it would go.
452  if (I.getType() == llvm::Type::getVoidTy(I.getContext()) || !I.hasOneUse() ||
453  llvm::isa<llvm::TerminatorInst>(I) || llvm::isa<llvm::CallInst>(I) || llvm::isa<llvm::PHINode>(I) ||
454  llvm::isa<llvm::LoadInst>(I) || llvm::isa<llvm::VAArgInst>(I) || llvm::isa<llvm::InsertElementInst>(I) ||
455  llvm::isa<llvm::InsertValueInst>(I) || llvm::isa<llvm::ExtractValueInst>(I) || llvm::isa<llvm::SelectInst>(I))
456  // Don't inline a load across a store or other bad things!
457  return false;
458 
459  // Must not be used in inline asm, extractelement, or shufflevector.
460  if (I.hasOneUse()) {
461  const llvm::Instruction &User = llvm::cast<llvm::Instruction>(*I.use_back());
462  if (isInlineAsm(User) || llvm::isa<llvm::ExtractElementInst>(User) ||
463  llvm::isa<llvm::ShuffleVectorInst>(User) || llvm::isa<llvm::AtomicRMWInst>(User) ||
464  llvm::isa<llvm::AtomicCmpXchgInst>(User))
465  return false;
466  }
467 
468  // Only inline instruction it if it's use is in the same BB as the inst.
469  return I.getParent() == llvm::cast<llvm::Instruction>(I.use_back())->getParent();
470  }

562 static std::string CBEMangle(const std::string &S) {
563  std::string Result;
564 
565  for (unsigned i = 0, e = S.size(); i != e; ++i) {
566  if (i+1 != e && ((S[i] == '>' && S[i+1] == '>') ||
567  (S[i] == '<' && S[i+1] == '<'))) {
568  Result += '_';
569  Result += 'A'+(S[i]&15);
570  Result += 'A'+((S[i]>>4)&15);
571  Result += '_';
572  i++;
573  } else if (isalnum(S[i]) || S[i] == '_' || S[i] == '<' || S[i] == '>') {
574  Result += S[i];
575  } else {
576  Result += '_';
577  Result += 'A'+(S[i]&15);
578  Result += 'A'+((S[i]>>4)&15);
579  Result += '_';
580  }
581  }
582  return Result;
583 }

1005 // isFPCSafeToPrint - Returns true if we may assume that CFP may be written out
1006 // textually as a double (rather than as a reference to a stack-allocated
1007 // variable). We decide this by converting CFP to a string and back into a
1008 // double, and then checking whether the conversion results in a bit-equal
1009 // double to the original value of CFP. This depends on us and the target C
1010 // compiler agreeing on the conversion process (which is pretty likely since we
1011 // only deal in IEEE FP).
1012 //
1013 static bool isFPCSafeToPrint(const llvm::ConstantFP *CFP) {
1014  bool ignored;
1015  // Do long doubles in hex for now.
1016  if (CFP->getType() != llvm::Type::getFloatTy(CFP->getContext()) &&
1017  CFP->getType() != llvm::Type::getDoubleTy(CFP->getContext()))
1018  return false;
1019  llvm::APFloat APF = llvm::APFloat(CFP->getValueAPF()); // copy
1020  if (CFP->getType() == llvm::Type::getFloatTy(CFP->getContext()))
1021  APF.convert(llvm::APFloat::IEEEdouble, llvm::APFloat::rmNearestTiesToEven, &ignored);
1022 #if HAVE_PRINTF_A && ENABLE_CBE_PRINTF_A
1023  char Buffer[100];
1024  sprintf(Buffer, "%a", APF.convertToDouble());
1025  if (!strncmp(Buffer, "0x", 2) ||
1026  !strncmp(Buffer, "-0x", 3) ||
1027  !strncmp(Buffer, "+0x", 3))
1028  return APF.bitwiseIsEqual(llvm::APFloat(atof(Buffer)));
1029  return false;
1030 #else
1031  std::string StrVal = ftostr(APF);
1032 
1033  while (StrVal[0] == ' ')
1034  StrVal.erase(StrVal.begin());
1035 
1036  // Check to make sure that the stringized number is not some string like "Inf"
1037  // or NaN. Check that the string matches the "[-+]?[0-9]" regex.
1038  if ((StrVal[0] >= '0' && StrVal[0] <= '9') ||
1039  ((StrVal[0] == '-' || StrVal[0] == '+') &&
1040  (StrVal[1] >= '0' && StrVal[1] <= '9')))
1041  // Reparse stringized version!
1042  return APF.bitwiseIsEqual(llvm::APFloat(atof(StrVal.c_str())));
1043  return false;
1044 #endif
1045 }

1099  case llvm::Instruction::Trunc:
