97 bool
98 SymbolTable::AddVariable(Symbol *symbol) {
99  Assert(symbol != NULL);
100 
101  // Check to see if a symbol of the same name has already been declared.
102  for (int i = (int)variables.size() - 1; i >= 0; --i) {
103  SymbolMapType &sm = *(variables[i]);
104  if (sm.find(symbol->name) != sm.end()) {
105  if (i == (int)variables.size()-1) {
106  // If a symbol of the same name was declared in the
107  // same scope, it's an error.
108  Error(symbol->pos, "Ignoring redeclaration of symbol \"%s\".",
109  symbol->name.c_str());
110  return false;
111  }
112  else {
113  // Otherwise it's just shadowing something else, which
114  // is legal but dangerous..
115  Warning(symbol->pos,
116  "Symbol \"%s\" shadows symbol declared in outer scope.",
117  symbol->name.c_str());
118  (*variables.back())[symbol->name] = symbol;
119  return true;
120  }
121  }
122  }
123 
124  // No matches, so go ahead and add it...
125  (*variables.back())[symbol->name] = symbol;
126  return true;
127 }

130 Symbol *
131 SymbolTable::LookupVariable(const char *name) {
132  // Note that we iterate through the variables vectors backwards, since
133  // we want to search from the innermost scope to the outermost, so that
134  // we get the right symbol if we have multiple variables in different
135  // scopes that shadow each other.
136  for (int i = (int)variables.size() - 1; i >= 0; --i) {
137  SymbolMapType &sm = *(variables[i]);
138  SymbolMapType::iterator iter = sm.find(name);
139  if (iter != sm.end())
140  return iter->second;
141  }
142  return NULL;
143 }

162 SymbolTable::LookupFunction(const char *name, std::vector<Symbol *> *matches) {
163  FunctionMapType::iterator iter = functions.find(name);
164  if (iter != functions.end()) {
165  if (matches == NULL)
166  return true;
167  else {
168  const std::vector<Symbol *> &funcs = iter->second;
169  for (int j = 0; j < (int)funcs.size(); ++j)

172  }
173  return matches ? (matches->size() > 0) : false;

192 SymbolTable::AddType(const char *name, const Type *type, SourcePos pos) {
193  const Type *t = LookupType(name);
194  if (t != NULL && CastType<UndefinedStructType>(t) == NULL) {

218 std::vector<std::string>
219 SymbolTable::ClosestVariableOrFunctionMatch(const char *str) const {
220  // This is a little wasteful, but we'll look through all of the
221  // variable and function symbols and compute the edit distance from the
222  // given string to them. If the edit distance is under maxDelta, then
223  // it goes in the entry of the matches[] array corresponding to its
224  // edit distance.
225  const int maxDelta = 2;
226  std::vector<std::string> matches[maxDelta+1];
227 
228  for (int i = 0; i < (int)variables.size(); ++i) {
229  const SymbolMapType &sv = *(variables[i]);
230  SymbolMapType::const_iterator iter;
231  for (iter = sv.begin(); iter != sv.end(); ++iter) {
232  const Symbol *sym = iter->second;
233  int dist = StringEditDistance(str, sym->name, maxDelta+1);
234  if (dist <= maxDelta)
235  matches[dist].push_back(sym->name);
236  }
237  }
238 
239  FunctionMapType::const_iterator iter;
240  for (iter = functions.begin(); iter != functions.end(); ++iter) {
241  int dist = StringEditDistance(str, iter->first, maxDelta+1);
242  if (dist <= maxDelta)
243  matches[dist].push_back(iter->first);
244  }
245 
246  // Now, return the first entry of matches[] that is non-empty, if any.
247  for (int i = 0; i <= maxDelta; ++i) {
248  if (matches[i].size())
249  return matches[i];
250  }
251 
252  // Otherwise, no joy.
253  return std::vector<std::string>();
254 }

269 std::vector<std::string>
270 SymbolTable::closestTypeMatch(const char *str, bool structsVsEnums) const {
271  // This follows the same approach as ClosestVariableOrFunctionMatch()
272  // above; compute all edit distances, keep the ones shorter than
273  // maxDelta, return the first non-empty vector of one or more sets of
274  // alternatives with minimal edit distance.
275  const int maxDelta = 2;
276  std::vector<std::string> matches[maxDelta+1];
277 
278  TypeMapType::const_iterator iter;
279  for (iter = types.begin(); iter != types.end(); ++iter) {
280  // Skip over either StructTypes or EnumTypes, depending on the
281  // value of the structsVsEnums parameter
282  bool isEnum = (CastType<EnumType>(iter->second) != NULL);
283  if (isEnum && structsVsEnums)
284  continue;
285  else if (!isEnum && !structsVsEnums)
286  continue;
287 
288  int dist = StringEditDistance(str, iter->first, maxDelta+1);
289  if (dist <= maxDelta)
290  matches[dist].push_back(iter->first);
291  }
292 
293  for (int i = 0; i <= maxDelta; ++i) {
294  if (matches[i].size())
295  return matches[i];
296  }
297  return std::vector<std::string>();
298 }

301 void
302 SymbolTable::Print() {
303  int depth = 0;
304  fprintf(stderr, "Variables:\n----------------\n");
305  for (int i = 0; i < (int)variables.size(); ++i) {
306  SymbolMapType &sm = *(variables[i]);
307  SymbolMapType::iterator iter;
308  for (iter = sm.begin(); iter != sm.end(); ++iter) {
309  fprintf(stderr, "%*c", depth, ' ');
310  Symbol *sym = iter->second;
311  fprintf(stderr, "%s [%s]", sym->name.c_str(),
312  sym->type->GetString().c_str());
313  }
314  fprintf(stderr, "\n");
315  depth += 4;
316  }
317 
318  fprintf(stderr, "Functions:\n----------------\n");
319  FunctionMapType::iterator fiter = functions.begin();
320  while (fiter != functions.end()) {
321  fprintf(stderr, "%s\n", fiter->first.c_str());
322  std::vector<Symbol *> &syms = fiter->second;
323  for (unsigned int j = 0; j < syms.size(); ++j)
324  fprintf(stderr, " %s\n", syms[j]->type->GetString().c_str());
325  ++fiter;
326  }
327 
328  depth = 0;
329  fprintf(stderr, "Named types:\n---------------\n");
330  TypeMapType::iterator siter = types.begin();
331  while (siter != types.end()) {
332  fprintf(stderr, "%*c", depth, ' ');
333  fprintf(stderr, "%s -> %s\n", siter->first.c_str(),
334  siter->second->GetString().c_str());
335  ++siter;
336  }
337 }
