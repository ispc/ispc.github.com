55 /** Types may have uniform, varying, SOA, or unbound variability; this
56  struct is used by Type implementations to record their variability.
57 */
58 struct Variability {
59  enum VarType { Unbound, Uniform, Varying, SOA };
60 
61  Variability(VarType t = Unbound, int w = 0) : type(t), soaWidth(w) { }
62 
63  bool operator==(const Variability &v) const {
64  return v.type == type && v.soaWidth == soaWidth;
65  }
66  bool operator!=(const Variability &v) const {
67  return v.type != type || v.soaWidth != soaWidth;
68  }
69 
70  bool operator==(const VarType &t) const { return type == t; }
71  bool operator!=(const VarType &t) const { return type != t; }
72 
73  std::string GetString() const;
74  std::string MangleString() const;
75 
76  VarType type;
77  int soaWidth;
78 };

412 /** @brief Type implementation for pointers to other types
413 
414  Pointers can have two additional properties beyond their variability
415  and the type of object that they are pointing to. Both of these
416  properties are used for internal bookkeeping and aren't directly
417  accessible from the language.
418 
419  - Slice: pointers that point to data with SOA layout have this
420  property--it indicates that the pointer has two components, where the
421  first (major) component is a regular pointer that points to an
422  instance of the soa<> type being indexed, and where the second
423  (minor) component is an integer that indicates which of the soa
424  slices in that instance the pointer points to.
425 
426  - Frozen: only slice pointers may have this property--it indicates that
427  any further indexing calculations should only be applied to the major
