73 ASTNode *
74 WalkAST(ASTNode *node, ASTPreCallBackFunc preFunc, ASTPostCallBackFunc postFunc,
75  void *data) {
76  if (node == NULL)
77  return node;
78 
79  // Call the callback function
80  if (preFunc != NULL) {
81  if (preFunc(node, data) == false)
82  // The function asked us to not continue recursively, so stop.
83  return node;
84  }
85 
86  ////////////////////////////////////////////////////////////////////////////
87  // Handle Statements
88  if (dynamic_cast<Stmt *>(node) != NULL) {
89  ExprStmt *es;
90  DeclStmt *ds;
91  IfStmt *is;
92  DoStmt *dos;
93  ForStmt *fs;
94  ForeachStmt *fes;
95  ForeachActiveStmt *fas;
96  ForeachUniqueStmt *fus;
97  CaseStmt *cs;
98  DefaultStmt *defs;
99  SwitchStmt *ss;
100  ReturnStmt *rs;
101  LabeledStmt *ls;
102  StmtList *sl;
103  PrintStmt *ps;
104  AssertStmt *as;
105  DeleteStmt *dels;
106  UnmaskedStmt *ums;
107 
108  if ((es = dynamic_cast<ExprStmt *>(node)) != NULL)
109  es->expr = (Expr *)WalkAST(es->expr, preFunc, postFunc, data);
110  else if ((ds = dynamic_cast<DeclStmt *>(node)) != NULL) {
111  for (unsigned int i = 0; i < ds->vars.size(); ++i)
112  ds->vars[i].init = (Expr *)WalkAST(ds->vars[i].init, preFunc,
113  postFunc, data);
114  }
115  else if ((is = dynamic_cast<IfStmt *>(node)) != NULL) {
116  is->test = (Expr *)WalkAST(is->test, preFunc, postFunc, data);
117  is->trueStmts = (Stmt *)WalkAST(is->trueStmts, preFunc,
118  postFunc, data);
119  is->falseStmts = (Stmt *)WalkAST(is->falseStmts, preFunc,
120  postFunc, data);
121  }
122  else if ((dos = dynamic_cast<DoStmt *>(node)) != NULL) {
123  dos->testExpr = (Expr *)WalkAST(dos->testExpr, preFunc,
124  postFunc, data);
125  dos->bodyStmts = (Stmt *)WalkAST(dos->bodyStmts, preFunc,
126  postFunc, data);
127  }
128  else if ((fs = dynamic_cast<ForStmt *>(node)) != NULL) {
129  fs->init = (Stmt *)WalkAST(fs->init, preFunc, postFunc, data);
130  fs->test = (Expr *)WalkAST(fs->test, preFunc, postFunc, data);
131  fs->step = (Stmt *)WalkAST(fs->step, preFunc, postFunc, data);
132  fs->stmts = (Stmt *)WalkAST(fs->stmts, preFunc, postFunc, data);
133  }
134  else if ((fes = dynamic_cast<ForeachStmt *>(node)) != NULL) {
135  for (unsigned int i = 0; i < fes->startExprs.size(); ++i)
136  fes->startExprs[i] = (Expr *)WalkAST(fes->startExprs[i], preFunc,
137  postFunc, data);
138  for (unsigned int i = 0; i < fes->endExprs.size(); ++i)
139  fes->endExprs[i] = (Expr *)WalkAST(fes->endExprs[i], preFunc,
140  postFunc, data);
141  fes->stmts = (Stmt *)WalkAST(fes->stmts, preFunc, postFunc, data);
142  }
143  else if ((fas = dynamic_cast<ForeachActiveStmt *>(node)) != NULL) {
144  fas->stmts = (Stmt *)WalkAST(fas->stmts, preFunc, postFunc, data);
145  }
146  else if ((fus = dynamic_cast<ForeachUniqueStmt *>(node)) != NULL) {
147  fus->expr = (Expr *)WalkAST(fus->expr, preFunc, postFunc, data);
148  fus->stmts = (Stmt *)WalkAST(fus->stmts, preFunc, postFunc, data);
149  }
150  else if ((cs = dynamic_cast<CaseStmt *>(node)) != NULL)
151  cs->stmts = (Stmt *)WalkAST(cs->stmts, preFunc, postFunc, data);
152  else if ((defs = dynamic_cast<DefaultStmt *>(node)) != NULL)
153  defs->stmts = (Stmt *)WalkAST(defs->stmts, preFunc, postFunc, data);
154  else if ((ss = dynamic_cast<SwitchStmt *>(node)) != NULL) {
155  ss->expr = (Expr *)WalkAST(ss->expr, preFunc, postFunc, data);
156  ss->stmts = (Stmt *)WalkAST(ss->stmts, preFunc, postFunc, data);
157  }
158  else if (dynamic_cast<BreakStmt *>(node) != NULL ||
159  dynamic_cast<ContinueStmt *>(node) != NULL ||
160  dynamic_cast<GotoStmt *>(node) != NULL) {
161  // nothing
162  }
163  else if ((ls = dynamic_cast<LabeledStmt *>(node)) != NULL)
164  ls->stmt = (Stmt *)WalkAST(ls->stmt, preFunc, postFunc, data);
165  else if ((rs = dynamic_cast<ReturnStmt *>(node)) != NULL)
166  rs->expr = (Expr *)WalkAST(rs->expr, preFunc, postFunc, data);
167  else if ((sl = dynamic_cast<StmtList *>(node)) != NULL) {
168  std::vector<Stmt *> &sls = sl->stmts;
169  for (unsigned int i = 0; i < sls.size(); ++i)
170  sls[i] = (Stmt *)WalkAST(sls[i], preFunc, postFunc, data);
171  }
172  else if ((ps = dynamic_cast<PrintStmt *>(node)) != NULL)
173  ps->values = (Expr *)WalkAST(ps->values, preFunc, postFunc, data);
174  else if ((as = dynamic_cast<AssertStmt *>(node)) != NULL)
175  as->expr = (Expr *)WalkAST(as->expr, preFunc, postFunc, data);
176  else if ((dels = dynamic_cast<DeleteStmt *>(node)) != NULL)
177  dels->expr = (Expr *)WalkAST(dels->expr, preFunc, postFunc, data);
178  else if ((ums = dynamic_cast<UnmaskedStmt *>(node)) != NULL)
179  ums->stmts = (Stmt *)WalkAST(ums->stmts, preFunc, postFunc, data);
180  else
181  FATAL("Unhandled statement type in WalkAST()");
182  }
183  else {
184  ///////////////////////////////////////////////////////////////////////////
185  // Handle expressions
186  Assert(dynamic_cast<Expr *>(node) != NULL);
187  UnaryExpr *ue;
188  BinaryExpr *be;
189  AssignExpr *ae;
190  SelectExpr *se;
191  ExprList *el;
192  FunctionCallExpr *fce;
193  IndexExpr *ie;
194  MemberExpr *me;
195  TypeCastExpr *tce;
196  ReferenceExpr *re;
197  PtrDerefExpr *ptrderef;
198  RefDerefExpr *refderef;
199  SizeOfExpr *soe;
200  AddressOfExpr *aoe;
201  NewExpr *newe;
202 
203  if ((ue = dynamic_cast<UnaryExpr *>(node)) != NULL)
204  ue->expr = (Expr *)WalkAST(ue->expr, preFunc, postFunc, data);
205  else if ((be = dynamic_cast<BinaryExpr *>(node)) != NULL) {
206  be->arg0 = (Expr *)WalkAST(be->arg0, preFunc, postFunc, data);
207  be->arg1 = (Expr *)WalkAST(be->arg1, preFunc, postFunc, data);
208  }
209  else if ((ae = dynamic_cast<AssignExpr *>(node)) != NULL) {
210  ae->lvalue = (Expr *)WalkAST(ae->lvalue, preFunc, postFunc, data);
211  ae->rvalue = (Expr *)WalkAST(ae->rvalue, preFunc, postFunc, data);
212  }
213  else if ((se = dynamic_cast<SelectExpr *>(node)) != NULL) {
214  se->test = (Expr *)WalkAST(se->test, preFunc, postFunc, data);
215  se->expr1 = (Expr *)WalkAST(se->expr1, preFunc, postFunc, data);
216  se->expr2 = (Expr *)WalkAST(se->expr2, preFunc, postFunc, data);
217  }
218  else if ((el = dynamic_cast<ExprList *>(node)) != NULL) {
219  for (unsigned int i = 0; i < el->exprs.size(); ++i)
220  el->exprs[i] = (Expr *)WalkAST(el->exprs[i], preFunc,
221  postFunc, data);
222  }
223  else if ((fce = dynamic_cast<FunctionCallExpr *>(node)) != NULL) {
224  fce->func = (Expr *)WalkAST(fce->func, preFunc, postFunc, data);
225  fce->args = (ExprList *)WalkAST(fce->args, preFunc, postFunc, data);
226  fce->launchCountExpr = (Expr *)WalkAST(fce->launchCountExpr, preFunc,
227  postFunc, data);
228  }
229  else if ((ie = dynamic_cast<IndexExpr *>(node)) != NULL) {
230  ie->baseExpr = (Expr *)WalkAST(ie->baseExpr, preFunc, postFunc, data);
231  ie->index = (Expr *)WalkAST(ie->index, preFunc, postFunc, data);
232  }
233  else if ((me = dynamic_cast<MemberExpr *>(node)) != NULL)
234  me->expr = (Expr *)WalkAST(me->expr, preFunc, postFunc, data);
235  else if ((tce = dynamic_cast<TypeCastExpr *>(node)) != NULL)
236  tce->expr = (Expr *)WalkAST(tce->expr, preFunc, postFunc, data);
237  else if ((re = dynamic_cast<ReferenceExpr *>(node)) != NULL)
238  re->expr = (Expr *)WalkAST(re->expr, preFunc, postFunc, data);
239  else if ((ptrderef = dynamic_cast<PtrDerefExpr *>(node)) != NULL)
240  ptrderef->expr = (Expr *)WalkAST(ptrderef->expr, preFunc, postFunc,
241  data);
242  else if ((refderef = dynamic_cast<RefDerefExpr *>(node)) != NULL)
243  refderef->expr = (Expr *)WalkAST(refderef->expr, preFunc, postFunc,
244  data);
245  else if ((soe = dynamic_cast<SizeOfExpr *>(node)) != NULL)
246  soe->expr = (Expr *)WalkAST(soe->expr, preFunc, postFunc, data);
247  else if ((aoe = dynamic_cast<AddressOfExpr *>(node)) != NULL)
248  aoe->expr = (Expr *)WalkAST(aoe->expr, preFunc, postFunc, data);
249  else if ((newe = dynamic_cast<NewExpr *>(node)) != NULL) {
250  newe->countExpr = (Expr *)WalkAST(newe->countExpr, preFunc,
251  postFunc, data);
252  newe->initExpr = (Expr *)WalkAST(newe->initExpr, preFunc,
253  postFunc, data);
254  }
255  else if (dynamic_cast<SymbolExpr *>(node) != NULL ||
256  dynamic_cast<ConstExpr *>(node) != NULL ||
257  dynamic_cast<FunctionSymbolExpr *>(node) != NULL ||
258  dynamic_cast<SyncExpr *>(node) != NULL ||
259  dynamic_cast<NullPointerExpr *>(node) != NULL) {
260  // nothing to do
261  }
262  else
263  FATAL("Unhandled expression type in WalkAST().");
264  }
265 
266  // Call the callback function
267  if (postFunc != NULL)
268  return postFunc(node, data);
269  else
270  return node;
271 }

330 static bool
331 lCostCallbackPre(ASTNode *node, void *d) {
332  CostData *data = (CostData *)d;
333  if (dynamic_cast<ForeachStmt *>(node) != NULL)
334  ++data->foreachDepth;
335  if (data->foreachDepth == 0)
336  data->cost += node->EstimateCost();
337  return true;

358 /** Given an AST node, check to see if it's safe if we happen to run the
359  code for that node with the execution mask all off.
360  */
361 static bool
362 lCheckAllOffSafety(ASTNode *node, void *data) {
363  bool *okPtr = (bool *)data;
364 
365  FunctionCallExpr *fce;
366  if ((fce = dynamic_cast<FunctionCallExpr *>(node)) != NULL) {
367  if (fce->func == NULL)
368  return false;
369 
370  const Type *type = fce->func->GetType();
371  const PointerType *pt = CastType<PointerType>(type);
372  if (pt != NULL)
373  type = pt->GetBaseType();
374  const FunctionType *ftype = CastType<FunctionType>(type);
375  Assert(ftype != NULL);
376 
377  if (ftype->isSafe == false) {
378  *okPtr = false;
379  return false;
380  }
381  }
382 
383  if (dynamic_cast<AssertStmt *>(node) != NULL) {
384  // While it's fine to run the assert for varying tests, it's not
385  // desirable to check an assert on a uniform variable if all of the
386  // lanes are off.
387  *okPtr = false;
388  return false;
389  }
390 
391  if (dynamic_cast<NewExpr *>(node) != NULL ||
392  dynamic_cast<DeleteStmt *>(node) != NULL) {
393  // We definitely don't want to run the uniform variants of these if
394  // the mask is all off. It's also worth skipping the overhead of
395  // executing the varying versions of them in the all-off mask case.
396  *okPtr = false;
397  return false;
398  }
399 
400  if (dynamic_cast<ForeachStmt *>(node) != NULL ||
401  dynamic_cast<ForeachActiveStmt *>(node) != NULL ||
402  dynamic_cast<ForeachUniqueStmt *>(node) != NULL ||
403  dynamic_cast<UnmaskedStmt *>(node) != NULL) {
404  // The various foreach statements also shouldn't be run with an
405  // all-off mask. Since they can re-establish an 'all on' mask,
406  // this would be pretty unintuitive. (More generally, it's
407  // possibly a little strange to allow foreach in the presence of
408  // any non-uniform control flow...)
409  //
410  // Similarly, the implementation of foreach_unique assumes as a
411  // precondition that the mask won't be all off going into it, so
412  // we'll enforce that here...
413  *okPtr = false;
414  return false;
415  }
416 
417  IndexExpr *ie;
418  if ((ie = dynamic_cast<IndexExpr *>(node)) != NULL && ie->baseExpr != NULL) {
419  const Type *type = ie->baseExpr->GetType();
420  if (type == NULL)
421  return true;
422  if (CastType<ReferenceType>(type) != NULL)
423  type = type->GetReferenceTarget();
424 
425  ConstExpr *ce = dynamic_cast<ConstExpr *>(ie->index);
426  if (ce == NULL) {
427  // indexing with a variable... -> not safe
428  *okPtr = false;
429  return false;
430  }
431 
432  const PointerType *pointerType = CastType<PointerType>(type);
433  if (pointerType != NULL) {
434  // pointer[index] -> can't be sure -> not safe
435  *okPtr = false;
436  return false;
437  }
438 
439  const SequentialType *seqType = CastType<SequentialType>(type);
440  Assert(seqType != NULL);
441  int nElements = seqType->GetElementCount();
442  if (nElements == 0) {
443  // Unsized array, so we can't be sure -> not safe
444  *okPtr = false;
445  return false;
446  }
447 
448  int32_t indices[ISPC_MAX_NVEC];
449  int count = ce->AsInt32(indices);
450  for (int i = 0; i < count; ++i) {
451  if (indices[i] < 0 || indices[i] >= nElements) {
452  // Index is out of bounds -> not safe
453  *okPtr = false;
454  return false;
455  }
456  }
457 
458  // All indices are in-bounds
459  return true;
460  }
461 
462  MemberExpr *me;
463  if ((me = dynamic_cast<MemberExpr *>(node)) != NULL &&
464  me->dereferenceExpr) {
465  *okPtr = false;
466  return false;
467  }
468 
469  if (dynamic_cast<PtrDerefExpr *>(node) != NULL) {
470  *okPtr = false;
471  return false;
472  }
473 
474  return true;
475 }
