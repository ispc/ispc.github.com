45 /** @brief Expr is the abstract base class that defines the interface that
46  all expression types must implement.
47  */
48 class Expr : public ASTNode {
49 public:
50  Expr(SourcePos p) : ASTNode(p) { }
51 
52  /** This is the main method for Expr implementations to implement. It
53  should call methods in the FunctionEmitContext to emit LLVM IR
54  instructions to the current basic block in order to generate an
55  llvm::Value that represents the expression's value. */
56  virtual llvm::Value *GetValue(FunctionEmitContext *ctx) const = 0;
57 
58  /** For expressions that can provide an lvalue (e.g. array indexing),
59  this function should emit IR that computes the expression's lvalue
60  and returns the corresponding llvm::Value. Expressions that can't
61  provide an lvalue should leave this unimplemented; the default
62  implementation returns NULL. */
63  virtual llvm::Value *GetLValue(FunctionEmitContext *ctx) const;
64 
65  /** Returns the Type of the expression. */
66  virtual const Type *GetType() const = 0;
67 
68  /** Returns the type of the value returned by GetLValueType(); this
69  should be a pointer type of some sort (uniform or varying). */
70  virtual const Type *GetLValueType() const;
71 
72  /** For expressions that have values based on a symbol (e.g. regular
73  symbol references, array indexing, etc.), this returns a pointer to
74  that symbol. */
75  virtual Symbol *GetBaseSymbol() const;
76 
77  /** If this is a constant expression that can be converted to a
78  constant of the given type, this method should return the
79  corresponding llvm::Constant value. Otherwise it should return
80  NULL. */
81  virtual llvm::Constant *GetConstant(const Type *type) const;
82 
83  /** This method should perform early optimizations of the expression
84  (constant folding, etc.) and return a pointer to the resulting
85  expression. If an error is encountered during optimization, NULL
86  should be returned. */
87  virtual Expr *Optimize() = 0;
88 
89  /** This method should perform type checking of the expression and
90  return a pointer to the resulting expression. If an error is
91  encountered, NULL should be returned. */
92  virtual Expr *TypeCheck() = 0;
93 
94  /** Prints the expression to standard output (used for debugging). */
95  virtual void Print() const = 0;
96 };
