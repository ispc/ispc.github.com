280 template <typename Predicate> void
281 SymbolTable::GetMatchingFunctions(Predicate pred,
282  std::vector<Symbol *> *matches) const {
283  // Iterate through all function symbols and apply the given predicate.
284  // If it returns true, add the Symbol * to the provided vector.
285  FunctionMapType::const_iterator iter;
286  for (iter = functions.begin(); iter != functions.end(); ++iter) {
287  const std::vector<Symbol *> &syms = iter->second;
288  for (unsigned int j = 0; j < syms.size(); ++j) {
289  if (pred(syms[j]))
290  matches->push_back(syms[j]);
291  }
292  }
293 }

296 template <typename Predicate> void
297 SymbolTable::GetMatchingVariables(Predicate pred,
298  std::vector<Symbol *> *matches) const {
299  for (unsigned int i = 0; i < variables.size(); ++i) {
300  SymbolMapType &sm = *(variables[i]);
301  SymbolMapType::const_iterator iter;
302  for (iter = sm.begin(); iter != sm.end(); ++iter) {
303  if (pred(iter->second))
304  matches->push_back(iter->second);
305  }
306  }
